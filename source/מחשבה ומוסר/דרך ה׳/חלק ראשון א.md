# דרך ה' חלק ראשון א

'''א. מְצִיאוּת ה':''' כָּל אִישׁ מִיִּשְׂרָאֵל צָרִיךְ שֶׁיַּאֲמִין וְיָדַע, שֶׁיֵּשׁ שָׁם מָצוּי רִאשׁוֹן קַדְמוֹן וְנִצְחִי, וְהוּא שֶׁהִמְצִיא וּמַמְצִיא כָּל מַה שֶּׁנִּמְצָא בִּמְצִיאוּת, וְהוּא הֶאָלֹק ב"ה.

**ב. שלימותו:** עוֹד צָרִיךְ שֶׁיֵּדַע, שֶׁהַמָּצוּי הַזֶּה ית"ש, אֵין אֲמִתַּת מְצִיאוּתוֹ מֻשֶּׂגֶת לְזוּלָתוֹ כְּלָל. וְרַק זֶה נוֹדַע בּוֹ, שֶׁהוּא מָצוּי שָׁלֵם בְּכָל מִינֵי שְׁלֵמוֹת, וְלֹא נִמְצָא בּוֹ חִסָּרוֹן כְּלָל. וְאוּלָם דְּבָרִים אֵלֶּה יִדַּעְנוּם בְּקַבָּלָה מִן הָאָבוֹת וּמִן הַנְּבִיאִים, וְהִשִּׂיגוּם כָּל יִשְׂרָאֵל בְּמַעֲמַד הַר סִינַי וְעָמְדוּ עַל אֲמִתָּתָם בְּבֵרוּר, וְלַמְּדוּם לִבְנֵיהֶם דּוֹר אַחַר דּוֹר כַּיּוֹם הַזֶּה, שֶׁכֵּן צִיּוּם מרע"ה מִפִּי הַגְּבוּרָה: "פֶּן תִּשְׁכַּח אֶת הַדְּבָרִים אֲשֶׁר רָאוּ עֵינֶיךָ" וְגוֹ' "וְהוֹדַעְתָּם לְבָנֶיךָ וְלִבְנֵי בָנֶיךָ". אָמְנָם גַּם מִצַּד הַחֲקִירָה בַּמּוֹפְתִים הַלִּמּוּדִיִּים יְאַמְּתוּ כָּל הָעִנְיָנִים הָאֵלֶּה, וְיֻכְרַח הֱיוֹתָם כֵּן מִכֹּחַ הַנִּמְצָאוֹת וּמַשִּׂיגֵיהֶם אֲשֶׁר אֲנַחְנוּ רוֹאִים בְּעֵינֵינוּ, ע"פ חָכְמַת הַטֶּבַע, הַהַנְדָּסָה, הַתְּכוּנָה וּשְׁאָר הַחֲכָמוֹת, שֶׁמֵּהֶם תַּלְקַחְנָה הַקְדָּמוֹת אֲמִתִּיּוּת אֲשֶׁר יִוָּלֵד מֵהֶן בֵּרוּר הָעִנְיָנִים הָאֲמִתִּיִּים הָאֵלֶּה. וְאָמְנָם לֹא נַאֲרִיךְ עַתָּה בָּזֶה, אֶלָּא נַצִּיעַ הַהַקְדָּמוֹת לַאֲמָתָם, וְנִסְדַּר הַדְּבָרִים עַל בּוּרִים, כְּפִי הַמָּסֹרֶת שֶׁבְּיָדֵינוּ וְהַמְּפֻרְסָם בְּכָל אֻמָּתֵנוּ:

**ג. הֶכְרֵחַ מְצִיאוּתוֹ:** עוֹד צָרִיךְ שֶׁיֵּדַע, שֶׁהַמָּצוּי הַזֶּה ית"ש, הִנֵּה מְצִיאוּתוֹ מְצִיאוּת מֻכְרָח שא"א הֶעְדֵּרוֹ כְּלָל:

**ד. הֱיוֹתוֹ בִּלְתִּי נִתְלֶה בְּאַחֵר:** עוֹד צָרִיךְ שֶׁיֵּדַע, שֶׁמְּצִיאוּתוֹ יָת' אֵינוֹ תָּלוּי בְּזוּלָתוֹ כְּלָל, אֶלָּא מֵעַצְמוֹ הוּא מֻכְרָח הַמְּצִיאוּת:

**ה. פְּשִׁיטוּתוֹ:** וְכֵן צָרִיךְ שֶׁיָּדַע, שֶׁמְּצִיאוּתוֹ יָת' - מְצִיאוּת פָּשׁוּט בְּלִי הַרְכָּבָה וְרִבּוּי כְּלָל, וְכָל הַשְּׁלֵמֻיּוֹת כֻּלָּם נִמְצָאִים בּוֹ בְּדֶרֶךְ פָּשׁוּט. פֵּי': כִּי הִנֵּה בַּנֶּפֶשׁ יִמְצְאוּ כֹּחוֹת רַבִּים שׁוֹנִים, שֶׁכָּל א' מֵהֶם גָּדְרוּ בִּפְנֵי עַצְמוֹ. ד"מ הַזִּכָּרוֹן כֹּחַ א', וְהָרָצוֹן כֹּחַ אַחֵר, וְהַדִּמְיוֹן כֹּחַ אַחֵר, וְאֵין א' מֵאֵלֶּה נִכְנָס בְּגֶדֶר חֲבֵרוֹ כְּלָל. כִּי הִנֵּה גֶּדֶר הַזִּכָּרוֹן - גָּדֵר א', וְגֶדֶר הָרָצוֹן - גָּדֵר אַחֵר, וְאֵין הָרָצוֹן נִכְנָס בְּגֶדֶר הַזִּכָּרוֹן, וְלֹא הַזִּכָּרוֹן בְּגֶדֶר הָרָצוֹן, וְכֵן כֻּלָּם. אַךְ הָאָדוֹן ית"ש אֵינֶנּוּ בַּעַל כֹּחוֹת שׁוֹנִים, אעפ"י שֶׁבֶּאֱמֶת יֵשׁ בּוֹ עִנְיָנִים שֶׁבָּנוּ הֵם שׁוֹנִים; כִּי הֲרֵי הוּא רוֹצֶה, וְהוּא חָכָם, וְהוּא יָכוֹל, וְהוּא שָׁלֵם בְּכָל שְׁלֵמוֹת. אָמְנָם אֲמִתַּת מְצִיאוּתוֹ הוּא עִנְיָן א' שֶׁכּוֹלֵל בַּאֲמִתָּתוֹ וְגָדְרוּ. (פֵּי' אֲמִתַּת עִנְיָנוֹ, כִּי אֵין שַׁיָּךְ גָּדֵר בּוֹ יָת', אֶלָּא עַל צַד הֶתֵּר לָשׁוֹן), כָּל מָה שֶׁהוּא שְׁלֵמוֹת. וְנִמְצָא שֶׁיֵּשׁ בּוֹ כָּל הַשְּׁלֵמֻיּוֹת, לֹא כְּדָבָר נוֹסָף עַל מַהוּתוֹ וַאֲמִתַּת עִנְיָנוֹ, אֶלָּא מִצַּד אֲמִתַּת עִנְיָנוֹ בְּעַצְמָהּ שֶׁכּוֹלֶלֶת בַּאֲמָתָהּ כָּל הַשְּׁלֵמֻיּוֹת, שא"א לְעִנְיָן הַהוּא מִבִּלְתִּי כָּל הַשְּׁלֵמֻיּוֹת מִצַּד עַצְמוֹ. וְהִנֵּה בֶּאֱמֶת הַדֶּרֶךְ הַזֶּה רָחוֹק מְאֹד מֵהַשָּׂגָתֵנוּ וְצִיּוּרֵנוּ, וְכִמְעַט שֶׁאֵין לָנוּ דֶּרֶךְ לְבֵאֲרוּ וּמִלּוֹת לְפָרְשׁוֹ. כִּי אֵין צִיּוּרֵנוּ וְדִמְיוֹנֵנוּ תּוֹפֵס אֶלָּא עִנְיָנִים מֻגְבָּלִים בִּגְבוּל הַטֶּבַע הַנִּבְרָא מִמֶּנּוּ יָת', שֶׁזֶּה מֶה שֶׁחוּשֵׁינוּ מַרְגִּישִׁים וּמְבִיאִים צִיּוּרוֹ אֶל הַשֵּׂכֶל; וּבִבְרוּאִים הִנֵּה הָעִנְיָנִים רַבִּים וְנִפְרָדִים. אוּלָם כְּבָר הִקְדַּמְנוּ, שֶׁאֲמִתַּת מְצִיאוּתוֹ יָת' אֵינָהּ מֻשֶּׂגֶת, וְאֵין לְהַקִּישׁ מִמָּה שֶׁרוֹאִים בַּבְּרוּאִים עַל הַבּוֹרֵא יִתְבָּרֵךְ, כִּי אֵין עִנְיָנָם וּמְצִיאוּתָם שָׁוֶה כְּלָל שֶׁנּוּכַל לְדִין מַזֶּה עַל זֶה. אֲבָל זֶה גַּם כֵּן מִן הַדְּבָרִים הַנּוֹדָעִים בַּקַּבָּלָה, כְּמוֹ שֶׁכָּתַבְנוּ, וּמְאֻמָּתִים בַּחֲקִירָה ע"פ הַטֶּבַע עַצְמוֹ בְּחֻקּוֹתָיו וּמִשְׁפָּטָיו, שא"א עכ"פ שֶׁלֹּא יִמְצָא מָצוּי אֶחָד מְשֻׁלָּל מִכָּל הַטֶּבַע חֻקּוֹת וּגְבוּלָיו, מִכָּל הָעֵדֶר וְחִסָּרוֹן, מִכָּל רִבּוּי וְהַרְכָּבָה, מִכָּל יַחַס וְעֵרֶךְ, וּמִכָּל מִקְּרֵי הַבְּרוּאִים, שֶׁיִּהְיֶה הוּא הַסִּבָּה הָאֲמִתִּית לְכָל הַנִּמְצָאוֹת וּלְכָל הַמִּתְיַלֵּד בָּם; כִּי זוּלַת זֶה, מְצִיאוֹת הַנִּמְצָאוֹת שֶׁאָנוּ רוֹאִים וְהַתְמָדָתָם הָיָה בִּלְתִּי אֶפְשָׁרִי:

**ו.** וּמִמָּה שֶׁצָּרִיךְ שֶׁיֵּדַע עוֹד, שֶׁהַמָּצוּי הַזֶּה ית"ש מֻכְרָח שֶׁיִּהְיֶה א' וְלֹא יוֹתֵר. פֵּי': שא"א שֶׁיִּמְצְאוּ מְצוּיִים רַבִּים שֶׁמְּצִיאוּתָם מֻכְרָח מֵעַצְמוֹ, אֶלָּא א' בִּלְבַד צָרִיךְ שֶׁיִּמָּצֵא בִּמְצִיאוּת הַמֻּכְרָח וְהַשְׁלֵם הַזֶּה. וְאִם שֶׁיִּמְצְאוּ נִמְצָאִים אֲחֵרִים, לֹא יִמְצְאוּ אֶלָּא מִפְּנֵי שֶׁהוּא יַמְצִיאֵם בִּרְצוֹנוֹ, וְנִמְצָאִים כֻּלָּם תְּלוּיִם בּוֹ וְלֹא מְצַוִּים מֵעַצְמָם:

**ז.** נִמְצָא כְּלָל הַיְּדִיעוֹת הַשָּׁרְשִׁיּוּת הָאֵלֶּה ו', וְהֵם: אֲמִתַּת מְצִיאוּתוֹ יָת', שֶׁלֵּמוּתוּ, הֶכְרֵחַ הִמָּצְאוּ, הֱיוֹתוֹ בִּלְתִּי נִתְלֶה בְּזוּלָתוֹ, פְּשִׁיטוּתוֹ, וְיִחוּדוֹ:
