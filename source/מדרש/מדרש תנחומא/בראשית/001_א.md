# מדרש תנחומא בראשית סימן א

, זה שאמר הכתוב וכשברא הקב"ה את עולמו, נתיעץ בתורה וברא את העולם, שנא' : לִי עֵצָה וְתוּשִׁיָּה אֲנִי בִינָה לִי גְבוּרָה, והתורה במה היתה כתובה? על גבי אש לבנה באש שחורה שנא' : קְוֻצּוֹתָיו תַּלְתַּלִּים שְׁחֹרוֹת כָּעוֹרֵב, מהו קוצותיו תלתלים? על כל קוץ וקוץ תילי תילים של הלכות. כיצד? כתוב בה : וְלֹא תְחַלְּלוּ אֶת שֵׁם קָדְשִׁי, אם אתה עושה חי"ת ה"א, אתה מחריב את העולם, כֹּל הַנְּשָׁמָה תְּהַלֵּל יָהּ , אם אתה עושה ה"א חי"ת, אתה מחריב את העולם. וכן : שְׁמַע יִשְׂרָאֵל ה' אֱלֹהֵינוּ ה' אֶחָד, אם אתה עושה דל"ת רי"ש, אתה מחריב את העולם, שנאמר : כִּי לֹא תִשְׁתַּחֲוֶה לְאֵל אַחֵר, כִּחֲשׁוּ בַּה' , אם אתה עושה בית כ"ף, תחריב את העולם, אֵין קָדוֹשׁ כַּה' , אם אתה עושה כ"ף בי"ת, תחריב את העולם, ואם אות אחת כך, כ"ש התיבה כולה, לכך נאמר: קוצותיו תלתלים, לפיכך דוד מקלס ואומר : רְחָבָה מִצְוָתְךָ מְאֹד, ואומר : אֲרֻכָּה מֵאֶרֶץ מִדָּהּ וגו', והיא היתה אומנת לכל מעשה בראשית, שנאמר : וָאֶהְיֶה אֶצְלוֹ אָמוֹן, אל תקרי אמון אלא אומן, ובה נטה שמים ויסד ארץ, שנא' : אִם לֹא בְרִיתִי יוֹמָם וָלָיְלָה וגו', ובה חתם ים אוקיינוס, שלא יצא וישטף את העולם, שנאמר : הַאוֹתִי לֹא תִירָאוּ נְאֻם ה' אִם מִפָּנַי לֹא תָחִילוּ וגו', ובה חתם את התהום, שלא יציף את העולם, שנא' : בְּחֻקוֹ חוּג עַל פְּנֵי תְהוֹם, ובה ברא חמה ולבנה, שנאמר : כֹּה אָמַר ה' נֹתֵן שֶׁמֶשׁ לְאוֹר יוֹמָם חֻקֹּת יָרֵחַ וְכוֹכָבִים לְאוֹר לָיְלָה רֹגַע הַיָּם וַיֶּהֱמוּ גַלָּיו ה' צְבָאוֹת שְׁמוֹ, הא למדת שהעולם לא נתיסד אלא על התורה, והקדוש ברוך הוא נתנה לישראל, שיתעסקו בה ובמצותיה יומם ולילה, שנא' : וְהָגִיתָ בּוֹ יוֹמָם וָלַיְלָה, ואומר : כִּי אִם בְּתוֹרַת ה' חֶפְצוֹ וגו', וְהָיָה כְּעֵץ שָׁתוּל עַל פַּלְגֵי מָיִם וגו', שבשביל שומרי התורה, העולם עומד, שכן אמרה חנה : כִּי לַה' מְצֻקֵי אֶרֶץ, ומי הם מצוקי ארץ? אלו שומרי התורה, שבזכותם הושתת התורה, שנאמר: וישת עליהם תבל, ותניא: אר"ש בן לקיש: למה נאמר במעשה בראשית יום אחד, יום שני, יום שלישי, יום רביעי, יום חמישי, יום הששי, ה"א יתירה למה, שהרי בכולן לא נאמר אלא יום אחד, יום שני, וכן לכולם? מלמד: שהתנה הקב"ה עם מעשה בראשית, ואמר להם: אם ישראל מקבלים התורה, שיש בה ה' ספרים, מוטב, ואם לאו, אני מחזיר אתכם לתוהו ובוהו, וכן בני קרח אמרו : נְמֹגִים אֶרֶץ וְכָל יֹשְׁבֶיהָ אָנֹכִי תִכַּנְתִּי וגו', זו התורה, שפתח הקדוש ברוך הוא, וזכה משה רבינו וקבלה, לפי שהתורה סוליתה ענוה וכתרה יראה, סוליתה ענוה שנאמר : עֵקֶב עֲנָוָה יִרְאַת ה', וכתרה יראה שנא' : רֵאשִׁית חָכְמָה יִרְאַת ה', ושניהם במשה רבינו שנא' : וְהָאִישׁ מֹשֶׁה עָנָו מְאֹד, יראה, דכתיב : כִּי יָרֵא מֵהַבִּיט אֶל הָאֱלֹהִים, ואמרו רבותינו: בשכר שלשה זכה לשלשה, בשכר ויסתר משה פניו, זכה לקלסתר פנים, בשכר כי ירא, וייראו מגשת אליו , בשכר מהביט, זכה וּתְמֻנַת ה' יַבִּיט , ואין מתן שכרה של תורה בעולם הזה אלא לעולם הבא, שנאמר : אֲשֶׁר אָנֹכִי מְצַוְּךָ הַיּוֹם לַעֲשׂוֹתָם, ולא לעולם הבא, היום לעשותם, ולא היום לטול שכרם, וכן אמר שלמה : עֹז וְהָדָר לְבוּשָׁהּ וַתִּשְׂחַק לְיוֹם אַחֲרוֹן, בוא ולמד מפרעה: על שאמר ליוסף אני פרעה נגדל מאד שנאמר : ויסר פרעה וגו', הקב"ה על כל מצוה אני ה' על אחת כמה וכמה מה מזה נלמוד שאין קצבה על מתן שכרה, צפה דוד ואמר : מָה רַב טוּבְךָ אֲשֶׁר צָפַנְתָּ לִּירֵאֶיךָ וגו', בכל מעשה בראשית לא כתיב פעולה, ובמתן שכרה כתיב פעולה, דכתיב : פעלת לחוסים בך, אתה מוצא שהקדוש ברוך הוא מראה בשעת פטירתן של עוסקי תורה מתן שכרן, מעשה בר' אבהו: כשהיה מסתלק מן העולם, הראה לו הקב"ה שלש עשרה נהרי אפרסמון, התחיל לומר בשעת מיתה לתלמידיו: אשריכם עוסקי התורה, אמרו לו: רבינו! מה ראית? אמר להם: שלש עשרה נהרי אפרסמון נתן לי הקדוש ברוך הוא בשכר תורתי התחיל לומר וַאֲנִי אָמַרְתִּי לְרִיק יָגַעְתִּי לְתֹהוּ וְהֶבֶל כֹּחִי כִלֵּיתִי אָכֵן מִשְׁפָּטִי אֶת ה' וּפְעֻלָּתִי אֶת אֱלֹהָי , וכן ישעיה אמר : אַשְׁרֵיכֶם זֹרְעֵי עַל כָּל מָיִם, אלו עוסקי תורה שנמשלה למים שנא' : הוֹי כָּל צָמֵא לְכוּ לַמַּיִם, משלחי רגל השור, זה משיח בן יוסף, שנמשל לשור, וחמור, זה משיח בן דוד, שנא' : עָנִי וְרֹכֵב עַל חֲמוֹר, כשיבואו על אותה שעה הוא אומר : וּמֵעוֹלָם לֹא שָׁמְעוּ ולֹא הֶאֱזִינוּ, עַיִן לֹא רָאָתָה אֱלֹהִים זוּלָתְךָ וגו', ואומר : אַשְׁרֵי תְמִימֵי דָרֶךְ הַהֹלְכִים בְּתוֹרַת ה', כלומר אשרי מכבדי בעלי תורה, ואומר : עֵץ חַיִּים הִיא לַמַּחֲזִיקִים בָּהּ וְתֹמְכֶיהָ מְאֻשָּׁר, וכך משה אמר : כִּי אִם שָׁמֹר תִּשְׁמְרוּן, אם שמרת בני תורה, תשמרון, וכן הוא אומר : כִּי מְכַבְּדַי אֲכַבֵּד וּבֹזַי יֵקָלּוּ, זה המכבד בני תורה, ותניא: את ה' אלהיך תירא, "את" לרבות בני תורה, לפי שאין מדה אחרת כיוצא בה, שנא' : יְקָרָה הִיא מִפְּנִינִים, יקרה היא מכהן גדול המשמש לפני ולפנים, והתורה צווחת : אַשְׁרֵי אָדָם שֹׁמֵעַ לִי וגו', כל השומע אל התורה, אינו ניזק, שנא' : וְשֹׁמֵעַ לִי יִשְׁכָּן בֶּטַח וְשַׁאֲנַן וגו', ואומר : בְּהִתְהַלֶּכְךָ תַּנְחֶה אֹתָךְ וגו':
