# טור אבן העזר סימן הקדמה

## הַקְדָּמַת רַבֵּנוּ יַעֲקֹב בַּעַל הַטּוּרִים לְחֵלֶק אֶבֶן הָעֵזֶר

יְהִי שָׁם ה' מְבֹרָךְ מֵעַתָּה וְעַד עוֹלָם. מְרוֹמָם עַל כָּל, וּכְבוֹדוֹ מֵעֵין כָּל נִסְתָּר וְנֶעֱלָם. וְאַף כִּי כְּבוֹדוֹ מָלֵא עוֹלָם. אֵיזֶה עַיִן תְּשׁוּרֵהוּ, וּלְשׁוֹן מִי תְּכִילֵהוּ, וְלֵב מִי יִמְצָאֵהוּ, וְאֵיזֶה שֵׂכֶל אֲשֶׁר יֵדָעֵהוּ, וּתְבוּנוֹת מִי תְּבִינֵהוּ, וְחָכְמַת מִי תַּשִּׂיגֵהוּ, עָמֹק עָמֹק מִי יַעֲלֵהוּ, גָּבֹהַּ גָּבֹהַּ מִי יוֹרִידֵהוּ, רָחוֹק מִי יְקָרְבֵהוּ, קָרוֹב מִי יְרַחֲקֵהוּ.

אֲבָל כֹּחַ מַעֲשָׂיו יַגִּידוּהוּ, עֲלִילוֹתָיו יוֹדִיעוּהוּ, וְנִפְלְאוֹתָיו יַשִּׂיגוּהוּ, נוֹרְאוֹתָיו יְסַהֲדוּהוּ, גֹּדֶל מִפְעָלָיו יְעִידוּהוּ, כִּי אֱלֹהֵי הָאֱלֹהִים וַאֲדוֹנֵי הָאֲדֹנִים, הוּא לְבַדּוֹ וְאֵין עוֹד כָּמוֹהוּ, הַשָּׁמַיִם וְכָל צְבָאָם כְּבוֹדוֹ יְסַפְּרוּהוּ, חַיּוֹת וְאוֹפַנִּים יְנַשְּׂאוּהוּ, שְׂרָפִים וְאֶרְאֶלִּים יַעֲרִיצוּהוּ, כְּרוּבִים וּמַלְאָכִים יְעַצְּמוּהוּ, עִירִין וְקַדִּישִׁין יַקְדִּישׁוּהוּ, שֶׁמֶשׁ וְיָרֵחַ יְצַלְצְלוּהוּ, הַכּוֹכָבִים מִמְּסִלּוֹתָם יְסַלְסְלוּהוּ, תֵּבֵל וּמְלוֹאָהּ תְּהִלּוֹתָיו יְהַלְּלוּהוּ, בְּנֵי יַעֲקֹב יִבְחָרוּהוּ, עֶרֶב וָבֹקֶר יְיַחֲדוּהוּ, עֲדַת לְאֻמִּים יַמְלִיכוּהוּ, קְהַל עָם יְרוֹמְמוּהוּ, מוֹשַׁב זְקֵנִים יַאֲמִירוּהוּ, יְשִׁישִׁים יַזְמִירוּהוּ, נְעָרִים יְגַדְּלוּהוּ, עוֹלָלִים יְקַלְּסוּהוּ, גְּמוּלֵי חָלָב יְפָאֲרוּהוּ, יוֹנְקֵי שָׁדַיִם יְבָרְכוּהוּ, גִּבּוֹרִים יַגְבִּירוּהוּ, חַלָּשִׁים יְאַמְּצוּהוּ, יַמִּים וְכָל רֹמֵשׂ בָּם יְשַׁבְּחוּהוּ, עֲצֵי הַיַּעַר יְרַנְּנוּהוּ, חַיּוֹת הַשָּׂדֶה יְכַבְּדוּהוּ, כָּל עוֹף כָּנָף יְשַׂגְּבוּהוּ, עִשְּׂבוֹת הָרִים יַאֲדִירוּהוּ, תְּנוּבוֹת שָׂדַי יְחַזְּקוּהוּ, דְּגֵי הַיָּם יַעֲלוּהוּ, אֵילֵי נְבָיוֹת יְשָׁרְתוּהוּ, צֹאן קֵדָר יַמְשִׁילוּהוּ, בְּהֵמוֹת בְּהַרְרֵי אֶלֶף יְפַחֲדוּהוּ, זִיז שָׂדַי יְהַדְּרוּהוּ.

כִּי שֵׁשֶׁת יָמִים עָשָׂה אֶת הַשָּׁמַיִם וְאֶת הָאָרֶץ הַיַּמִּים וְכָל אֲשֶׁר בָּהֶם, הַכֹּל לִכְבוֹדוֹ בְּרָאוֹ, וּבְאַחֲרִית כָּל בָּרָא הָאָדָם יָפֶה בְּעִתּוֹ, יְצָרוֹ אַף עֲשָׂאוֹ, כְּחָתָן שֶׁעוֹשֶׂה סְעוּדָה וְאַחַר כָּךְ יָבִיא קְרוּאָיו, כִּי אַחֲרֵי שֶׁבָּרָא הַשָּׁמַיִם וּכְסִילֵיהֶם, הָאָרֶץ וְתוֹלְדוֹתֶיהָ, הַיַּמִּים וְכָל אֲשֶׁר בָּהֶם, בָּרָא אֶת הָאָדָם וְהִמְשִׁילוֹ עַל כָּל הַתַּחְתּוֹנִים, כְּדִכְתִיב (תְּהִלִּים ח,ז): "תַּמְשִׁילֵהוּ בְּמַעֲשֵׂה יָדֶיךָ כָּל שַׁתָּה תַחַת רַגְלָיו". גַּם אֶת הָעֶלְיוֹנִים נָתַן לְשַׁמְּשׁוֹ, כְּדִכְתִיב (בְּרֵאשִׁית א,טו): "וְהָיוּ לִמְאוֹרֹת בִּרְקִיעַ הַשָּׁמַיִם לְהָאִיר עַל הָאָרֶץ".

וְעוֹד הִפְלִיא וְהִגְדִּיל עִמּוֹ בְּעֵזֶר אֲשֶׁר עֲזָרוֹ, וּבָעֲטָרָה אֲשֶׁר עִטְּרוֹ, וּכְחָתָן פֵּאֲרוֹ, וַיָּשֶׂם עָלָיו צִיץ נִזְרוֹ, וְעַל הַכֹּל הָאִשָּׁה אֲשֶׁר מִמֶּנּוּ לֻקֳחָה, וְלֹא כִּשְׁאַר הַבְּרוּאִים אֲשֶׁר בְּרָאָם שְׁנַיִם שְׁנַיִם, אֲבָל בְּרָאוֹ לְבַדּוֹ, וַיִּקַּח אַחַת מִצַּלְעוֹתָיו וַיִּבֶן אוֹתָהּ לְאִשָּׁה, וַיְבִיאֶהָ לוֹ לְמַעַן הֱיוֹתָהּ לוֹ לְעֵזֶר וּלְהוֹעִיל.

כַּאֲשֶׁר כָּתַב הֶחָכָם רַבִּי אַבְרָהָם בַּ"ר דָּוִד בְּסֵפֶר הַנֶּפֶשׁ - כִּי לְטוֹבַת הָאָדָם וְלַהֲנָאָתוֹ הַגְּדוֹלָה עָשָׂה זֶה, שֶׁאִלּוּ נִבְרָא זָכָר וּנְקֵבָה כִּשְׁאַר הַבְּרוּאִים, הָיְתָה הָאִשָּׁה אֵצֶל הָאָדָם כִּנְקֵבוֹת הַבְּהֵמָה אֵצֶל הַזָּכָר, שֶׁאֵינָהּ מְקַבֶּלֶת שִׁלְטוֹנוֹת הַזָּכָר, וְלֹא עוֹמֶדֶת אֶצְלוֹ לְשַׁמְּשׁוֹ, אַךְ זֶה חוֹטֵף מִזֶּה וְזֶה בּוֹעֵט בָּזֶה, וְאִישׁ לְמַעֲבָרוֹ יִפְנוּ. וְגַם אֵינָם מְיֻחָדִים זֶה לָזֶה, לְפִי שֶׁבַּתְּחִלָּה נִבְרְאוּ זֶה בִּפְנֵי עַצְמוֹ וְזֶה בִּפְנֵי עַצְמוֹ, וְלָכֵן רָאָה הַבּוֹרֵא יִתְבָּרַךְ צֹרֶךְ הָאָדָם וַהֲנָאוֹתָיו וּבָרָא אוֹתוֹ יְחִידִי, וְלָקַח אַחַת מִצַּלְעוֹתָיו וּבָנָה הָאִשָּׁה מִמֶּנָּה, וֶהֱבִיאָהּ אֶל הָאָדָם לִהְיוֹת לוֹ לְאִשָּׁה, וְלִהְיוֹתָהּ לוֹ לְעֵזֶר וּלְמִסְעָד, מִפְּנֵי שֶׁהִיא נֶחְשֶׁבֶת לוֹ כְּאֶחָד מֵאֲבָרָיו אֲשֶׁר לוֹ לְשַׁמְּשׁוֹ, וְכִי יִהְיֶה מוֹשֵׁל עָלֶיהָ כְּמוֹשְׁלוֹ עַל אֲבָרָיו, וְכִי תִּהְיֶה מִשְׁתּוֹקֶקֶת אֵלָיו כַּאֲשֶׁר אֲבָרָיו מִשְׁתּוֹקְקִים לַהֲנָאַת גּוּפוֹ. וְזֶהוּ מַה שֶׁאָמַר הַכָּתוּב (בְּרֵאשִׁית ב,כ): "וּלְאָדָם לֹא מָצָא עֵזֶר כְּנֶגְדּוֹ", מְצִיאָה זוֹ אֵינָהּ אַחַר בְּדִיקָה וְחִפּוּשׂ כִּשְׁאַר כָּל הַמְּצִיאוֹת, שֶׁאֵין נָכוֹן לְדַבֵּר כֵּן כְּנֶגֶד הַבּוֹרֵא יִתְבָּרַךְ, אַךְ הִיא בִּמְצִיאוּת הַמַּחְשָׁבָה הַקַּדְמוֹנִית, כַּאֲשֶׁר בָּרָא אֶת הַנִּבְרָאִים מִן הָאֲדָמָה זָכָר וּנְקֵבָה, וְהִבִּיט וְרָאָה בְּטוֹבַת הָאָדָם וְלֹא מָצָא עֵזֶר לָאָדָם בַּבְּרִיאָה הַהִיא, וְאָמַר (בְּרֵאשִׁית ב,יד): "לֹא טוֹב הֱיוֹת הָאָדָם לְבַדּוֹ", כְּלוֹמַר אֵין טוֹב שֶׁיִּהְיֶה הָאָדָם מִתְבּוֹדֵד עִם הָאִשָּׁה, כְּמוֹ הַבְּהֵמוֹת שֶׁהֵם מִתְבּוֹדְדִים וְאֵין הַנְּקֵבוֹת מְיֻחָדוֹת לַזְּכָרִים. "אֶעֱשֶׂה לוֹ עֵזֶר מִמֶּנּוּ כְּנֶגְדּוֹ" – כְּלוֹמַר אֶבְרָאֶנּוּ בְּעִנְיָן שֶׁיִּהְיֶה לוֹ עֵזֶר מִמֶּנּוּ וְשֶׁתִּהְיֶה מְיֻחֶדֶת לוֹ, וְזֶהוּ 'כְּנֶגְדּוֹ' – כְּלוֹמַר שֶׁתַּעֲמֹד אֶצְלוֹ יוֹם וָלַיְלָה. עַד כָּאן לְשׁוֹן הָרַב.

וְעַל כֵּן הִפְלִיגוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה מְאֹד לְדַבֵּר בְּמִי שֶׁאֵין לוֹ אִשָּׁה, עַד שֶׁאָמְרוּ (יְבָמוֹת סב,ב): 'כָּל מִי שֶׁאֵין לוֹ אִשָּׁה שָׁרוּי בְּלֹא טוֹבָה, בְּלֹא שִׂמְחָה, בְּלֹא בְּרָכָה, בְּלֹא דִּירָה, בְּלֹא תּוֹרָה, בְּלֹא חוֹמָה, בְּלֹא שָׁלוֹם. בְּלֹא טוֹבָה - דִּכְתִיב (בְּרֵאשִׁית ב,יח): "לֹא טוֹב הֱיוֹת הָאָדָם לְבַדּוֹ". בְּלֹא שִׂמְחָה - דִּכְתִיב (דְּבָרִים י,כו): "וְשָׂמַחְתָּ בְּחַגֶּךָ אַתָּה וּבֵיתֶךָ". בְּלֹא בְּרָכָה – דִּכְתִיב (יְחֶזְקֵאל מד,ל): "לְהָנִיחַ בְּרָכָה אֶל בֵּיתֶךָ". בְּלֹא דִּירָה – דִּכְתִיב (בְּרֵאשִׁית שָׁם): "אֶעֱשֶׂה לּוֹ עֵזֶר כְּנֶגְדּוֹ". בְּלֹא תּוֹרָה – דִּכְתִיב (אִיּוֹב ו,יג): "הַאִם אֵין עֶזְרָתִי בִי וְתֻשִׁיָּה נִדְּחָה מִמֶּנִּי". בְּלֹא חוֹמָה – דִּכְתִיב (יִרְמְיָה לא,כא): "נְקֵבָה תְּסוֹבֵב גָּבֶר". בְּלֹא שָׁלוֹם – דִּכְתִיב (אִיּוֹב ה,כד): "וְיָדַעְתָּ כִּי שָׁלוֹם אָהֳלֶךָ וּפָקַדְתָּ נָוְךָ וְלֹא תֶחֱטָא".

וְהָאָדָם בִּרְאוֹתוֹ אוֹתָהּ וְיָדַע כִּי מִמֶּנּוּ לֻקְּחָה, אָמַר (בְּרֵאשִׁית ב,כז) – "עַל כֵּן יַעֲזָב אִישׁ אֶת אָבִיו וְאֶת אִמּוֹ וְדָבַק בְּאִשְׁתּוֹ וְהָיוּ לְבָשָׂר אֶחָד", כְּלוֹמַר רְאוּיָה זוֹ שֶׁתִּהְיֶה מְיֻחֶדֶת לָאָדָם וְאָדָם אֵלֶיהָ.

וְעַל כֵּן רָאוּי לָאָדָם לֶאֱהֹב אֶת אִשְׁתּוֹ כְּגוּפוֹ, וּלְכַבְּדָהּ יוֹתֵר מִגּוּפוֹ (יְבָמוֹת סב,ב), וּלְרַחֵם עָלֶיהָ וּלְשָׁמְרָהּ, כַּאֲשֶׁר יִשְׁמֹר אֶחָד מֵאֲבָרָיו. וְכֵן הִיא חַיֶּבֶת לְעוֹבְדוֹ וּלְאוֹהֲבוֹ כְּנַפְשָׁהּ, כִּי מִמֶּנּוּ לֻקְּחָה.

וְעַל כֵּן צִוָּה הַכָּתוּב עַל הָאָדָם "שְׁאֵרָהּ כְּסוּתָהּ וְעֹנָתָהּ לֹא יִגְרָע" (שְׁמוֹת כא,י), וְעוֹד מִצְוֹת רַבּוֹת נִצְטַוּוּ בְּעִנְיָן חִבּוּר הָאָדָם עִם אִשְׁתּוֹ, כָּל יְמֵי הֱיוֹתָם יַחַד עַד הִפַּרְדָם, וְעוֹד תַּקָּנוֹת רַבּוֹת נִתְקְנוּ בָּהֶן אַחֲרֵי כֵן. וְאָמַרְתִּי לְחַבֵּר בָּהֶן סֵפֶר, וְרָאִיתִי בַּסְּפָרִים, וּבִינוֹתִי בְּדִבְרֵי הַמְּחַבְּרִים, וְלִקַּטְתִּי בָּעֳמָרִים מֵאַחֲרֵי הַקּוֹצְרִים, וְחִבַּרְתִּי אֶת הַסֵּפֶר הַזֶּה, וּקְרָאתִיו אֶבֶן הָעֵזֶר.

וְזֶה סִדּוּרוֹ – תְּחִלָּה אֲבָאֵר חִיּוּב הָאָדָם לִשָּׂא וְגֹדֶל שְׂכָרוֹ, אֶת מִי רָאוּי שֶׁיִּשָּׂא, וְאֵיזוֹ חֲזָקָה מוֹעֶלֶת לְעִנְיַן כְּהֻנָּה, וְשֶׁאֵינָן רְאוּיוֹת לִנָּשֵׁא לְיִשְׂרָאֵל וּלְכֹהֵן אוֹ מֵחֲמַת פְּסוּל שֶׁבְּגוּפָן, אוֹ חִסָּרוֹן שֶׁבְּגוּפָן, אוֹ מֵחֲמַת מִקְרֶה שֶׁאֵרַע בָּהֶן, אוֹ מֵחֲמַת אִסּוּר עֶרְוָה, וְאִסּוּר גּוֹיָה וְשִׁפְחָה, וְהֶתֵּר אֵשֶׁת אִישׁ בְּעֵדוּת שֶׁמֵּת בַּעֲלָהּ, וּבְאֵיזֶה בִּיאָה מִתְחַיֵּב הָאִישׁ וְהָאִשָּׁה, וְהַרְחָקַת מִן הָעֲרָיוֹת, וְאִסּוּר יִחוּד, וְאִסּוּר הוֹצָאַת זֶרַע לְבַטָּלָה, וְכֵיצַד יִתְנַהֵג עִם אִשְׁתּוֹ וְחִיּוּב עוֹנָתוֹ עִמָּהּ.

וְהִלְכוֹת קִדּוּשִׁין, וְכָל דִּין הָאֲרוּסָה כָּל מַה שֶׁנּוֹהֵג בָּהּ עַד שֶׁתִּנָּשֵׂא, וְדִין הַסִּבְלוֹנוֹת וְהַחֻפָּה, וּבִרְכַּת נִשּׂוּאִין, וּמִצְוַת שִׂמְחַת חָתָן וְכַלָּה, וְשִׁעוּר הַכְּתוּבָה לְכָל אַחַת וְאַחַת, וְטַעֲנַת בְּתוּלִים, וְחִיּוּב הָאִישׁ בִּכְתֻבָּה וְכָל תְּנָאֶיהָ, וְהֵיאַךְ נִגְבֵּית, וּמַה שֶׁהוּא זוֹכֶה מִכֹּחַ אִשְׁתּוֹ.

וְהִלְכוֹת גֵּרוּשִׁין וּמֵאוּן, וְהִלְכוֹת יִבּוּם וַחֲלִיצָה, וְהִלְכוֹת אֹנֶס וּמְפַתֶּה וּמוֹצִיא שֵׁם רַע, וְהִלְכוֹת סוֹטָה.

וְרָאִיתִי עוֹד לַעֲשׂוֹת סִימָנִים לִכְלֹל כָּל עִנְיָן וְעִנְיָן בְּמִלּוֹת קְצָרוֹת, וּלְכוֹתְבָם בִּתְחִלַּת הַסֵּפֶר בְּמִנְיָן, לְמַעַן הֱיוֹת נָקֵל לִמְצֹא כָּל עִנְיָן וְעִנְיָן.

וְאֵל אֱלֹהִים יֹדֵעַ וְיִשְׂרָאֵל הוּא יֵדַע כִּי לֹא בַּעֲבוּר מַעֲלָה עָשִׂיתִי הַדָּבָר הַזֶּה, וְגַם לֹא עָשִׂיתִי דָּבָר שֶׁיִּהְיֶה כְּדֵי עִלּוּי, שֶׁאֵינִי אֶלָּא כְּתַלְמִיד הַלּוֹמֵד מִפִּי הַמְּלַמֵּד וְכוֹתֵב דְּבָרָיו כְּדֵי לְהִתְלַמֵּד, וּכְלוֹקֵט הַמְּלַקֵּט מֵאַחֲרֵי הַקּוֹצְרִים, וְכִמְאַסֵּף מִבֵּין הָעֳמָרִים אַחַת הֵנָּה וְאַחַת הֵנָּה בֵּינִי חִטֵּי, לִקּוּטֵי בָּתַר לִקּוּטֵי.

וַאֲבַקֵּשׁ מֵאַל הָעוֹזֵר לִהְיוֹת לִי לְעֵזֶר, כְּדִכְתִיב (תְּהִלִּים קכא,ב): "עֶזְרִי מֵעִם ה' עוֹשֶׂה שָׁמַיִם וָאָרֶץ", וּכְתִיב (שָׁם קטו,יא): "יִרְאֵי ה' בִּטְחוּ בַה' עֶזְרָם וּמָגִנָּם הוּא", וּכְתִיב (שָׁם כח,ז): "ה' עֻזִּי וּמָעֻזִּי בּוֹ בָטַח לִבִּי וְנֶעֱזָרְתִּי וַיַּעֲלֹז לִבִּי וּמִשִּׁירִי אֲהוֹדֶנּוּ", וּכְתִיב (שָׁם מ,יח): "וַאֲנִי עָנִי וְאֶבְיוֹן ה' יַחֲשֹׁב לִי עֶזְרָתִי וּמְפַלְטִי אַתָּה אֱלֹהַי אַל תְּאַחַר", וּכְתִיב (שָׁם קיח,ז): "ה' לִי בְּעֹזְרָי וַאֲנִי אֶרְאֶה בְשֹׂנְאָי", וּכְתִיב (שָׁם קִי,יג): "דַּחֹה דְחִיתַנִי לִנְפֹּל וַה' עֲזָרָנִי", וּכְתִיב (שָׁם מַד,כז): "קוּמָה ה' עֶזְרָתָה לָּנוּ וּפְדֵנוּ לְמַעַן חַסְדֶּךָ", וּכְתִיב (שָׁם ס,יג): "הָבָה לָּנוּ עֶזְרָת מִצָּר וְשָׁוְא תְּשׁוּעַת אָדָם", וּכְתִיב (שְׁמוּאֵל א ז,יב): "וַיִּקְרָא שְׁמוֹ אֶבֶן הָעֵזֶר לֵאמֹר עַד הֵנָּה עֲזָרָנִי ה'"
