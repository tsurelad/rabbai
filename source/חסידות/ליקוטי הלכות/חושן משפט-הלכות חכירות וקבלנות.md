# ליקוטי הלכות/חושן משפט/הלכות חכירות וקבלנות

# חכירות וקבלנות א

הַחוֹכֵר שֶׁנּוֹתֵן מִכָּל הַשָּדֶה דָּבָר קָצוּב כָּךְ וְכָךְ כּוּרִין לְשָׁנָה, הַמְקַבֵּל לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ:

## אות א

יֵשׁ לְרַמֵּז בָּזֶה עַל פִּי מַה שֶּׁכָּתַב רַבֵּנוּ בַּמַּאֲמָר וַיֹּאמֶר בֹּעַז אֶל רוּת וְכוּ' סִימָן סה שֶׁיֵּשׁ שָֹדֶה וְשָׁם גְּדֵלִים אִילָנוֹת וַעֲשָֹבִים הַיְנוּ נְשָׁמוֹת. וְהֵם צְרִיכִים לְבַעַל הַשָּדֶה שֶׁעוֹסֵק בְּתִקּוּנָם בְּכָל תִּקּוּנֵי הַשָּדֶה עַיֵּן שָׁם הֵיטֵב. וְהָעִקָּר שֶׁמֵּבִיא אוֹתָם אֶל הַתַּכְלִית שֶׁהִוא כֻּלּוֹ טוֹב שֶׁעַל יְדֵי זֶה יְכוֹלִים לְהִתְפַּלֵּל עַיֵּן שָׁם הֵיטֵב:

וְזֶה הַצַּדִּיק הַדּוֹר הָעוֹסֵק בְּתִקּוּנֵי הַשָּדֶה הַנַּ"ל הוּא בְּחִינַת מְקַבֵּל וְאָרִיס שֶׁיּוֹרֵד לְתוֹךְ שָֹדֶה שֶׁל חֲבֵרוֹ כִּי ה' יִתְבָּרַךְ כִּבְיָכוֹל נִקְרָא חֲבֵרוֹ שֶׁל הַצַּדִּיק הָאֱמֶת כְּמוֹ שֶׁכָּתוּב אִכְלוּ רֵעִים לְמַעַן אַחַי וְרֵעָי וְכוּ' כִּי הַצַּדִּיק הָאֱמֶת שֶׁעוֹסֵק בְּתִקּוּנֵי הַשָּדֶה שֶׁזֶּהוּ תִּקּוּן כָּל הָעוֹלָמוֹת הוּא נַעֲשֶֹה שֻׁתָּף לְהַקָּדוֹשׁ בָּרוּךְ הוּא בְּמַעֲשֵֹה בְּרֵאשִׁית וְעַל כֵּן נִקְרָא בְּחִינַת מְקַבֵּל שָֹדֶה מֵחֲבֵרוֹ, שָֹדֶה הַנַּ"ל וְעוֹסֵק בְּכָל תִּקּוּנִים שֶׁל הַשָּדֶה הַנַּ"ל בַּחֲרִישָׁה וּקְצִירָה זְרִיעָה וּנְטִיעָה וּלְנַכֵּשׁ הָעֲשָֹבִים כִּי כָּל אֵלּוּ הַבְּחִינוֹת נִמְצָאִים בִּבְחִינַת הַשָּדֶה שֶׁל הַנְּשָׁמוֹת הַנַּ"ל, וְלִפְעָמִים מְקַבְּלוֹ בִּפְשִׁיטוּת שֶׁנּוֹתֵן דֶּרֶךְ כְּלָל כָּךְ וְכָךְ כּוּרִין לְשָׁנָה. וְלִפְעָמִים לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ, זֶה בְּחִינַת שִׁיר פָּשׁוּט כָּפוּל מְשֻׁלָּשׁ מְרֻבָּע בְּחִינַת עֲשָֹרָה מִינֵי נְגִינָה כִּי זֶה הַצַּדִּיק בַּעַל הַשָּדֶה הַנַּ"ל עוֹשֶֹה נִגּוּנִים. וְזֶה כְּלַל וְעִקַּר תִּקּוּנֵי הַשָּדֶה בְּחִינַת שְׁלֵמוּת הַתְּפִלָּה כִּי הוּא צָרִיךְ לְתַקְּנָם וּלְהַעֲלוֹתָם וּלְבָרְרָם מִמָּקוֹם שֶׁהֵם לַהֲבִיאָם אֶל הַתַּכְלִית הַטּוֹב. וְזֶה בְּחִינַת נִגּוּנִים כַּמּוּבָא בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה (סִימָן רפב) עַל פָּסוּק וְעוֹד מְעַט וְאֵין רָשָׁע וְהִתְבּוֹנַנְתָּ עַל מְקוֹמוֹ וְאֵינֶנּוּ, שֶׁהַצַּדִּיק דָּן אֶת הַכֹּל לְכַף זְכוּת וּמוֹצֵא זְכוּת אֲפִלּוּ בְּרָשָׁע גָּמוּר וְעַל יְדֵי זֶה מַעֲלֵהוּ לְכַף זְכוּת בֶּאֱמֶת וּמְתַקְּנוֹ וּמַחֲזִירוֹ בִּתְשׁוּבָה וְעַל יְדֵי זֶה נַעֲשִֹין נִגּוּנִים בִּבְחִינַת אֲזַמְּרָה לֵאלֹקַי בְּעוֹדִי וְזֶה בְּחִינַת שְׁלִיחַ צִבּוּר שֶׁהוּא מְלַקֵּט וּמְבָרֵר הַטּוֹב מִכָּל הַמִּתְפַּלְּלִים וְכוּ' עַיֵּן שָׁם הֵיטֵב. נִמְצָא שֶׁהַתִּקּוּן שֶׁמְּתַקֵּן אֶת הַנְּשָׁמוֹת וּמְבָרְרָם וּמַעֲלֵם וּמְקַבֵּץ הַנְּקֻדּוֹת טוֹבוֹת שֶׁיֵּשׁ בָּהֶם זֶה בְּחִינַת נִגּוּנִים כַּנַּ"ל וְכַמְבֹאָר שָׁם הֵיטֵב וְזֶה בְּחִינַת תִּקּוּנֵי הַשָּדֶה הַנַּ"ל כִּי תִּקּוּן הַשָּדֶה הוּא תִּקּוּן הַנְּשָׁמוֹת לְקַבֵּץ הַטּוֹב שֶׁלָּהֶם וּלְהַעֲלוֹתָם אֶל הַקְּדֻשָּׁה שֶׁזֶּה בְּחִינַת נִגּוּנִים כַּנַּ"ל. וְזֶה שֶׁכָּתַב רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה שָׁם שֶׁתִּקּוּן הַשָּדֶה הִיא שְׁלֵמוּת הַתְּפִלָּה. כִּי אֵלּוּ הַנִּגּוּנִים הֵם בְּחִינַת הַשְּׁלִיחַ צִבּוּר שֶׁהוּא עוֹשֶֹה הַנִּגּוּנִים הַנַּ"ל כַּנַּ"ל וְהַשְּׁלִיחַ צִבּוּר הוּא שְׁלֵמוּת הַתְּפִלָּה כִּי עִקַּר הַתְּפִלָּה עַל יְדֵי הַשְּׁלִיחַ צִבּוּר וְזֶה בְּחִינַת הַמִּנְיָן שֶׁצָּרִיךְ לִהְיוֹת בַּעֲשָֹרָה וְאָז הַשְּׁלִיחַ צִבּוּר יוֹרֵד לִפְנֵי הַתֵּבָה וַאֲזַי הוּא שְׁלֵמוּת הַתְּפִלָּה. זֶה בְּחִינַת עֲשָֹרָה מִינֵי נְגִינָה שֶׁנַּעֲשִֹין עַל יְדֵי מִנְיָן שֶׁל עֲשָֹרָה הַמִּתְפַּלְּלִין וְאָז הֵם רְאוּיִים לִשְׁלִיחַ צִבּוּר שֶׁהוּא מְקַבֵּץ הַטּוֹב וְעוֹשֶֹה הַנִּגּוּנִים כַּנַּ"ל. וְזֶה בְּחִינַת הַמְקַבֵּל לְמֶחֱצָה לִשְׁלִישׁ וְלִרְבִיעַ וְהַחוֹכֵר מְקַבֵּל בִּפְשִׁיטוּת בְּכָךְ וְכָךְ כּוּרִין זֶה בְּחִינַת שִׁיר פָּשׁוּט כָּפוּל מְשֻׁלָּשׁ מְרֻבָּע בְּחִינַת נִגּוּנִים הַנַּ"ל שֶׁעוֹשֶֹה הַיּוֹרֵד לְתוֹךְ הַשָּדֶה לְתַקְּנָהּ שֶׁזֶּה עִקַּר תִּקּוּן הַשָּדֶה כַּנַּ"ל כִּי הוּא נוֹתֵן פֵּרוֹת הַשָּדֶה לְבַעַל הַשָּדֶה יִתְבָּרַךְ שְׁמוֹ לְמֶחֱצָה לִשְׁלִישׁ וְכוּ' כְּדֶרֶךְ הַמְקַבֵּל דְּהַיְנוּ שֶׁמַּעֲלֶה הַטּוֹב שֶׁבָּהֶם לְמַעְלָה אֵלָיו יִתְבָּרַךְ כְּפִי מַה שֶּׁיָּכוֹל לְבָרֵר לְפִי בְּחִינַת הַשָּדֶה לִפְעָמִים מֶחֱצָה אוֹ שְׁלִישׁ וְכוּ' וּמִזֶּה נַעֲשִֹין נִגּוּנִים וְשִׁירוֹת. בִּבְחִינַת שִׁיר פָּשׁוּט כָּפוּל וְכוּ' וְזֶה הַכֹּל כְּמִנְהַג הַמְּדִינָה. כְּפִי מִנְהָג וְהַנְהָגָה שֶׁנּוֹהֲגִין בַּמְדִינָה כִּי כְּפִי הַנְהָגָתָם בִּקְדֻשָּׁה כֵּן הוּא מְבָרֵר וּמַעֲלֶה הַטּוֹב שֶׁבָּהֶם אִם רַב אוֹ מְעַט כִּי לְפִי הַנְהָגָתָם וּקְדֻשָּׁתָם כֵּן הוּא מַעֲלֶה הַטּוֹב שֶׁבָּהֶם:

## אות ב

וְזֶה בְּחִינַת הַמְקַבֵּל אֶת הַשָּדֶה לְשָׁנִים מֻעָטוֹת לֹא יִזְרָעֶנָּה פִּשְׁתָּן וְאֵין לוֹ בְּקוֹרַת שִׁקְמָה כִּי מוּבָן מִדִּבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה שָׁם שֶׁיֵּשׁ מְקַבְּלֵי הַשָּדֶה הַנַּ"ל שֶׁאֵינָם יְכוֹלִים לְקַבֵּל אוֹתָהּ רַק לְשָׁנִים מֻעָטוֹת כִּי מֻכְרָחִים לְהִסְתַּלֵּק מֵחֲמַת זֶה. אֲבָל יֵשׁ שֶׁיְּכוֹלִים לְקַבֵּל אוֹתָהּ לְשָׁנִים מְרֻבּוֹת מֵחֲמַת שֶׁהֵם גְּדוֹלִים וּמֻפְלָגִים בְּמַעֲלָה עֶלְיוֹנָה וְנוֹרָאָה מְאֹד מְאֹד. וְזֶה שֶׁאֵין כֹּחוֹ גָּדוֹל כָּל כָּךְ וְאֵין יָכוֹל לְקַבֵּל הַשָּדֶה וְלַעֲסֹק בְּתִקּוּנִים כִּי אִם שָׁנִים מֻעָטוֹת כַּנַּ"ל עָלָיו הִזְהִירוּ שֶׁלֹּא יִזְרַע פִּשְׁתָּן. דְּהַיְנוּ שֶׁלֹּא יַעֲסֹק בְּתִקּוּנִים הָרְחוֹקִים מְאֹד מִמֶּנּוּ יִתְבָּרַךְ וְהֵם מְכֻנִּים בִּבְחִינַת פִּשְׁתָּן שֶׁהוּא בְּחִינַת גָּרוּעַ כְּמוֹ שֶׁכָּתוּב (בְּרֵאשִׁית ד) וַיָּבֵא קַיִן מִפְּרִי הָאֲדָמָה וְאִיתָא שֶׁהֵבִיא מִן הַגָּרוּעַ דְּהַיְנוּ פִּשְׁתָּן. גַּם אִיתָא בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה (סִימָן לח) שֶׁפְּגַם קַיִן הוּא בְּחִינַת קֵץ כָּל בָּשָֹר שֶׁהוּא בְּחִינַת שֶׁמְּחַפֵּשֹ אַחַר חוֹבוֹת בְּנֵי אָדָם וּמְחַזֵּר אַחַר הַגְּרִיעוּת שֶׁבָּהֶן. וְזֶה בְּחִינַת שֶׁהֵבִיא מִן הַגָּרוּעַ שֶׁמֵּבִיא לְמַעְלָה הַגְּרִיעוּת וְהַחוֹבוֹת שֶׁל בְּנֵי אָדָם. נִמְצָא שֶׁפִּשְׁתָּן שֶׁהוּא קָרְבַּן קַיִן הוּא בְּחִינַת חוֹבוֹת וּגְרִיעוּת בְּנֵי אָדָם כִּי פִּשְׁתָּן הוּא בְּחִינַת גְּבוּרוֹת שֶׁמִּשָּׁם אֲחִיזַת הַסִּטְרָא אָחֳרָא כְּמוֹ שֶׁכָּתוּב אִישׁ אֶחָד (גַּבְרִיאֵל) לָבוּשׁ בַּדִּים הוּא פִּשְׁתָּן. וְאוֹתָן הָרְחוֹקִים וְהַגְּרוּעִים בְּיוֹתֵר שֶׁהֵם בְּחִינַת פִּשְׁתָּן אָסוּר לַעֲסֹק בְּתִקּוּנָם מִי שֶׁאֵין כֹּחוֹ חָזָק וְגָדוֹל בְּיוֹתֵר וְאֵין יָכוֹל לְקַבֵּל הַשָּדֶה כִּי אִם לְשָׁנִים מֻעָטוֹת כִּי יוּכְלוּ לְהַזִּיק לוֹ מְאֹד כַּמּוּבָן בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה. וְזֶה אֵין לוֹ בְּקוֹרַת שִׁקְמָה הוּא גַּם כֵּן דֻּגְמָתָם דְּהַיְנוּ שֶׁכְּבָר הִזְקִינוּ בְּרִשְׁעָתָם וְנִתְעַבּוּ וְנִתְגַּשְּׁמוּ כְּקוֹרַת שִׁקְמָה כִּי הַיֵּצֶר הָרָע נִקְרָא בְּשֵׁם קוֹרָה כְּמוֹ שֶׁאָמְרוּ, טֹל קוֹרָה מִבֵּין עֵינֶיךָ וְהוּא בְּחִינַת קוֹרַת שִׁקְמָה שֶׁהִוא אִילָן זָקֵן בְּחִינַת מֶלֶךְ זָקֵן וְכוּ' כִּי שִׁקְמִים הֵם חִלּוּף אֲרָזִים כְּמוֹ שֶׁכָּתוּב שִׁקְמִים גֻּדָּעוּ וַאֲרָזִים נַחֲלִיף. וּכְמוֹ שֶׁכָּתוּב בִּשְׁלֹמֹה וַיִּתֵּן אֶת הַכֶּסֶף בִּירוּשָׁלַיִם כָּאֲבָנִים וְאֶת הָאֲרָזִים נָתַן כַּשִּׁקְמִים. נִמְצָא שֶׁשִּׁקְמִים הֵם גְּרוּעִים וְהֵם חִלּוּף אֲרָזִים וְעַל כֵּן הַתַּנָּא מְכַנֶּה אֶת הָרְחוֹקִים בְּיוֹתֵר בִּבְחִינַת קוֹרַת שִׁקְמָה. כִּי הַצַּדִּיק נִקְרָא אֶרֶז כְּמוֹ שֶׁכָּתוּב צַדִּיק כַּתָּמָר יִפְרָח כְּאֶרֶז בַּלְּבָנוֹן יִשְֹגֶּה וְהָרְחוֹקִים שֶׁהֵם הֵפֶךְ הַצַּדִּיק הֵם בְּחִינַת שִׁקְמִים שֶׁהֵם הֵפֶךְ אֲרָזִים כַּנַּ"ל. אֲבָל מִי שֶׁמְּקַבֵּל הַשָּדֶה לְשָׁנִים מְרֻבּוֹת יָכוֹל לְזָרְעָהּ אֲפִלּוּ פִּשְׁתָּן וְגַם יֵשׁ לוֹ בְּקוֹרַת שִׁקְמָה דְּהַיְנוּ כְּשֶׁכֹּחוֹ אַבִּיר וְחָזָק מְאֹד מְאֹד עַד שֶׁיָּכוֹל לְקַבֵּל אֶת הַשָּדֶה לְשָׁנִים מְרֻבּוֹת דְּהַיְנוּ שֶׁיַּעֲסֹק בְּתִקּוּנֵי הַשָּדֶה וְאַף עַל פִּי כֵן יַאֲרִיךְ יָמִים וְשָׁנִים עַל מַמְלַכְתּוֹ, זֶה יָכוֹל לַעֲסֹק אֲפִלּוּ בְּפִשְׁתָּן וַאֲפִלּוּ בְּהַזְקֵנִים בְּרִשְׁעָתָם כְּקוֹרַת שִׁקְמָה שֶׁאֲפִלּוּ אוֹתָן הָרְחוֹקִים בְּיוֹתֵר וְנִזְקְנוּ בְּרִשְׁעָתָם, יָכוֹל לְקָרְבָם וּלְתַקְּנָם מֵחֲמַת שֶׁכֹּחוֹ גָּדוֹל וְחָזָק מְאֹד מְאֹד כַּנַּ"ל:

## אות ג

וְשִׁעוּר שָׁנִים מֻעָטוֹת הוּא עַד שֶׁבַע שָׁנִים, עַל פִּי מַה דְּאִיתָא בַּמַּאֲמָר פָּתַח רַבִּי שִׁמְעוֹן וְאָמַר עֵת לַעֲשֹוֹת לַה' וְכוּ' (סִימָן ס) שֶׁיֵּשׁ בְּנֵי אָדָם שֶׁנּוֹפְלִים לִבְחִינַת שֵׁנָה וְאוֹבְדִים פְּנֵיהֶם דְּהַיְנוּ שֶׁנִּסְתַּלֵּק הַמֹּחִין שֶׁלָּהֶם וְיֵשׁ שִׁבְעִים פָּנִים שֶׁהֵם בְּחִינַת שִׁבְעִים שָׁנִים וְיֵשׁ שֶׁנָּפְלוּ מִפָּנִים אֶחָד אוֹ שְׁנַים אוֹ יוֹתֵר. וּמַחֲזִירִין לָהֶם פָּנִים שֶׁלָּהֶם עַל יְדֵי סִפּוּרֵי מַעֲשִֹיּוֹת שֶׁל שִׁבְעִים שָׁנִים שֶׁהֵם בְּחִינַת בְּקֶרֶב שָׁנִים. אֲבָל יֵשׁ שֶׁנָּפְלוּ מִכָּל הַשִּׁבְעִים פָּנִים שִׁבְעִים שָׁנִים וְאִי אֶפְשָׁר לְהַחֲזִיר לָהֶם פְּנֵיהֶם כִּי אִם עַל יְדֵי סִפּוּרֵי מַעֲשִֹיּוֹת שֶׁל שָׁנִים קַדְמוֹנִיּוֹת עַיֵּן שָׁם הֵיטֵב. יֵשׁ לְרַמֵּז שֶׁזֶּה בְּחִינַת שְׁנֵי מִינֵי מְקַבְּלִים הַנַּ"ל שֶׁיֵּשׁ מְקַבֵּל הַשָּדֶה לְשָׁנִים מֻעָטוֹת דְּהַיְנוּ עַד שֶׁבַע שָׁנִים דְּהַיְנוּ שֶׁהוּא עוֹסֵק בְּתִקּוּנֵי הַשָּדֶה הַנַּ"ל רַק בִּבְחִינַת שָׁנִים מֻעָטוֹת שֶׁהוּא בְּחִינַת בְּקֶרֶב שָׁנִים וְעַל כֵּן הַשִּׁעוּר שֶׁבַע שָׁנִים, הוּא בְּחִינַת כְּלָלוּת הַשִּׁבְעִים שָׁנָה שֶׁנִּכְלָלִין בְּזַיִּי"ן, וְהָא בְּהָא תַּלְיָא שֶׁמֵּחֲמַת זֶה שֶׁאֵין עוֹסֵק כִּי אִם בִּבְחִינַת שָׁנִים מֻעָטוֹת דְּהַיְנוּ בְּחִינַת בְּקֶרֶב שָׁנִים כַּנַּ"ל עַל כֵּן בֶּאֱמֶת אֵינוֹ מְקַבֵּל אֶת הַשָּדֶה כִּי אִם לְשָׁנִים מֻעָטוֹת כִּי מֻכְרָח לְהִסְתַּלֵּק וְאֵינוֹ יָכוֹל לְהַאֲרִיךְ יָמִים מְרֻבִּים מֵאַחַר שֶׁלֹּא הִגִּיעַ לִבְחִינַת שָׁנִים מְרֻבּוֹת שֶׁהֵם בְּחִינַת שָׁנִים קַדְמוֹנִיּוֹת כִּי שָׁם אֲרִיכוּת הַיָּמִים כִּי הֵם בְּחִינַת עַתִּיק אֲרִיכוּת יָמִים כְּמוֹ שֶׁמְּבֹאָר שָׁם בַּמַּאֲמָר הַנַּ"ל:

וְעַל כֵּן זֶה שֶׁאֵין יָכוֹל לַעֲסֹק כִּי אִם בִּבְחִינַת שָׁנִים מֻעָטוֹת כַּנַּ"ל אָסוּר לְקָרֵב הָרְחוֹקִים בְּיוֹתֵר כַּנַּ"ל דְּהַיְנוּ שֶׁהֵם רְחוֹקִים מִכָּל הַשִּׁבְעִים פָּנִים וְהֵם בְּחִינַת פִּשְׁתָּן כִּי פִּשְׁתָּן מְקַלְקֵל הַשָּדֶה עַד שֶׁבַע שָׁנִים רֶמֶז עַל כָּל הַשִּׁבְעִים פָּנִים שִׁבְעִים שָׁנִים שֶׁכְּלוּלִים בְּזַיִּן כַּנַּ"ל, וְהֵם מְקַלְקְלִים עַד שֶׁבַע שָׁנִים דְּהַיְנוּ שֶׁהֵם מְקַלְקְלִים וְנוֹפְלִים מִכָּל הַשִּׁבְעִים פָּנִים כַּנַּ"ל וְאֵינָם יְכוֹלִים לְהִתְתַּקֵּן כִּי אִם לְאַחַר שֶׁבַע שָׁנִים דְּהַיְנוּ לְמַעְלָה מִזַּיִן הַיְנוּ לְמַעְלָה מֵשִּׁבְעִים שָׁנִים שִׁבְעִים פָּנִים הַיְנוּ בְּחִינַת שָׁנִים קַדְמוֹנִיּוֹת שֶׁהֵם לְמַעְלָה מֵשִּׁבְעִים שָׁנִים כַּנַּ"ל וְהֵם בְּחִינַת שָׁנִים מְרֻבּוֹת כַּנַּ"ל. וְעַל כֵּן מִי שֶׁמְּקַבֵּל הַשָּדֶה לְשָׁנִים מְרֻבּוֹת דְּהַיְנוּ שֶׁיָּכוֹל לַעֲסֹק בְּתִקּוּנֵי הַשָּדֶה בִּבְחִינַת שָׁנִים מְרֻבּוֹת שֶׁהֵם בְּחִינַת שָׁנִים קַדְמוֹנִיּוֹת הוּא יָכוֹל לִזְרֹעַ וּלְתַקֵּן אֲפִלּוּ בְּחִינַת פִּשְׁתָּן כַּנַּ"ל. וְדַיְקָא בְּשָׁנִים הָרִאשׁוֹנוֹת קֹדֶם שֶׁמַּגִּיעַ לְשֶׁבַע שָׁנִים אַחֲרוֹנִים כַּמּוּבָא הַדִּין בַּשֻּׁלְחָן עָרוּךְ הַיְנוּ שֶׁאֵינוֹ יָכוֹל לַעֲסֹק בְּתִקּוּנָם כִּי אִם בִּבְחִינַת שָׁנִים הָרִאשׁוֹנוֹת שֶׁהֵם בְּחִינַת שָׁנִים קַדְמוֹנִיּוֹת כַּנַּ"ל. וּמֵחֲמַת שֶׁהוּא עוֹסֵק בִּבְחִינַת שָׁנִים קַדְמוֹנִיּוֹת עַל כֵּן גַּם בֶּאֱמֶת מְקַבֵּל הַשָּדֶה לְשָׁנִים מְרֻבּוֹת מַמָּשׁ כִּי זוֹכֶה לַאֲרִיכוּת יָמִים כִּי שָׁם אֲרִיכוּת הַיָּמִים בְּחִינַת עַתִּיק כַּנַּ"ל:

וְקַיִן שֶׁהֵבִיא פִּשְׁתָּן נֶאֱמַר בּוֹ וַיִּפְּלוּ פָּנָיו לְשׁוֹן רַבִּים שֶׁנָּפְלוּ כָּל פָּנָיו כַּנַּ"ל:

## אות ד

וְעַל כֵּן נִקְרָא יוֹרֵד כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה הַיּוֹרֵד לְתוֹךְ שְֹדֵה חֲבֵרוֹ. הַיְנוּ בְּחִינַת שְׁלִיחַ צִבּוּר שֶׁנִּקְרָא גַּם כֵּן יוֹרֵד כְּמוֹ שֶׁאָמְרוּ הַיּוֹרֵד לִפְנֵי הַתֵּבָה. וְהַטַּעַם כִּי הוּא בְּחִינַת יוֹרֵד מַמָּשׁ שֶׁצָּרִיךְ לֵירֵד לְעִמְקֵי עֲמָקִים לְהַעֲלוֹת הַטּוֹב מִכָּל הָרְחוֹקִים כַּנַּ"ל שֶׁזֶּה בְּחִינַת הַשְּׁלִיחַ צִבּוּר כַּנַּ"ל. וְזֶה בְּחִינַת מְקַבֵּל הַשָּדֶה שֶׁהוּא בְּחִינַת שְׁלִיחַ צִבּוּר שֶׁעוֹשֶֹה נִגּוּנִים כַּנַּ"ל. וְזֶה שֶׁנִּקְרָא אָרִיס וּמְקַבֵּל. כִּי זֶה הַצַּדִּיק הַמֻּפְלָג בְּיוֹתֵר שֶׁיָּכוֹל לַעֲסֹק בְּתִקּוּנֵי הַשָּדֶה הַנַּ"ל הוּא בְּחִינַת מֹשֶׁה כִּי הֶעֱלָה נַפְשׁוֹת יִשְֹרָאֵל מִמַּדְרֵגָה פְּחוּתָה מְאֹד מְטֻּמְאַת מִצְרַיִם לְמַדְרֵגָה נוֹרָאָה מְאֹד לְקַבָּלַת הַתּוֹרָה כִּי מֹשֶׁה הָיָה עָנָו מִכָּל הָאָדָם וְהָיָה מוֹצֵא זְכוּת בְּכָל אֶחָד מִיִּשְֹרָאֵל וְהָיָה יָכֹל לְהַלְבִּישׁ הַתּוֹרָה בְּסִפּוּרֵי מַעֲשִֹיּוֹת שֶׁל שָׁנִים קַדְמוֹנִיּוֹת הַיְנוּ רֹב סֵפֶר בְּרֵאשִׁית שֶׁהֵם סִפּוּרֵי מַעֲשִֹיּוֹת שֶׁל שָׁנִים קַדְמוֹנִיּוֹת. כִּי הַדּוֹרוֹת הָרִאשׁוֹנִים הָיוּ יְמֵיהֶם אֲרֻכִּים בְּיוֹתֵר. וְאִיתָא שֶׁזֶּה מֵחֲמַת שֶׁהָיוּ יְמֵיהֶם מִבְּחִינַת שָׁנִים קַדְמוֹנִיּוֹת. וּמֹשֶׁה רַבֵּנוּ עָלָיו הַשָּׁלוֹם סִפֵּר מַעֲשִֹיּוֹת שֶׁלָּהֶם בְּסֵפֶר בְּרֵאשִׁית. שֶׁזֶּה בְּחִינַת סִפּוּרֵי מַעֲשִֹיּוֹת שֶׁל שָׁנִים קַדְמוֹנִיּוֹת. וְעַל כֵּן זָכָה מֹשֶׁה שֶׁלִּמְּדוֹ ה' יִתְבָּרַךְ לִהְיוֹת שְׁלִיחַ צִבּוּר, כְּמוֹ שֶׁכָּתוּב וַיַּעֲבֹר ה' עַל פָּנָיו. וְאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁנִּתְעַטֵּף הַקָּדוֹשׁ בָּרוּךְ הוּא כִּשְׁלִיחַ צִבּוּר וְלִמְּדוֹ סֵדֶר תְּפִלָּה. וְעַל כֵּן הָיָה יָכֹל לְהַמְשִׁיךְ בְּחִינַת פָּנִים לְיִשְֹרָאֵל. כְּמוֹ שֶׁמָּצִינוּ כְּשֶׁרָצָה ה' יִתְבָּרַךְ לְסַלֵּק בְּחִינַת הַפָּנִים עַל יְדֵי פְּגַם יִשְֹרָאֵל עָמַד מֹשֶׁה בִּתְפִלָּה וּפָעַל בַּקָּשָׁתוֹ. וְהִמְשִׁיךְ הַפָּנִים כְּמוֹ שֶׁכָּתוּב פָּנַי יֵלְכוּ וְכוּ'. וְעַל כֵּן זָכָה לְקֵרוּן אוֹר הַפָּנִים וְדִבֵּר ה' אֶל מֹשֶׁה פָּנִים אֶל פָּנִים עַל יְדֵי שֶׁהָיָה מַמְשִׁיךְ וּמַלְבִּישׁ אֶת יִשְֹרָאֵל בְּאוֹר הַפָּנִים וְעַל יְדֵי זֶה זָכָה לְקַבֵּל הַתּוֹרָה. כְּמוֹ שֶׁאִיתָא בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה (סִימָן יג) שֶׁעַל יְדֵי שֶׁמַּעֲלִין נַפְשׁוֹת יִשְֹרָאֵל עַל יְדֵי זֶה זוֹכִין לְקַבֵּל אֶת הַתּוֹרָה בִּבְחִינַת וְלֹקֵחַ נְפָשׁוֹת חָכָם בְּחִינַת עִיר גִּבּוֹרִים עָלָה חָכָם וַיּוֹרֶד עֹז מִבְטֶחָה עַיֵּן שָׁם. גַּם אִיתָא שֶׁעַל יְדֵי תְּפִלָּה שֶׁכּוֹלֵל נַפְשׁוֹ עִם נְפָשׁוֹת רַבִּים וּמִתְפַּלֵּל עַל יְדֵי זֶה מַמְשִׁיךְ תּוֹרָה וּשְׁנֵי הַבְּחִינוֹת הֵם אֶחָד, כִּי עַל יְדֵי שֶׁמַּעֲלֶה נַפְשׁוֹת יִשְֹרָאֵל דְּהַיְנוּ שֶׁמַּעֲלֶה הַטּוֹב שֶׁבָּהֶם עַל יְדֵי זֶה הוּא שְׁלֵמוּת הַתְּפִלָּה בִּבְחִינַת תְּפִלּוֹת רַבִּים שֶׁזֶּה בְּחִינַת הַשְּׁלִיחַ צִבּוּר כַּנַּ"ל וְעַל יְדֵי זֶה מְקַבֵּל הַתּוֹרָה כַּנַּ"ל וְעַל כֵּן נִקְרָא מְקַבֵּל בְּחִינַת מֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי. וְזֶה שֶׁנִּקְרָא אָרִיס שֶׁזֶּה בְּחִינַת מֹשֶׁה שֶׁהַתּוֹרָה מְאֹרָסָה לוֹ כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה עַל פָּסוּק תּוֹרָה צִוָּה לָנוּ מֹשֶׁה מוֹרָשָׁה אַל תִּקְרֵי מוֹרָשָׁה אֶלָּא מְאֹרָסָה. וְעַל כֵּן נִקְרָא אָרִיס כַּנַּ"ל:

## אות ה

עַל פִּי מַה שֶּׁמּוּבָא לְעֵיל דִּבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה. שֶׁבְּחִינַת שְׁלִיחַ צִבּוּר הוּא שֶׁמַּעֲלֶה הַטּוֹב מִכָּל אֶחָד וְאֶחָד וְעוֹשֶֹה מִזֶּה נִגּוּנִים וְעוֹמֵד וּמִתְפַּלֵּל עִם כָּל הַטּוֹב עַיֵּן שָׁם בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה, עַל פִּי זֶה מְבֹאָר הֵיטֵב סֵדֶר הַתְּפִלָּה שֶׁבִּתְחִלָּה אוֹמְרִים סֵדֶר הַקָּרְבָּנוֹת וּקְטֹרֶת שֶׁזֶּה לְבָרֵר וּלְהַעֲלוֹת הַטּוֹב מֵעִמְקֵי הַקְּלִפּוֹת כַּמְבֹאָר וְעַל יְדֵי זֶה שֶׁמַּעֲלִין וּמְבָרְרִין הַנְּקֻדּוֹת טוֹבוֹת מִן הָרַע עַל יְדֵי זֶה נַעֲשִֹין נִגּוּנִים וּזְמִירוֹת כַּנַּ"ל. וְזֶה שֶׁאוֹמְרִין תֵּכֶף אַחַר כָּךְ פְּסוּקֵי דְּזִמְרָה שֶׁהֵם בְּחִינַת הַזְּמִירוֹת וְנִגּוּנִים שֶׁנַּעֲשִֹין עַל יְדֵי בֵּרוּר הַנְּקֻדּוֹת טוֹבוֹת שֶׁהָיוּ מְפֻזָּרִים בְּתוֹךְ הָרַע כַּנַּ"ל שֶׁזֶּה בְּחִינַת אֲזַמְּרָה לֵאלֹקַי בְּעוֹדִי כַּנַּ"ל בְּחִינַת פְּסוּקֵי דְּזִמְרָה כַּנַּ"ל. וְעַל כֵּן אוֹמְרִים בְּסוֹפָם שִׁירַת הַיָּם שֶׁהִוא שֶׁבַח נוֹרָא מְאֹד וְנֶאֱמַר בְּדֶרֶךְ שִׁירָה כִּי זֶה אָנוּ צְרִיכִין עַתָּה לוֹמַר שִׁירָה הַיְנוּ זְמִירוֹת וְשִׁירוֹת בְּחִינַת נִגּוּנִים שֶׁנַּעֲשִֹין מִקִּבּוּץ וּבֵרוּר הַטּוֹב כַּנַּ"ל. וְעַל כֵּן בְּיִשְׁתַּבַּח שֶׁהוּא גְּמַר וְסִיּוּם הַשִּׁירוֹת וְהַזְּמִירוֹת אָז צָרִיךְ הַשְּׁלִיחַ צִבּוּר לֵירֵד לִפְנֵי הַתֵּבָה כְּמוֹ שֶׁכָּתוּב בְּשֻׁלְחָן עָרוּךְ אֹרַח חַיִּים שֶׁעִקָּר שֶׁהַשְּׁלִיחַ צִבּוּר יוֹרֵד לִפְנֵי הַתֵּבָה הוּא מִן יִשְׁתַּבַּח כִּי אֲזַי הוּא הַזְּמַן שֶׁל הַשְּׁלִיחַ צִבּוּר הַיְנוּ בִּגְמַר הַזְּמִירוֹת וְשִׁירוֹת אָז יוֹרֵד לִפְנֵי הַתֵּבָה וּכְלַל כָּל הַזְּמִירוֹת וְהַנִּגּוּנִים בְּיִשְׁתַּבַּח שֶׁהוּא סִיּוּם וַחֲתִימַת כָּל הַזְּמִירוֹת וְכוֹלֵל כֻּלָּם בִּשְׁלֹשׁ עֶשְֹרֵה מִינֵי זִמְרָה כִּי עִקַּר בְּחִינַת הַשְּׁלִיחַ צִבּוּר בִּבְחִינַת הַנִּגּוּנִים כַּנַּ"ל וְעַל כֵּן הַמִּנְהָג שֶׁהַשְּׁלִיחַ צִבּוּר מַתְחִיל לְנַגֵּן בְּיִשְׁתַּבַּח. וְזֶה שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁשִּׁירַת הַיָּם נֶאֶמְרָה בְּסֵדֶר שְׁלִיחַ צִבּוּר וְעוֹנִים כְּמוֹ שֶׁאָמְרוּ מֹשֶׁה אָמַר אָשִׁירָה וְהֵם עוֹנִים וְכוּ' כִּי עִקַּר בְּחִינַת הַשְּׁלִיחַ צִבּוּר בִּבְחִינַת שִׁירָה וְנִגּוּן כַּנַּ"ל, וְאָז סָמוּךְ לְשִׁירַת הַיָּם הַשְּׁלִיחַ צִבּוּר יוֹרֵד לִפְנֵי הַתֵּבָה וְאוֹמֵר יִשְׁתַּבַּח כַּנַּ"ל. וְאַחַר כָּךְ אוֹמְרִים קְרִיאַת שְׁמַע שֶׁהִוא בְּחִינַת כֻּלּוֹ אֶחָד כֻּלּוֹ טוֹב שֶׁלָּזֶה זוֹכִין עַל יְדֵי הַנִּגּוּנִים וְהַזְּמִירוֹת דְּהַיְנוּ עַל יְדֵי שֶׁמְּקַבְּצִין וּמַעֲלִין הַטּוֹב מֵהָרַע וּמְקַבְּצִין אוֹתוֹ וְעוֹשִֹין מִזֶּה נִגּוּנִים כַּנַּ"ל וַאֲזַי הוּא בְּחִינַת כֻּלּוֹ טוֹב כִּי הָרַע נִתְבַּטֵּל עַל יְדֵי שֶׁבֵּרְרוּ מִשָּׁם הַנִּיצוֹצוֹת וְהַנְּקֻדּוֹת טוֹבוֹת וְנַעֲשֶֹה כֻּלּוֹ טוֹב, עַל כֵּן אֲזַי אוֹמְרִים קְרִיאַת שְׁמַע בְּחִינַת ה' אֶחָד בְּחִינַת כֻּלּוֹ טוֹב כֻּלּוֹ אֶחָד כַּנַּ"ל, אַחַר כָּךְ עוֹמְדִין בִּתְפִלָּה כִּי עַתָּה אָנוּ יְכוֹלִין לְהִתְפַּלֵּל בִּשְׁלֵמוּת בְּתַכְלִית כִּי עִקַּר שְׁלֵמוּת הַתְּפִלָּה עַל יְדֵי בְּחִינַת כֻּלּוֹ אֶחָד כֻּלּוֹ טוֹב כַּנַּ"ל כַּמְבֹאָר הֵיטֵב כָּל זֶה בַּמַּאֲמָר וַיֹּאמֶר בֹּעַז הַנַּ"ל. אַחַר כָּךְ הַשְּׁלִיחַ צִבּוּר חוֹזֵר הַתְּפִלָּה שֶׁזֶּה בְּחִינַת שְׁלֵמוּת הַתְּפִלָּה בִּבְחִינַת כֻּלּוֹ אֶחָד וְטוֹב דְּהַיְנוּ כְּשֶׁעוֹמֵד בְּסוֹף הַתְּפִלָּה עֲדַיִן הוּא בִּתְחִלַּת הַתֵּבָה שֶׁל תְּחִלַּת הַתְּפִלָּה שֶׁזֶּה בְּחִינַת חֲזָרַת הַשְּׁלִיחַ צִבּוּר שֶׁאַחַר שֶׁהַצִּבּוּר גּוֹמְרִים וְעוֹמְדִים בְּסוֹף הַתְּפִלָּה עֲדַיִן הוּא בַּכֹּל מָשָׁלָה וְכוּ' עַיֵּן שָׁם כָּל זֶה הֵיטֵב:

וְהַכְּלָל שֶׁכְּשֶׁמְּקַשְּׁרִין אֶת מַחֲשַׁבְתּוֹ בְּהַקָּדוֹשׁ בָּרוּךְ הוּא בְּשֵׁם הויה בָּרוּךְ הוּא בִּבְחִינַת מַ"ה אֲזַי מַמְשִׁיכִין חִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא מִבְּחִינַת כַּסֵּדֶר לִבְחִינַת מַלְכוּת שֶׁהִיא בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וַאֲזַי בַּאֲתַר דְּאִית דְּכַר נוּקְבָא לֹא אִדְכַּר תַּמָּן וַאֲזַי נִתְבַּטֵּל בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וְנַעֲשֶֹה הַכֹּל כַּסֵּדֶר וַאֲזַי הוֹלֵךְ לְהָאָדָם הַכֹּל כַּסֵּדֶר. אֲבָל כְּשֶׁמַּפְרִיד מַחֲשַׁבְתּוֹ מֵהַקָּדוֹשׁ בָּרוּךְ הוּא וְאֵין מַמְשִׁיךְ חִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא מִבְּחִינַת מַ"ה מִבְּחִינַת כַּסֵּדֶר לִבְחִינַת מַלְכוּת בְּחִינַת שֶׁלֹּא כַּסֵּדֶר אֲזַי כִּבְיָכוֹל נִפְרָד בְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר מֵהַקָּדוֹשׁ בָּרוּךְ הוּא מִבְּחִינַת כַּסֵּדֶר וַאֲזַי נַעֲשֶֹה חַס וְשָׁלוֹם בְּחִינַת שֶׁלֹּא כַּסֵּדֶר מַמָּשׁ וַאֲזַי הוֹלֵךְ לְהָאָדָם שֶׁלֹּא כַּסֵּדֶר כִּי הוּא בְּתוֹךְ בְּחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל. וְעִקַּר הַפְּגָם עַל יְדֵי גַּדְלוּת שֶׁהִוא עֲבוֹדָה זָרָה שֶׁזֶּהוּ בְּחִינַת אֲנָא אֶמְלֹךְ דְּהַיְנוּ שֶׁמַּפְרִיד בְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר מֵהַקָּדוֹשׁ בָּרוּךְ הוּא וְכוּ' כַּנַּ"ל. וְזֶה עִקַּר עֲבוֹדַת הָאָדָם לְהַמְשִׁיךְ חִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא לִבְחִינַת מַלְכוּת שֶׁהִוא בְּחִינַת הַנְהָגַת הָעוֹלָם כְּדֵי שֶׁיִּתְקַשֵּׁר בְּחִינַת מַלְכוּת בְּהַקָּדוֹשׁ בָּרוּךְ הוּא וְאָז חוֹזֵר הַכֹּל כַּסֵּדֶר כַּמְבֹאָר לְעֵיל וְהָבֵן שָׁם הֵיטֵב:

## אות ג

וְזֶה בְּחִינַת עֲבוֹדַת הַשָּדֶה כִּי הַשָּדֶה הוּא בְּחִינַת מַלְכוּת בִּבְחִינַת כִּי בַּשָּדֶה מְצָאָהּ כַּמּוּבָא. וְזֶה בְּחִינַת מֶלֶךְ לַשָּדֶה נֶעֱבָד. כִּי הַשָּדֶה הִיא בִּבְחִינַת מַלְכוּת שֶׁהִיא חַוָּה בְּחִינַת שֶׁלֹּא כַּסֵּדֶר. וְעִקַּר עֲבוֹדַת הַשָּדֶה כְּדֵי לְהַמְשִׁיךְ חִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא שֶׁהוּא בְּחִינַת שָׁמַיִם כְּמוֹ שֶׁכָּתוּב וְאַתָּה תִּשְׁמַע הַשָּׁמַיִם, לִבְחִינַת אֶרֶץ עַל יְדֵי הַמָּטָר שֶׁהִיא בְּחִינַת הַחִיּוּת וְהַשֶּׁפַע הַנִּשְׁפָּע מִשָּׁמַיִם לָאָרֶץ מֵאָדָם לְחַוָּה כִּי שָׁמַיִם וָאָרֶץ הֵם בְּחִינַת אָדָם וְחַוָּה כַּיָּדוּעַ. וְזֶהוּ עֹצֶם מַעֲלַת הַגְּשָׁמִים כַּמּוּבָא בְּדִבְרֵי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כְּמוֹ שֶׁכָּתוּב אִם בְּחֻקֹּתַי תֵּלֵכוּ וְכוּ' וְנָתַתִּי גִשְׁמֵיכֶם בְּעִתָּם וְכוּ'. וְכֵן לְהֵפֶךְ וְסַרְתֶּם וַעֲבַדְתֶּם אֱלֹהִים אֲחֵרִים וְכוּ' וְעָצַר אֶת הַשָּׁמַיִם וְלֹא יִהְיֶה מָטָר וְכוּ'. נִמְצָא שֶׁהַתּוֹרָה תָּלְאָה הַשָּכָר וְהָעֹנֶשׁ בְּהַמָּטָר כִּי בֶּאֱמֶת הַמָּטָר הַיּוֹרֵד מִשָּׁמַיִם לָאָרֶץ זֶה מְרַמֵּז עַל הַהַשְׁפָּעָה הַנִּשְׁפָּע מִבְּחִינַת אָדָם מַ"ה דְּהַיְנוּ מֵהַקָּדוֹשׁ בָּרוּךְ הוּא לִבְחִינַת מַלְכוּת כַּנַּ"ל. וְזֶה עִקַּר תִּקּוּן כָּל הַתּוֹרָה וְהַמִּצְוֹת כַּיָּדוּעַ וּכְמוֹ שֶׁמּוּבָא בַּזֹּהַר הַקָּדוֹשׁ שֶׁעִקַּר הַמָּטָר בָּא כְּשֶׁחוֹזְרִין אַנְפִּין בְּאַנְפִּין כְּמוֹ שֶׁאִיתָא בַּזֹּהַר בַּמַּעֲשֶֹה כְּשֶׁבָּא לְרַבִּי שִׁמְעוֹן עֲבוּר גְּשָׁמִים עַיֵּן שָׁם כִּי עִקַּר הַמָּטָר הוּא כְּשֶׁבְּחִינַת אָדָם מַ"ה כַּסֵּדֶר מַשְׁפִּיעַ לִבְחִינַת חַוָּה וְכוּ' וְאָז חוֹזֵר הַכֹּל כַּסֵּדֶר וְאָז הוּא בְּחִינַת אַנְפִּין בְּאַנְפִּין וְאָז יוֹרֵד שֶׁפַע הַמָּטָר מִשָּׁמַיִם לָאָרֶץ שֶׁהֵם בְּחִינַת אָדָם וְחַוָּה כַּנַּ"ל. וְעַל כֵּן עַל יְדֵי פְּגַם עֲבוֹדָה זָרָה שֶׁעַל יְדֵי זֶה הוּא בְּחִינַת הַסְתָּרַת פָּנִים, אֲזַי מַפְרִיד הַשֶּׁפַע מֵהַקָּדוֹשׁ בָּרוּךְ הוּא לִבְחִינַת מַלְכוּת כִּי זֶה הוּא בְּחִינַת וְנִרְגָּן מַפְרִיד אַלּוּף, הויה בְּמִלּוּי אַלְפִין וְאָז וְעָצַר אֶת הַשָּׁמַיִם וְלֹא יִהְיֶה מָטָר כַּנַּ"ל:

## אות ד

וְעִקַּר עֲבוֹדַת הָאֲדָמָה מַה שֶּׁצְּרִיכִין לְכַמָּה טְרָחוֹת וַעֲבוֹדוֹת קָשׁוֹת לַעֲבֹד אֶת הַשָּדֶה, זֶה הוּא מֵחֲמַת חֵטְא אָדָם הָרִאשׁוֹן שֶׁנִּתְקַלְּלָה הָאֲדָמָה עֲבוּרוֹ כְּמוֹ שֶׁכָּתוּב בְּזֵעַת אַפֶּךָ תֹּאכַל לֶחֶם וְכוּ'. דְּהַיְנוּ שֶׁצְּרִיכִין כַּמָּה עֲבוֹדוֹת לַעֲבֹד בַּשָּדֶה קֹדֶם שֶׁאוֹכְלִין לֶחֶם. כִּי עִקַּר חֵטְא אָדָם הָרִאשׁוֹן הָיָה בִּבְחִינָה הַנַּ"ל בִּבְחִינַת נִרְגָּן מַפְרִיד אַלּוּף וְכוּ' כַּנַּ"ל. כִּי אָדָם הָרִאשׁוֹן כָּפַר בָּעִקָּר כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה וְהִפְרִיד הַחִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא לִבְחִינַת מַלְכוּת. כִּי אָדָם נִקְרָא עַל שֵׁם אָדָם הָעֶלְיוֹן וְהָיָה לוֹ כֹּחַ לְהַמְשִׁיךְ חִיּוּת הַנַּ"ל בְּחִינַת מַ"ה לִבְחִינַת מַלְכוּת וְכוּ' כַּנַּ"ל וְעַל יְדֵי חֶטְאוֹ פָּגַם בָּזֶה וְעָשָֹה פֵּרוּד כִּבְיָכוֹל. וְאִם הָיָה מַמְשִׁיךְ הַחִיּוּת בְּחִינַת מַ"ה לִבְחִינַת חַוָּה מַלְכוּת הָיָה הַכֹּל כַּסֵּדֶר כַּנַּ"ל. וְעַל יְדֵי חֶטְאוֹ שֶׁהִפְרִיד הַחִיּוּת כִּבְיָכוֹל מִבְּחִינַת מַלְכוּת עַל יְדֵי זֶה חָזַר הַכֹּל שֶׁלֹּא כַּסֵּדֶר וְעַל יְדֵי זֶה נִתְקַלְּלָה הָאֲדָמָה וְכָל הַדְּבָרִים נִתְקַלְּלוּ כִּי הַכֹּל נַעֲשָֹה בִּבְחִינַת שֶׁלֹּא כַּסֵּדֶר מֵחֲמַת שֶׁהִפְרִיד הַחִיּוּת מִבְּחִינַת כַּסֵּדֶר כַּנַּ"ל. וְעִקַּר הַקִּלְקוּל הָיָה בְּהָאֲדָמָה כְּמוֹ שֶׁכָּתוּב אֲרוּרָה הָאֲדָמָה וְכוּ' כִּי הָאֲדָמָה הִיא בְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל שֶׁשָּׁם עִקַּר הַפְּגָם כַּנַּ"ל עַל יְדֵי שֶׁלֹּא הִמְשִׁיךְ הַחִיּוּת מִבְּחִינַת כַּסֵּדֶר לִבְחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל, וּמֵחֲמַת זֶה צְרִיכָה הַשָּדֶה לְכַמָּה יְגִיעוֹת וַעֲבוֹדוֹת בְּחִינַת בְּזֵעַת אַפֶּךָ תֹּאכַל לֶחֶם כַּנַּ"ל וְעַל יְדֵי אֵלּוּ הָעֲבוֹדוֹת מוֹצִיאִין הָאֲדָמָה מִקִּלְלָתָהּ כִּי עַל יְדֵי הָעֲבוֹדוֹת שֶׁל הַשָּדֶה מְתַקְּנִין אֶת הַשָּדֶה וְיוֹרֵד עָלֶיהָ מָטָר שֶׁהוּא בְּחִינַת שֶׁפַע וְחִיּוּת מִשָּׁמַיִם לָאָרֶץ וְאָז חוֹזֵר הַכֹּל כַּסֵּדֶר כַּנַּ"ל:

## אות ה

וְזֶה בְּחִינַת שֶׁהָעֵת שֶׁהַתְּבוּאָה נִגְמֶרֶת נִקְרֵאת אָבִיב כְּמוֹ שֶׁכָּתוּב שָׁמוֹר אֶת חֹדֶשׁ הָאָבִיב וְכוּ'. אָבִיב זֶה בְּחִינַת א"ב כַּסֵּדֶר כַּמּוּבָא. כִּי בְּתִשְׁרֵי זוֹרְעִין תִּשְׁרֵי הוּא בְּחִינַת תַּשְׁרַ"ק וְאָז הוּא עֲדַיִן קֹדֶם הַתִּקּוּן שֶׁל הַשָּדֶה וְאָז הוּא עֲדַיִן בִּבְחִינַת שֶׁלֹּא כַּסֵּדֶר. וְאַחַר כָּךְ כָּל הָעֲבוֹדוֹת וְהַתִּקּוּנִין שֶׁעוֹשִֹין בַּשָּדֶה וְיוֹרְדִין עָלֶיהָ גְּשָׁמִים כָּל יְמוֹת הַגְּשָׁמִים וְעַל יְדֵי זֶה נִתְתַּקֵּן בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וּמְקַבֵּל חִיּוּת דְּהַיְנוּ שֶׁפַע הַגְּשָׁמִים מִבְּחִינַת כַּסֵּדֶר. וְאָז בִּשְׁעַת גְּמַר הַתְּבוּאָה נִתְתַּקֵּן הַכֹּל וְחוֹזֵר הַכֹּל כַּסֵּדֶר וְעַל כֵּן אָז הוּא בְּחִינַת אָבִיב כַּסֵּדֶר כַּנַּ"ל. כִּי יֵשׁ כַּמָּה מִצְוֹת הַנְּהוּגוֹת בַּקַּרְקַע מֵעֵת הַחֲרִישָׁה וְהַזְּרִיעָה עַד שֶׁנִּגְמַר וְנַעֲשָֹה פַּת כְּמוֹ שֶׁכָּתוּב בְּטוּר אֹרַח חַיִּים סִימָן קסז, וְעַל יְדֵי אֵלּוּ הַמִּצְוֹת מְתַקְּנִין חֵטְא אָדָם הָרִאשׁוֹן שֶׁעִקַּר הַפְּגָם הָיָה בִּבְחִינַת הָאֲדָמָה שֶׁהִיא בְּחִינַת מַלְכוּת כַּנַּ"ל בְּחִינַת אֲרוּרָה הָאֲדָמָה וְכוּ' וְעַל כֵּן הֵם יוּ"ד מִצְוֹת כְּדֵי לְהַמְשִׁיךְ הַחִיּוּת מִבְּחִינַת כַּסֵּדֶר שֶׁהִיא בְּחִינַת יוּ"ד לִבְחִינַת מַלְכוּת שֶׁהִיא בְּחִינַת הַשָּדֶה כְּדֵי שֶׁיַּחֲזֹר הַכֹּל כַּסֵּדֶר וְזֶה בְּחִינַת אֶרֶץ נָתְנָה יְבוּלָהּ יְבָרְכֵנוּ אֱלֹקִים אֱלֹקֵינוּ כִּי בְּעֵת שֶׁנִּתְבַּשְּׁלָה הַתְּבוּאָה בְּחִינַת אָבִיב כַּסֵּדֶר אָז נִתְבַּטְּלָה הַקְּלָלָה שֶׁל הָאֲדָמָה שֶׁנִּמְשְׁכָה מִבְּחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל וְאָז נִמְשַׁךְ בְּרָכָה בְּחִינַת יְבָרְכֵנוּ אֱלֹקִים וְכוּ':

שֶׁהֵם בְּחִינַת תְּרֵי מִמֵּאָה אֲדָנִים שֶׁנַּעֲשִֹין וְנִמְשָׁכִין מִבְּחִינַת מַ"ה לְמֵאָה בִּבְחִינַת אַל תִּקְרֵי מַה אֶלָּא מֵאָה כַּנַּ"ל, וְזֶהוּ שִׁעוּר תְּרוּמָה תְּרֵי מִמֵּאָה כְּדֵי לְיַחֵד וּלְקַשֵּׁר וּלְהַעֲלוֹת בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וְכַסֵּדֶר אָדָם וְחַוָּה וְכוּ' כַּנַּ"ל שֶׁהֵם בְּחִינַת תְּרֵי מִמֵּאָה כִּי עַל יָדָם נַעֲשֶֹה בְּחִינַת מֵאָה מִבְּחִינַת מַ"ה עַל יְדֵי שֶׁנִּמְשָׁךְ הַחִיּוּת מֵאָדָם לְחַוָּה מִבְּחִינַת מַ"ה לִבְחִינַת מַלְכוּת וְאָז נַעֲשֶֹה מִמַּה מֵאָה כַּיָּדוּעַ כִּי מַ"ה הוּא בְּחִינַת יוּד כַּנַּ"ל וּכְשֶׁנִּמְשָׁךְ הַחִיּוּת לִבְחִינַת מַלְכוּת כַּנַּ"ל אֲזַי נַעֲשֵֹית גַּם כֵּן בִּבְחִינַת יוּד כִּי וָאו דָּלֶ"ת שֶׁהוּא בְּחִינַת הַמְשָׁכַת הַחִיּוּת לִבְחִינַת מַלְכוּת הוּא גַּם כֵּן בְּחִינַת יוּד תַּתָּאָה כַּיָּדוּעַ ## אות ז

וְעַל כֵּן נוֹתְנִין הַתְּרוּמָה לַכֹּהֵן כִּי הַכֹּהֵן הוּא בְּחִינַת חֶסֶד בְּחִינַת אַבְרָהָם בְּחִינַת בְּרָכָה שֶׁזֶּהוּ בְּחִינַת כַּסֵּדֶר שֶׁהוּא בְּחִינַת חֶסֶד בְּרָכָה. וְאַחַר כָּךְ נוֹתְנִין מַעֲשֵֹר לַלֵּוִי, לֵוִי הוּא בְּחִינַת גְּבוּרוֹת וְדִינִים בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וְנוֹתְנִין לוֹ מַעֲשֵֹר בְּחִינַת יוּד כְּדֵי לְהַמְשִׁיךְ הַיּוּד הַנַּ"ל לִבְחִינַת שֶׁלֹּא כַּסֵּדֶר שֶׁהוּא בְּחִינַת לֵוִי. וְאַחַר כָּךְ חוֹזְרִין וְנוֹתְנִין תְּרוּמָה, מַעֲשֵֹר מִן הַמַּעֲשֵֹר לַכֹּהֵן כִּי כְּשֶׁמְּקַבֵּל בְּחִינַת שֶׁלֹּא כַּסֵּדֶר מִבְּחִינַת כַּסֵּדֶר שֶׁהוּא בְּחִינַת יוּד אֲזַי חוֹזֵר וְנַעֲשֶֹה הַכֹּל כַּסֵּדֶר כִּי בַּאֲתַר דְּאִית דְּכַר נוּקְבָא לֹא אִדְכַּר תַּמָּן כַּנַּ"ל. וְזֶהוּ בְּחִינַת שֶׁחוֹזְרִין וְנוֹתְנִין תְּרוּמַת מַעֲשֵֹר לַכֹּהֵן כִּי חוֹזְרִין וּמְקַשְּׁרִין הַכֹּל לַכֹּהֵן בְּחִינַת כַּסֵּדֶר כְּדֵי שֶׁיִּהְיֶה הַכֹּל כַּסֵּדֶר כִּי בַּאֲתַר דְּאִית דְּכַר וְכוּ' הַיְנוּ כִּי כְּשֶׁבְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר הַיְנוּ כְּלָלִיּוּת הָעוֹלָמוֹת מְקַבְּלִין חִיּוּת מֵהַקָּדוֹשׁ בָּרוּךְ הוּא מִבְּחִינַת מַ"ה הֵם חוֹזְרִים וּמִתְקַשְּׁרִים בּוֹ יִתְבָּרַךְ בִּבְחִינַת פָּנִים בְּפָנִים וַאֲזַי נַעֲשֶֹה הַכֹּל כַּסֵּדֶר כַּנַּ"ל. וְזֶהוּ בְּחִינַת תְּרוּמַת מַעֲשֵֹר לַכֹּהֵן כִּי בִּתְחִלָּה מַעֲלִין הַכֹּל לִבְחִינַת כֹּהֵן עַל יְדֵי הַתְּרוּמָה וְנִכְלָל הַכֹּל בִּבְחִינַת כַּסֵּדֶר בִּבְחִינַת כֹּהֵן וְאַחַר כָּךְ מַמְשִׁיכִין הַחִיּוּת בִּבְחִינַת יוּד הַנַּ"ל גַּם לְמַטָּה לִבְחִינַת הָעוֹלָמוֹת לִבְחִינַת שֶׁלֹּא כַּסֵּדֶר וְזֶהוּ בְּחִינַת מַעֲשֵֹר לַלֵּוִי. וְאַחַר כָּךְ חוֹזְרִין וּמִתְקַשְּׁרִין לַכֹּהֵן לִבְחִינַת כַּסֵּדֶר עַל יְדֵי תְּרוּמַת מַעֲשֵֹר אֶחָד מִמֵּאָה כְּדֵי שֶׁבְּחִינַת שֶׁלֹּא כַּסֵּדֶר יִהְיֶה תָּמִיד מְקֻשָּׁר בִּבְחִינַת כַּסֵּדֶר כְּדֵי שֶׁיִּהְיֶה נַעֲשֶֹה הַכֹּל כַּסֵּדֶר כִּי בַּאֲתַר דְּאִית דְּכַר וְכוּ' כַּנַּ"ל:

## אות ח

וְזֶה בְּחִינַת מַתְּנוֹת עֲנִיִּים לֶקֶט שִׁכְחָה וּפֵאָה שֶׁנּוֹתְנִין מֵהַשָּדֶה כִּי עִקַּר הַהַשְׁפָּעָה שֶׁנִּשְׁפָּע מִבְּחִינַת כַּסֵּדֶר בְּחִינַת יוּד לִבְחִינַת מַלְכוּת וְכוּ' הוּא עַל יְדֵי צְדָקָה כַּמּוּבָא בְּחִינַת צֶדֶק לְפָנָיו יְהַלֵּךְ בְּחִינַת פָּנִים בְּחִינַת קֶדֶם שֶׁהוּא בְּחִינַת כַּסֵּדֶר הַיְנוּ עַל יְדֵי צְדָקָה מַעֲלִין אֶת הַמַּלְכוּת לִבְחִינַת כַּסֵּדֶר בְּחִינַת פָּנִים וְנַעֲשֶֹה מִצֶּדֶק צְדָקָה וְכוּ' כַּנַּ"ל:

## אות ט

וְזֶה בְּחִינַת מַה שֶּׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁאֵין הַתְּבוּאָה נִקְבַּעַת לְמַעַשְֹרוֹת עַד שֶׁיִּרְאֶה פְּנֵי הַבַּיִת. כִּי צְרִיכִין לְבָרֵר וּלְהַעֲלוֹת הַתְּבוּאָה מִבְּחִינַת שָֹדֶה שֶׁהוּא בְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר לִבְחִינַת בַּיִת שֶׁהוּא דִּירַת אָדָם בְּחִינַת כַּסֵּדֶר. וְזֶה בְּחִינַת בֵּית מִבְּרֵאשִׁית שֶׁהוּא הַתְחָלַת הַתּוֹרָה שֶׁהוּא בְּחִינַת אָדָם מַ"ה בְּחִינַת חָכְמָה בִּבְחִינַת זֹאת הַתּוֹרָה אָדָם. וְעַל כֵּן כְּשֶׁהַתְּבוּאָה רוֹאָה פְּנֵי הַבַּיִת, פְּנֵי הַבַּיִת דַּיְקָא בְּחִינַת פָּנִים בְּחִינַת קֶדֶם הַנַּ"ל שֶׁהוּא בְּחִינַת פָּנִים, הַיְנוּ כְּשֶׁהַתְּבוּאָה נִכְנֶסֶת מִשָּדֶה לְבַיִת הַיְנוּ מִיָּד שֶׁמַּתְחִיל הַתִּקּוּן שֶׁל הַתְּבוּאָה שֶׁעוֹלָה מִבְּחִינַת שֶׁלֹּא כַּסֵּדֶר לִבְחִינַת כַּסֵּדֶר אָז נִקְבַּעַת לִתְרוּמָה וּמַעַשְֹרוֹת כְּדֵי לִגְמֹר הַתִּקּוּן וּלְהַמְשִׁיךְ הַחִיּוּת מִבְּחִינַת כַּסֵּדֶר לִבְחִינַת שֶׁלֹּא כַּסֵּדֶר וְכוּ' כַּנַּ"ל:

## אות י

נִמְצָא שֶׁעִקַּר עֲבוֹדַת הַשָּדֶה הוּא כְּדֵי לְבָרְרָהּ וּלְתַקְּנָהּ לְהַמְשִׁיךְ לָהּ חִיּוּת מִבְּחִינַת כַּסֵּדֶר, לִבְחִינַת הַשָּדֶה שֶׁהִיא בְּחִינַת שֶׁלֹּא כַּסֵּדֶר. וְעַל כֵּן כָּל וּמְקַבֵּל כַּיָּדוּעַ. וּכְשֶׁהַנּוֹתֵן וְהַמְקַבֵּל אוֹ הַחוֹכֵר מִתְנַהֲגִים כָּרָאוּי שֶׁהַנּוֹתֵן נוֹתֵן חֶלְקוֹ שֶׁל הַמְקַבֵּל בִּשְׁלֵמוּת וְהַמְקַבֵּל אֵינוֹ נוֹטֵל לְעַצְמוֹ כְּלוּם יוֹתֵר וּלְקַשֵּׁר הַכֹּל לְמִי שֶׁשַּׁיָּךְ לוֹ לִבְחִינַת נוֹתֵן שֶׁהוּא בְּחִינַת אָדָם כַּנַּ"ל וְלִבְלִי לְקַבֵּל לְעַצְמוֹ רַק מַה שֶּׁמַּגִּיעַ לוֹ לֹא יוֹתֵר כְּדֵי שֶׁלֹּא לִפְגֹּם בִּבְחִינַת ## אות יא

וְזֶה בְּחִינַת הַחִלּוּק שֶׁבֵּין הַנּוֹתֵן וְהַחוֹכֵר שֶׁבְּהַנּוֹתֵן אֵין תּוֹלִין הַקִּלְקוּל אַף עַל פִּי שֶׁנִּשְׁתַּדְּפוּ רֹב שְֹדוֹתָיו (כִּי אִם כְּשֶׁיֵּשׁ מַכַּת מְדִינָה) אֲבָל בְּהַחוֹכֵר תּוֹלִין הַקִּלְקוּל כְּשֶׁנִּשְׁתַּדְּפוּ רֹב שְֹדוֹתָיו כִּי אָז אַף עַל פִּי שֶׁיֵּשׁ מַכַּת מְדִינָה שֶׁהַדִּין נוֹתֵן לְנַכּוֹת לוֹ מֵחֲכוּרוֹ אַף עַל פִּי כֵן אֵין מְנַכֶּה לוֹ וְכוּ' כַּנַּ"ל. כִּי הַנּוֹתֵן הוּא בִּבְחִינַת כַּסֵּדֶר וְהַחוֹכֵר הוּא בִּבְחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל וְכָל הַקִּלְקוּלִים שֶׁבָּאִים לְהַשָּדֶה חַס וְשָׁלוֹם הַכֹּל הוּא מִבְּחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל וְעַל כֵּן לְעוֹלָם תּוֹלִין יוֹתֵר הַקַּלְקָלָה בִּמְקַבֵּל דְּהַיְנוּ בְּחוֹכֵר מִבְּנוֹתֵן כִּי עִקַּר הַקַּלְקָלָה מִבְּחִינַת שֶׁלֹּא כַּסֵּדֶר כַּנַּ"ל:

## אות יב

וְזֶה בְּחִינַת עִנְיַן בִּלְעָם שֶׁהָפַךְ ה' הַקְּלָלָה לִבְרָכָה כְּמוֹ שֶׁכָּתוּב וַיַּהֲפֹךְ ה' אֱלֹקֶיךָ לְךָ אֶת הַקְּלָלָה לִבְרָכָה כִּי אֲהֵבְךָ ה' אֱלֹקֶיךָ. כִּי בִּלְעָם חָשַׁב לְקַלֵּל אֶת יִשְֹרָאֵל חַס וְשָׁלוֹם עַל יְדֵי חֵטְא הָעֵגֶל כְּמוֹ שֶׁכָּתוּב וַיָּשֶׁת אֶל הַמִּדְבָּר פָּנָיו כְּמוֹ שֶׁפֵּרֵשׁ רַשִׁ"י כִּי עַל יְדֵי עֲבוֹדָה זָרָה מַפְרִידִין הַחִיּוּת הַנִּמְשָׁךְ מִבְּחִינַת כַּסֵּדֶר מִבְּחִינַת מַ"ה וַאֲזַי הוֹלֵךְ שֶׁלֹּא כַּסֵּדֶר וַאֲזַי תּוּכַל הַקְּלָלָה חַס וְשָׁלוֹם לָחוּל כִּי עִקַּר אֲחִיזַת הַקְּלָלָה שֶׁהִוא דִּינִים הוּא עַל יְדֵי בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וְזֶה בְּחִינַת הַקְּלָלָה שֶׁנִּתְקַלְּלוּ מֵחֵטְא אָדָם הָרִאשׁוֹן כַּנַּ"ל. וְהַכֹּל תָּלוּי בְּהַדִּבּוּר שֶׁהוּא בְּחִינַת מַלְכוּת שֶׁלֹּא כַּסֵּדֶר שֶׁכָּל בְּחִינַת שֶׁלֹּא כַּסֵּדֶר נַעֲשֶֹה עַל יְדֵי שֶׁנִּפְרַד מִבְּחִינַת כַּסֵּדֶר עַל יְדֵי חֵטְא עֲבוֹדָה זָרָה בְּחִינַת נִרְגָּן מַפְרִיד אַלּוּף כַּנַּ"ל אֲזַי יְכוֹלִין לַהֲפֹךְ הַדִּבּוּר לִקְלָלָה חַס וְשָׁלוֹם. אֲבָל ה' יִתְבָּרַךְ חָמַל עַל יִשְֹרָאֵל וְהָפַךְ הַקְּלָלָה לִבְרָכָה כְּמוֹ שֶׁכָּתוּב וַיַּהֲפֹךְ ה' וְכוּ' וַיַּהֲפֹךְ דַּיְקָא. כִּי בְּרַחֲמָיו חָמַל עַל יִשְֹרָאֵל וְהָיָה מַמְשִׁיךְ חִיּוּת כַּסֵּדֶר שֶׁמִּשָּׁם הִשְׁתַּלְשְׁלוּת אֲחִיזַת הַקְּלָלָה חַס וְשָׁלוֹם לִבְחִינַת פָּנִים לִבְחִינַת בְּרָכָה כַּנַּ"ל. וְכָל זֶה הָיָה עַל יְדֵי מֹשֶׁה כִּי אִיתָא שָׁם בַּמַּאֲמָר הַנַּ"ל אֱלֹקוּת אֲפִלּוּ בְּהַקַּל שֶׁבַּקַּלִּים וְעַל יְדֵי זֶה מְקַשְּׁרוֹ לַה' יִתְבָּרַךְ. וְכָל זֶה פָּתַח מֹשֶׁה דֶּרֶךְ כְּבוּשָׁה בְּאַרְבָּעִים יָמִים אַחֲרוֹנִים שֶׁל קַבָּלַת הַתּוֹרָה כַּמּוּבָא שָׁם כִּי עַל יְדֵי הַתּוֹרָה הַקְּדוֹשָׁה שֶׁהִמְשִׁיךְ מֹשֶׁה לְיִשְֹרָאֵל עַל יְדֵי זֶה יִשְֹרָאֵל מְקֻשָּׁרִים לְעוֹלָם בּוֹ יִתְבָּרַךְ כִּי אֲפִלּוּ פּוֹשְׁעֵי יִשְֹרָאֵל מְלֵאִים מִצְוֹת כָּרִמּוֹן, וְתוֹרַת ה' תְּמִימָה מְשִׁיבַת נָפֶשׁ מִדַּרְכֵי מִיתָה לְדַרְכֵי חַיִּים כִּי הַתּוֹרָה מַחְכִּימַת פֶּתִי. וַאֲפִלּוּ מִי שֶׁנָּפַל לְמָקוֹם שֶׁנָּפַל וְהוּא רָחוֹק מְאֹד מֵה' יִתְבָּרַךְ עַל יְדֵי מַעֲשָֹיו הָרָעִים אַף עַל פִּי כֵן יָכוֹל לְהִתְקָרֵב לַה' יִתְבָּרַךְ בְּקַל עַל יְדֵי לִמּוּד הַתּוֹרָה הַקְּדוֹשָׁה כִּי הַתּוֹרָה הַקְּדוֹשָׁה יוֹרֶדֶת אֲפִלּוּ לְמִי ## אות יג

וְזֶה בְּחִינַת מַה שֶּׁפֵּרֵשׁ רַשִׁ"י עַל פָּסוּק וַיָּקָם בִּלְעָם בַּבֹּקֶר וּפֵרֵשׁ רַשִׁ"י רָשָׁע כְּבָר קְדָמְךָ אַבְרָהָם שֶׁנֶּאֱמַר וַיַּשְׁכֵּם אַבְרָהָם בַּבֹּקֶר. כִּי אַבְרָהָם הִשְׁכִּים בַּבֹּקֶר לַעֲקֹד אֶת יִצְחָק בְּנוֹ וְהָעֲקֵדָה הִוא בְּחִינַת הַמְתָּקַת הַדִּינִים כִּי אַבְרָהָם הוּא בְּחִינַת מַה כַּמּוּבָא בְּחִינַת חֶסֶד כַּסֵּדֶר כַּנַּ"ל וְיִצְחָק הוּא בְּחִינַת גְּבוּרוֹת בְּחִינַת שֶׁלֹּא כַּסֵּדֶר וְאַבְרָהָם עַל יְדֵי הָעֲקֵדָה קָשַׁר וְעָקַד אֶת בְּחִינַת יִצְחָק בְּחִינַת שֶׁלֹּא כַּסֵּדֶר לִבְחִינַת אַבְרָהָם בְּחִינַת מַה כַּסֵּדֶר וְעַל יְדֵי זֶה נִכְלָל שֶׁלֹּא כַּסֵּדֶר בִּבְחִינַת כַּסֵּדֶר כִּי נִכְלַל יִצְחָק בְּאַבְרָהָם עַל יְדֵי הָעֲקֵדָה. וְזֹאת הָעֲקֵדָה עוֹמֶדֶת לְיִשְֹרָאֵל תָּמִיד לְכַפֵּר עֲלֵיהֶם בְּכָל עֵת שֶׁנִּתְרַחֲקִין מֵה' יִתְבָּרַךְ חַס וְשָׁלוֹם עַל יְדֵי חֲטָאֵיהֶם חַס וְשָׁלוֹם. כִּי עַל יְדֵי בְּחִינַת הָעֲקֵדָה נִמְתָּק בְּחִינַת שֶׁלֹּא כַּסֵּדֶר בְּחִינַת יִצְחָק וְנִכְלָל בְּכַסֵּדֶר וְעַל יְדֵי זֶה אֵין שׁוּם דִּין שׁוֹרֶה עַל יִשְֹרָאֵל חַס וְשָׁלוֹם כַּנַּ"ל, עַל יְדֵי שֶׁנִּמְשָׁךְ הַחִיּוּת תָּמִיד מִבְּחִינַת כַּסֵּדֶר מַ"ה בְּחִינַת אַבְרָהָם לִבְחִינַת יִצְחָק וְכוּ' כַּנַּ"ל וְעַל כֵּן עַל יְדֵי זֶה נִתְבַּטֵּל וְנִתְהַפֵּךְ הַקְּלָלָה לִבְרָכָה כַּנַּ"ל. וְזֶהוּ שֶׁפֵּרֵשׁ רַשִׁ"י רָשָׁע כְּבָר קְדָמְךָ אַבְרָהָם שֶׁנֶּאֱמַר וַיַּשְׁכֵּם אַבְרָהָם וְכוּ' כִּי עַל יְדֵי הָעֲקֵדָה נִמְתַּק וְנִכְלַל שֶׁלֹּא כַּסֵּדֶר בִּבְחִינַת כַּסֵּדֶר בִּבְחִינַת מַ"ה וְעַל יְדֵי זֶה נִמְשָׁכִין כָּל הַבְּרָכוֹת לְיִשְֹרָאֵל לְעוֹלָם וָעֶד:
