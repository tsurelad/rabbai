# ליקוטי הלכות/יורה דעה/הלכות מליחה

# מְלִיחָה א

עַל-פִּי הַתּוֹרָה 'תְּפִלָּה לַחֲבַקּוּק' סִימָן יט הַנַּ"ל בְּהִלְכוֹת טְרֵפוֹת הֲלָכָה א:

## אות א

וְזֶהוּ עִנְיַן מְלִיחָה, כְּמוֹ שֶׁאָמַר רַבֵּנוּ, נֵרוֹ יָאִיר, בַּבֵּאוּר בְּסִימָן כג; שֶׁהַמֶּלַח הוּא בְּחִינַת בְּרִית מֶלַח עוֹלָם, שֶׁהוּא מַמְתִּיק מְרִירוּתָא דְּעָלְמָא. וְעַל יָדוֹ נִפְלָט הַדָּמִים רָעִים שֶׁהֵם בִּבְחִינַת פְּגַם הַבְּרִית, כַּנִּזְכָּר לְעֵיל בְּהִלְכוֹת חֵלֶב וָדָם הֲלָכָה ה. וְזֶהוּ בְּחִינַת הַהֲדָחָה בְּמַיִם, לְהַכְשִׁיר הַבָּשָֹר עַל-יְדֵי שֶׁמַּכְנִיסִין וְשׁוֹרִין אוֹתוֹ בְּתוֹךְ הַמַּיִם, כִּי הַמַּיִם מְטַהֲרִין אוֹתוֹ מִכָּל הַטֻּמְאוֹת, בִּפְרָט מִפְּגַם הַבְּרִית בִּבְחִינַת מִקְוֵה מַיִם. כִּי הַמִּקְוֶה הוּא תִּקּוּן גָּדוֹל מְאֹד לִפְגַם הַבְּרִית. כִּי הַמִּקְוֶה הוּא סוֹד הִתְעַלְּמוּת בְּתוֹךְ הַמַּיִם כַּמּוּבָא שֶׁזֶּה בְּחִינַת תִּקּוּן הַבְּרִית בִּבְחִינַת זַרְעוֹ שֶׁל יוֹסֵף, שֶׁהֵם בִּבְחִינַת דָּגִים שֶׁבַּיָּם שֶׁהַמַּיִם מְכַסֶּה עֲלֵיהֶם וְאֵין עַיִן הָרָע שֶׁל שִׁבְעִים אֻמּוֹת שׁוֹלֵט בָּהֶם כַּנַּ"ל. (בְּהִלְכוֹת בָּשָֹר שֶׁנִּתְעַלֵּם מִן הָעַיִן ב). וְעַל-כֵּן מְתַקְּנִין וּמַכְשִׁירִין הַבָּשָֹר עַל-יְדֵי הַמַּיִם כְּדֵי לְהַעֲבִיר מֵעָלָיו טֻמְאַת הַדָּמִים שֶׁהֵם בְּחִינַת פְּגַם הַבְּרִית, כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, מִשׁוּם דָּם בְּעֵין שֶׁעָלָיו, שֶׁהוּא עוֹבֵר וְנוֹפֵל עַל-יְדֵי הַמַּיִם שֶׁהֵם בְּחִינַת תִּקּוּן הַבְּרִית, בִּבְחִינַת דָּגִים שֶׁבַּיָּם כַּנַּ"ל. וְזֶה בְּחִינַת הֲדָחָה רִאשׁוֹנָה וַהֲדָחָה אַחֲרוֹנָה. כִּי הַנֶּפֶשׁ שֶׁבַּחַי שֶׁנָּפַל מִמַּדְרֵגָתוֹ, מִמְּדַבֵּר לְחַי כַּנַּ"ל צָרִיךְ כַּמָּה תִּקּוּנִים תִּקּוּן אַחַר תִּקּוּן עַד שֶׁיַּעֲלֶה וְיִתְתַּקֵּן בִּשְׁלֵמוּת לַהֲשִׁיבוֹ לְשָׁרְשׁוֹ בִּבְחִינַת תְּשׁוּבָה, כִּי כָּל תִּקּוּנֵי הַבָּשָֹר שֶׁעוֹשִֹים הֵם בְּחִינַת תִּקּוּנֵי הַתְּשׁוּבָה. כִּי כֵן צְרִיכִין לְתַקֵּן הַנֶּפֶשׁ שֶׁבַּחַי לַהֲשִׁיבוֹ לְשָׁרְשׁוֹ שֶׁזֶּה בְּחִינַת תְּשׁוּבָה. וְעַל-כֵּן מִתְּחִלָּה צָרִיךְ לָהּ שְׁחִיטָה לְהַכְנִיעַ וּלְשַׁבֵּר רַע הַכּוֹלֵל כַּנַּ"ל, שֶׁזֶּהוּ רֵאשִׁית הַתְּשׁוּבָה לָסוּר מֵרַע הַנַּ"ל וּלְשָׁבְרוֹ, וְאַחַר כָּךְ לִפְנוֹת וּלְנַקֵּר אֶת כָּל הַחֲלָבִים רָעִים וְכָל הַגִּידִין הַנִּמְצָאִים בְּעֵין, בְּתוֹכוֹ, כְּדֵי שֶׁלֹּא יִהְיֶה 'טוֹבֵל וְשֶׁרֶץ בְּיָדוֹ'. עַל-כֵּן אַחַר שֶׁסָּר מֵרַע הַכּוֹלֵל הַנַּ"ל שֶׁהוּא עִקַּר הַהַתְחָלָה וְהָרֵאשִׁית אֲזַי צָרִיךְ לָסוּר כָּל הָרָעוֹת מִקִּרְבּוֹ. וְזֶה נַעֲשֶֹה בְּקַל אַחַר הַשְּׁחִיטָה, כִּי עִקַּר הַיְגִיעָה וְהַטִּרְחָא הוּא הַשְּׁחִיטָה שֶׁהוּא הַכְנָעַת הָרַע הַנַּ"ל. שֶׁצָּרִיךְ אֻמָּן לָזֶה מֻמְחֶה וּמֻחְזָק. וּבְדִיקַת הַסַּכִּין וּשְׁאָר הַזְהָרוֹת הַשְּׁחִיטָה כַּנַּ"ל. אֲבָל אַחַר כָּךְ נַעֲשֶֹה הַנִּקּוּר בְּנָקֵל; שֶׁמְּנַקְּרִין וּמוֹצִיאִין מִשָּׁם שְׁאָר כָּל הָרָעוֹת הַנִּמְצָאִין שָׁם. וַאֲזַי זוֹכִין לִבְחִינַת מִקְוֵה מַיִם, שֶׁזֶּה בְּחִינַת הֲדָחָה הָרִאשׁוֹנָה וְאַחַר כָּךְ צָרִיךְ לִשְׁהוֹת בְּמִלְחוֹ הַנַּ"ל. וְאָז הוּא גְּמַר הַתִּקּוּן שֶׁהוּא הֲדָחָה הָאַחֲרוֹנָה, כִּי הַבָּא לִטָּהֵר מְסַיְּעִין לוֹ, שֶׁאוֹמְרִים לוֹ הַמְתֵּן (כְּמוֹ שֶׁמּוּבָא בְּדִבְרֵי רַבֵּנוּ, נֵרוֹ יָאִיר, סִימָן ו), כִּי אִי אֶפְשָׁר לִכָּנֵס בְּפַעַם אֶחָד אֶל הַתְּשׁוּבָה וְצָרִיךְ לְהַמְתִּין וְלִשְׁהוֹת וְלִסְבֹּל שְׁפִיכַת דָּמִים הַרְבֵּה כְּדֵי לְהַכְנִיעַ הַדָּם הָרָע שֶׁבֶּחָלָל הַשְּמָאלִי שֶׁהִתְגַּבֵּר עַל-יְדֵי עֲווֹנוֹתָיו:

## אות ב

וְזֶה בְּחִינַת הַשְּׁהִיָּה שֶׁצָּרִיךְ לִשְׁהוֹת בְּמִלְחוֹ וְלִהְיוֹת כָּבוּשׁ בְּתוֹךְ הַצִּיר בִּכְלִי מְנֻקָּב בִּבְחִינַת אוֹמְרִים לוֹ הַמְתֵּן הַנַּ"ל, כְּדֵי לְהַפְקִיעַ הַדָּמִים רָעִים שֶׁנִּבְלְעוּ בְּתוֹךְ הַבָּשָֹר, שֶׁנִּכְנְסוּ לְעֹמֶק הַגּוּף. וְעַל-כֵּן אִי אֶפְשָׁר לְתַקֵּן הַבָּשָֹר בִּמְהִירוּת, כִּי אִי אֶפְשָׁר לְהַכְנִיסוֹ בְּפַעַם אַחַת אֶל הַתְּשׁוּבָה עַד שֶׁיִּשְׁהֶה בְּמִלְחוֹ שֶׁהוּא בְּחִינַת הַתִּקּוּן, בְּחִינַת בְּרִית מֶלַח, כְּדֵי לְהַמְתִּיק מְרִירוּתוֹ וְלִפְלֹט כָּל הַדָּמִים כַּנַּ"ל. וְעַל-כֵּן הוּא בֵּין שְׁנֵי הֲדָחוֹת שֶׁהֵם בְּחִינַת מִקְוֵה מַיִם כַּנַּ"ל עַל-פִּי מַה שֶּׁמּוּבָא בְּשֵׁם הַבַּעַל שֵׁם טוֹב זַ"ל. וּמְבֹאָר לְעֵיל כַּמָּה פְּעָמִים; שֶׁבִּתְחִלַּת הַתְּשׁוּבָה מַעֲלִין אֶת הָאָדָם לִבְחִינַת מַדְרֵגָה גְּדוֹלָה לְמַעְלָה מִמַּדְרֵגָתוֹ וְאַחַר כָּךְ מוֹרִידִין אוֹתוֹ מִשָּׁם כְּדֵי לְהַבְחִין וּלְהִצְטָרֵף וּלְהִזְדַּכֵּךְ, כִּי צָרִיךְ לָשׁוּב וְלַעֲבֹר בְּכָל אוֹתָן הַדְּרָכִים שֶׁהָיָה שָׁם בִּבְחִינַת תְּשׁוּבַת הַמִּשְׁקָל כַּנַּ"ל:

## אות ג

וְזֶה בְּחִינַת הֲדָחָה רִאשׁוֹנָה וַהֲדָחָה אַחֲרוֹנָה שֶׁהֵם בִּבְחִינַת מִקְוֵה מַיִם כַּנַּ"ל, שֶׁהֵם עֲלִיָּה גְּדוֹלָה וְתִקּוּן גָּדוֹל מְאֹד, שֶׁזֶּה עִקַּר בְּחִינַת הַתְּשׁוּבָה כְּשֶׁזּוֹכֶה לִכָּנֵס לִבְחִינַת מִקְוֵה מַיִם שֶׁהוּא בְּחִינַת עָלְמָא דְּאָתֵי, בְּחִינַת תְּשׁוּבָה כַּנַּ"ל. וּבִתְחִלָּה מַעֲלִין אוֹתוֹ מִלְּמַעְלָה לְמַדְרֵגָה גְּדוֹלָה כַּנַּ"ל, לִבְחִינַת תְּשׁוּבָה, בְּחִינַת מִקְוֵה מַיִם כַּנַּ"ל וְזֶהוּ בְּחִינַת הֲדָחָה רִאשׁוֹנָה. וְאַף שֶׁכְּבָר זָכָה לְתַכְלִית הַתִּקּוּן שֶׁהוּא בְּחִינַת מִקְוֶה כַּנַּ"ל, אַחַר כָּךְ מוֹרִידִין אוֹתוֹ מִשָּׁם כְּדֵי שֶׁיִּצְטָרֵף וְיִזְדַּכֵּךְ כַּנַּ"ל בִּבְחִינַת אוֹמְרִים לוֹ הַמְתֵּן כַּנַּ"ל. וְזֶהוּ בְּחִינַת מַה שֶּׁהוּא צָרִיךְ לִשְׁהוֹת בְּמִלְחוֹ זְמַן מְסֻיָּם, כְּדֵי לְהִזְדַּכֵּךְ וּלְהִצְטָרֵף וְלִפְלֹט כָּל דָּמָיו הָרָעִים שֶׁנִּבְלְעוּ בְּתוֹכוֹ כַּנַּ"ל. וַאֲזַי אַחַר כָּךְ זוֹכִין שֵׁנִית לִגְמֹר הַתִּקּוּן בֶּאֱמֶת וְזֶהוּ בְּחִינַת הֲדָחָה אַחֲרוֹנָה, בְּחִינַת מִקְוֵה מַיִם, שֶׁמַּכְנִיסִין אוֹתוֹ אַחַר כָּךְ כְּשֶׁזָּכָה לְהִזְדַּכֵּךְ, לִפְלֹט דָּמָיו הָרָעִים כַּנַּ"ל. וְזֶה בְּחִינַת יְצִיאַת מִצְרַיִם שֶׁהָיָה בִּבְחִינַת תְּשׁוּבָה, בְּחִינַת יוֹבֵל כַּמּוּבָא. וּבִתְחִלָּה הֶעֱלָה אוֹתָם לְמַעְלָה מִמַּדְרֵגָתָם מְאֹד כַּמּוּבָא. וְזֶה בְּחִינַת מִילָה וּטְבִילָה שֶׁזָּכוּ בְּעֶרֶב פֶּסַח שֶׁתֵּכֶף אַחַר הַמִּילָה שֶׁהִכְנִיעוּ הָרַע הַכּוֹלֵל שֶׁזֶּה בְּחִינַת מִילָה, אֲזַי תֵּכֶף זָכוּ לִבְחִינַת יְצִיאַת מִצְרַיִם שֶׁהוּא בְּחִינַת תְּשׁוּבָה שֶׁהָיָה אָז לְמַעְלָה מִמַּדְרֵגָתָם מְאֹד כַּמּוּבָא:

## אות ד

וְזֶהוּ בְּחִינַת הַמִּקְוֵה מַיִם שֶׁזָּכוּ אָז בְּעֶרֶב פֶּסַח מִיָּד וְאַחַר כָּךְ הוֹרִידָם מִמַּדְרֵגָתָם וְזֶה בְּחִינַת הַסְּפִירָה כַּמּוּבָא, שֶׁהוּא כְּאִשָּׁה הַמִּטַּהֶרֶת לְבַעְלָהּ, שֶׁהוּא בְּחִינַת הַמְתֵּן, שֶׁצָּרִיךְ לְהַמְתִּין וְלִשְׁהוֹת כְּדֵי לְטַהֵר עַצְמוֹ מִדַּם נִדּוֹת כַּנַּ"ל. וְאָז זָכוּ לִגְמַר הַתִּקּוּן בְּשָׁבוּעוֹת. וְאָז זָכוּ לִכְנֹס לַמִּקְוֶה בְּחִינַת מִקְוֶה עֶלְיוֹנָה מְאֹד, שֶׁהוּא הַתִּקּוּן וְזֶה בְּחִינַת שְׁנֵי הַהֲדָחוֹת, בְּחִינַת שְׁנֵי הַמִּקְוָאוֹת הַנַּ"ל, שֶׁצְּרִיכִין כְּדֵי לְהִתְתַּקֵּן, הַיְנוּ בִּתְחִלַּת הַתְּשׁוּבָה וּבְסוֹף הַתְּשׁוּבָה כַּנַּ"ל. וּבָאֶמְצַע הוּא הַשְּׁהִיָה בְּמִלְחוֹ, שֶׁהוּא בְּחִינַת הַמְתֵּן כַּנַּ"ל. כְּדֵי לִפְלֹט הַדָּמִים רָעִים כַּנַּ"ל. וְכֵן מְבֹאָר בַּמַּאֲמַר הַמְדַבֵּר מִזֶּה (בְּסִימָן ו, לִקּוּטֵי חֵלֶק א), שֶׁבְּחִינַת הַמְתֵּן הַנַּ"ל, הוּא בְּחִינַת שְׁפִיכַת דָּמִים, שֶׁצְּרִיכִין לִסְבֹּל כַּנַּ"ל, שֶׁאֲזַי זוֹכִין לִבְחִינַת תְּשׁוּבָה שֶׁהוּא בְּחִינַת אקיה (כְּמוֹ שֶׁמְּבֹאָר שָׁם). וְזֶהוּ בְּחִינַת הַמִּקְוֵה מַיִם שֶׁהוּא בְּחִינַת אקיה, כִּי מִקְוֶה בְּגִמַטְרִיָּא קנ"א, כַּמּוּבָא וְכַנַּ"ל. וּכְבָר מְבֹאָר בְּהִלְכוֹת פֶּסַח הֲלָכָה א, שֶׁעַל-כֵּן שִׁעוּר מְלִיחָה ח"י מִינוּטִין, שֶׁזֶּה בְּחִינַת תִּקּוּן הַבְּרִית, כְּדֵי לָשׁוּב אֶל הַחַיִּים. וְזֶה שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה; שֶׁהַשִּׁעוּר הוּא כְּדֵי הִלּוּךְ מִיל, כִּי הַמִּיל הוּא אַלְפַּיִם אַמָּה תְּחוּם שַׁבָּת, שֶׁהוּא בְּחִינַת גְּבוּל הַקְּדֻשָּׁה כַּנִּזְכָּר לְעֵיל. (בְּהִלְכוֹת תְּחוּמִין, הֲלָכָה א). וְזֶהוּ הַשִּׁעוּר בְּעַצְמוֹ שֶׁל ח"י מִינוּטִין, כִּי שָׁם הַחַיִּים. (כְּמוֹ שֶׁמְּבֹאָר שָׁם). וְעַל-כֵּן צָרִיךְ לִשְׁהוֹת בְּתִקּוּן הַמְּלִיחָה כַּשִּׁעוּר הַזֶּה, כְּדֵי לִזְכּוֹת לָשׁוּב לְתוֹךְ גְּבוּל וּתְחוּם הַקְּדֻשָּׁה, כְּדֵי לִזְכּוֹת לְחַיִּים בְּחִינַת צַדִּיק חַי עָלְמִין, וְזֶהוּ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, הַצְּלִי אֵין צָרִיךְ מְלִיחָה, כִּי אֵשׁ הוּא הַתִּקּוּן בִּבְחִינַת כָּל דָּבָר אֲשֶׁר יָבוֹא בָאֵשׁ תַּעֲבִירוּ בָאֵשׁ וְטָהֵר. וְהוּא מְכַלֶּה אֶת הַזֻּהֲמָא, אֲבָל כְּשֶׁזּוֹכִין לְהִתְקַשֵּׁר לְהַצַּדִּיק שֶׁהוּא בְּחִינַת בְּרִית מֶלַח, וַאֲזַי מְקַבְּלִין מִמֶּנּוּ בְּחִינַת מִלְחָא, בְּחִינַת תִּקּוּנִים לְהַמְתִּיק מְרִירוּתוֹ וְאָז נִצּוֹלִין מִן הָאֵשׁ שֶׁהָיוּ צְרִיכִין לְהִצְטָרֵף חַס וְשָׁלוֹם, בִּבְחִינַת תַּעֲבִירוּ בָאֵשׁ כַּנַּ"ל, כִּי מָלִיחַ הֲרֵי הוּא כְּרוֹתֵחַ דִּצְלִי שֶׁנִּצְּלָה בָּאֵשׁ, שֶׁעַל-יְדֵי בְּחִינַת הַמְּלִיחָה שֶׁהוּא בְּחִינַת הִתְקַשְּׁרוּת לְהַצַּדִּיק כַּנַּ"ל. נִזְדַּכְּכִין יָפֶה וְנִצּוֹלִין מִן הָאֵשׁ וְיוֹצְאִין זַךְ וְיָפֶה וְשָׁלֵם בְּלִי שׁוּם פְּגָם וְזוֹכִין לְתַכְלִית הַתִּקּוּן כַּנַּ"ל:

מְלִיחָה הֲלָכָה ב נִכְלֶלֶת בְּהִלְכוֹת חֵלֶב וָדָם הֲלָכָה ד, וַהֲלָכָה ג נִכְלֶלֶת בְּהִלְכוֹת דָּם הֲלָכָה א:
