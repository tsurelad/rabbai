# ליקוטי הלכות/יורה דעה/הלכות סימני עוף טהור

# סִימָנֵי עוֹף טָהוֹר א

סִימָנֵי עוֹף טָהוֹר לֹא נֶאֶמְרוּ בַּתּוֹרָה

## אות א

כִּי הָעוֹפוֹת הֵם בִּבְחִינַת תּוֹרָה שֶׁבְּעַל פֶּה, בְּחִינַת מִשְׁנָה, בְּחִינַת מט"ט (מְטַטְרוֹן), כַּנִּזְכָּר לְעֵיל (בְּהִלְכוֹת בָּשָֹר שֶׁנִּתְעַלֵּם מִן הָעַיִן, הֲלָכָה א). וְעַל-כֵּן לֹא נִכְתְּבוּ סִימָנֵיהֶם בַּתּוֹרָה וְנִמְסְרוּ לַחֲכָמִים בְּעַל פֶּה, אֲבָל בְּהֵמוֹת וְחַיּוֹת שֶׁהֵם לְמַטָּה מִן הָעוֹפוֹת כַּנַּ"ל, שֶׁשָּׁם הִתְגַּבְּרוּת כֹּחַ הַמְדַמֶּה וְכֹחַ הַבְּהֵמִיּוּת בְּיוֹתֵר. וְעַל-כֵּן אֲחִיזָתָם בִּבְחִינַת תּוֹרָה שֶׁבִּכְתָב, שֶׁהוּא בְּחִינַת כְּתֹב זֹאת זִכָּרוֹן בַּסֵּפֶר, שֶׁשָּׁם אֲחִיזַת הַמְדַמֶּה, כִּי הַזִּכָּרוֹן הוּא בְּחִינַת כֹּחַ הַמְדַמֶּה, כְּמוֹ שֶׁאָמַר רַבֵּנוּ שָׁם (בְּסִימָן יט), עַיֵּן שָׁם. וְעַל-כֵּן נִכְתְּבוּ סִימָנֵיהֶם בַּתּוֹרָה שֶׁבִּכְתָב, כִּי שָׁם אֲחִיזָתָם כַּנַּ"ל. וְכֵן אִיתָא בַּכְּתָבִים, שֶׁבְּחִינַת תּוֹרָה שֶׁבִּכְתָב יוֹרֶדֶת לְהִתְלַבֵּשׁ יוֹתֵר לְמַטָּה בִּבְחִינַת עֲשִֹיָּה, שֶׁשָּׁם עִקַּר הִתְגַּבְּרוּת כֹּחַ הַמְדַמֶּה, כֹּחַ הַבְּהֵמִיּוּת, שֶׁשָּׁם בְּחִינַת בְּהֵמוֹת וְחַיּוֹת שֶׁמִּתְגַּבֵּר בָּהֶם כֹּחַ הַבְּהֵמִיּוּת יוֹתֵר מִן הָעוֹפוֹת, כִּי הֵם הוֹלְכִים עַל אַרְבַּע, כְּמוֹ שֶׁנִּתְבָּאֵר כָּל זֶה בַּבֵּאוּר לְעֵיל (בְּהִלְכוֹת סִימָנֵי בְּהֵמָה וְחַיָּה טְהוֹרָה, הֲלָכָה א), אֲבָל תּוֹרָה שֶׁבְּעַל פֶּה הִיא בְּחִינַת יְצִירָה שֶׁהִיא בְּחִינַת מט"ט כַַּנַּ"ל שֶׁמִּשָּׁם הָעוֹפוֹת. וְעַל-כֵּן לֹא נִכְתְּבוּ סִימָנֵיהֶם בַּתּוֹרָה:

## אות ב

וְעַל-כֵּן עוֹפוֹת טְהוֹרִים הֵם יוֹתֵר מִן הַטְּמֵאִים וּבַבְּהֵמוֹת לְהִפּוּךְ, כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, כִּי לְמַטָּה בַּעֲשִֹיָּה אֲחִיזָתָם בְּיוֹתֵר. וְעַל-כֵּן הַבְּהֵמוֹת שֶׁהֵם מֵעֲשִֹיָּה כַּנַּ"ל, עַל-כֵּן הַטְּמֵאִים יוֹתֵר מֵהַטְּהוֹרִים, אֲבָל הָעוֹפוֹת שֶׁהֵם מִבְּחִינַת יְצִירָה שֶׁשָּׁם אֵין אֲחִיזַת הַסִּטְרָא אָחֳרָא כָּל כָּךְ, עַל-כֵּן הַטְּהוֹרִים בְּיוֹתֵר מִן הַטְּמֵאִים:

## אות ג

וְזֶהוּ גַּם כֵּן מַה שֶּׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, עוֹף טָהוֹר נֶאֱכָל בְּמָסֹרֶת; שֶׁנִּמְסָר מִפֶּה אֶל פֶּה שֶׁהוּא עוֹף טָהוֹר, וּכְמוֹ שֶׁאָמְרוּ, נֶאֱמָן הַצַּיָּד לוֹמַר עוֹף פְּלוֹנִי הִתִּיר לִי רַבִּי הַצַּיָּד, כִּי עִקַּר סִימָנֵי הָעוֹפוֹת וְהֶתֵּרָם הוּא רַק בְּמָסֹרֶת, מִפֶּה אֶל פֶּה, בְּחִינַת תּוֹרָה שֶׁבְּעַל פֶּה כַּנַּ"ל, כִּי אִי אֶפְשָׁר לִסְמֹךְ עַל הַסִּימָנִים הַכְּתוּבִים בְּעוֹפוֹת, כִּי הֵם בִּבְחִינַת תּוֹרָה שֶׁבְּעַל פֶּה שֶׁצְּרִיכָה מְסִירָה מִפֶּה אֶל פֶּה דַּיְקָא, כְּמוֹ שֶׁאָמְרוּ (אָבוֹת פֶּרֶק א), "מֹשֶׁה קִבֵּל תּוֹרָה מִסִּינַי וּמְסָרָהּ לִיהוֹשֻׁעַ וְכוּ'", כִּי דְּבָרִים שֶׁבְּעַל פֶּה אִי אַתָּה רַשַּׁאי לְאָמְרָם בִּכְתָב:

# סִימָנֵי עוֹף טָהוֹר ב

סִימָנֵי עוֹף טָהוֹר לֹא נִתְפָּרְשׁוּ בַּתּוֹרָה וְאָמְרוּ חֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה עוֹף טָהוֹר נֶאֱכָל בְּמָסֹרֶת

כִּי אִיתָא בְּדִבְרֵי רַבֵּנוּ בַּמַּאֲמָר הַמַּתְחִיל, "בְּטַח בַּה' וַעֲשֵֹה טוֹב וְכוּ'" (סִימָן עט) שֶׁיֵּשׁ בְּחִינַת מט"ט וּבְחִינַת שַׁדַּי, הַיְנוּ בְּחִינַת שֵׁשֶׁת יְמֵי הַחֹל וּבְחִינַת שַׁבָּת, הַיְנוּ כִּי יֵשׁ בְּנֵי אָדָם שֶׁהֵם בִּבְחִינַת שֵׁשֶׁת יְמֵי הַחֹל, בְּחִינַת מט"ט, שֶׁהוּא בְּחִינַת כָּשֵׁר וּפָסוּל, טָמֵא וְטָהוֹר וְכוּ', הַיְנוּ שֶׁאֵין לָהֶם נַיְחָא בְּחִינַת שַׁבָּת, כִּי אֵינָם קְבוּעִים בַּעֲבוֹדָתָם וּבִתְשׁוּבָתָם בִּשְׁלֵמוּת תָּמִיד, רַק לִפְעָמִים נוֹפֵל לוֹ הִרְהוּרֵי תְּשׁוּבָה וְאַחַר כָּךְ חוֹזֵר לִמְקוֹמוֹ. וְהוּא פַּעַם כָּשֵׁר, פַּעַם פָּסוּל, פַּעַם טָהוֹר, פַּעַם טָמֵא וְכוּ'. אֲבָל הַצַּדִּיקִים הֵם בִּבְחִינַת שַׁבָּת, כִּי שַׁבָּת הוּא בְּחִינַת תְּשׁוּבָה, בְּחִינַת שַׁדַּי, הַיְנוּ שֶׁיֵּשׁ לָהֶם נַיְחָא, שֶׁתְּשׁוּבָה שֶׁלָּהֶם בִּמְנוּחָה, בִּקְבִיעוּת תָּמִיד, כִּי הֵם דּוֹחִים הָרַע לְגַמְרֵי וְכוּ' וְאֵלּוּ הַצַּדִּיקִים הֵם מוֹסְרִים נַפְשָׁם בִּשְׁבִיל יִשְֹרָאֵל, כִּי הֵם רוֹאִים שִׁפְלוּתָם וַחֲשִׁיבוּת וּגְדֻלַּת יִשְֹרָאֵל וְכוּ' (כַּמְבֹאָר שָׁם הֵיטֵב, עַיֵּן שָׁם):

## אות א

וְהִנֵּה עוֹף דָּא מט"ט, כַּמּוּבָא בַּתִּקּוּנִים (הַקְדָּמָה דַּף ג) וּבְחִינַת מט"ט הוּא בְּחִינַת עֵץ הַדַּעַת טוֹב וָרָע, שֶׁהוּא בְּחִינַת כָּשֵׁר וּפָסוּל, טָמֵא וְטָהוֹר וְכוּ'. וּכְתִיב, "מֵעֵץ הַדַּעַת טוֹב וָרָע לֹא תֹּאכַל מִמֶּנּוּ וְכוּ'. וְכָל הַדְּבָרִים שֶׁהֵם בִּבְחִינַת עֵץ הַדַּעַת טוֹב וָרָע, שֶׁמְּעֹרָב טוֹב וָרָע, אָסוּר לָנוּ לֶאֱכֹל מֵהֶם. וְעַל-כֵּן עִקַּר תִּקּוּן כָּל אֲכִילוֹתֵינוּ הוּא עַל-יְדֵי תּוֹרָה שֶׁבְּעַל פֶּה שֶׁנִּמְסְרָה לַצַּדִּיקִים שֶׁהִיא בְּחִינַת שִׁשָּׁה סִדְרֵי מִשְׁנָה שֶׁהֵם מְבָרְרִים הַטּוֹב מִן הָרַע, הַמֻּתָּר מִן הָאָסוּר וְכוּ'. וְעַל-כֵּן אָנוּ מְצֻּוִּין לַעֲשׂוֹת כְּכָל אֲשֶׁר יוֹרוּ אוֹתָנוּ הַצַּדִּיקִים וְהַחֲכָמִים הָאֲמִתִּיִּים אֲשֶׁר בְּכָל דּוֹר וָדוֹר, כְּמוֹ שֶׁכָּתוּב, "לֹא תָסוּר מִן הַדָּבָר אֲשֶׁר יַגִּידוּ לְךָ יָמִין וּשְֹמֹאל", אֲפִלּוּ אוֹמֵר לְךָ וְכוּ', כִּי בְּכָל דּוֹר וָדוֹר צְרִיכִין לְבָרֵר בֵּרוּרִים לְהַעֲלוֹת הַטּוֹב מִן הָרַע שֶׁנִּתְעָרְבוּ עַל-יְדֵי חֵטְא עֵץ הַדַּעַת שֶׁאָכַל אָדָם הָרִאשׁוֹן. וְאֵין כֹּחַ בְּיַד כָּל אָדָם לְבָרֵר כִּי אִם עַל-יְדֵי הַצַּדִּיק הָאֱמֶת אֲשֶׁר בְּכָל דּוֹר וָדוֹר, כִּי אֲנָשִׁים פְּשׁוּטִים אֲשֶׁר עֲדַיִן לֹא זָכוּ לִתְשׁוּבָה גְּמוּרָה בִּשְׁלֵמוּת כַּנַּ"ל, וְאֵין לָהֶם נַיְחָא עֲדַיִן כַּנַּ"ל; הֲרֵי הֵם בְּעַצְמָם עֲדַיִן בִּבְחִינַת עֵץ הַדַּעַת טוֹב וָרָע, כִּי עֲדַיִן הֵם מְעֹרָבִים פַּעַם כָּשֵׁר וְכוּ' כַּנַּ"ל. וְעַל-כֵּן בְּוַדַּאי אֵין בְּיָדָם לְבַדָּם כֹּחַ לְבָרֵר הַטּוֹב מִן הָרַע, כִּי אִם הַצַּדִּיק הָאֱמֶת אֲשֶׁר בְּכָל דּוֹר, אֲשֶׁר כְּבָר דָּחָה הָרַע לְגַמְרֵי וְהוּא בִּבְחִינַת שַׁבָּת כַּנַּ"ל, הוּא יֵשׁ לוֹ כֹּחַ לְבָרֵר הַבֵּרוּרִים וְהַנִּיצוֹצוֹת שֶׁיֵּשׁ בְּכָל דָּבָר, דְּהַיְנוּ לְבָרֵר הַטּוֹב מִן הָרַע, הַהֶתֵּר מִן הָאִסּוּר וְכוּ'. וְעַל-כֵּן כָּל אֲכִילוֹתֵינוּ בְּכַשְׁרוּת עִקַּר הַתִּקּוּן וְהַבֵּרוּר שֶׁלָּהֶם הוּא עַל-יְדֵי צַדִּיקֵי הַדּוֹר, אֲשֶׁר לָהֶם נִמְסְרָה הַתּוֹרָה שֶׁבְּעַל פֶּה, אֲשֶׁר בְּיָדָם לְבָרֵר כַּנַּ"ל. כִּי אַף-עַל-פִּי שֶׁמְּפֹרָשׁ בַּתּוֹרָה כָּל הַמִּצְווֹת עִם כָּל זֹאת מֵהַתּוֹרָה בְּעַצְמָהּ אֵין יְכוֹלִים לֵידַע דָּבָר, כִּי אִם עַל-יְדֵי הַצַּדִּיקִים וְהַחֲכָמִים אֲשֶׁר לָהֶם נִמְסְרָה הַתּוֹרָה שֶׁבְּעַל פֶּה אֲשֶׁר הֵם יוֹדְעִים פְּרָטֵי כָּל דָּבָר וְדָבָר אֵיךְ לְהִתְנַהֵג בְּכָל דָּבָר וְדָבָר כְּפִי צֹרֶךְ הַבֵּרוּר שֶׁל אוֹתוֹ דָּבָר. וְזֶה עָשָֹה ה' יִתְבָּרַךְ בְּכַוָּנָה וְלֹא בֵּאֵר כָּל דָּבָר בְּפֵרוּשׁ בַּתּוֹרָה # סִימָנֵי עוֹף טָהוֹר ג

בָּעוֹלָם, חַס וְשָׁלוֹם, כַּנַּ"ל. וְעַל-כֵּן לֹא נִתְפָּרְשׁוּ סִימָנֵי עוֹף טָהוֹר בַּתּוֹרָה, כִּי אִי אֶפְשָׁר לְפָרֵשׁ הַסִּימָנִים בַּתּוֹרָה שֶׁבִּכְתָב, כִּי אִי אֶפְשָׁר לְבָרֵר בִּכְתָב הַסִּימָנִים שֶׁבֵּין חָכְמַת הָאֱמֶת שֶׁמִּתְגַּבֶּרֶת עַל-יְדֵי כַּנְפֵי הַקְּדֻשָּׁה, שֶׁמִּשָּׁם יְנִיקַת עוֹפוֹת טְהוֹרִים, וּבֵין חָכְמַת שֶׁקֶר שֶׁל חַכְמֵי הַטֶּבַע שֶׁבָּאִים לִכְפִירוֹת עַל-יְדֵי חָכְמָתָם הָרָעָה, כִּי הַשֶּׁקֶר מִתְגַּבֵּר כְּנֶגֶד הָאֱמֶת עַד שֶׁקָּשֶׁה לְהַבְחִין בֵּינֵיהֶם וְאִי אֶפְשָׁר לְבָאֵר בִּכְתָב הַסִּימָנִים שֶׁבֵּינֵיהֶם, כִּי זֶה יָדוּעַ שֶׁיֵּשׁ כַּמָּה מְבוּכוֹת וּכְפִירוֹת הַנִּמְשָׁכִין מֵחַכְמֵי הַטֶּבַע שֶׁאִי אֶפְשָׁר לְיַשְּׁבָן עַל-פִּי דַּעַת וָשֵֹכֶל כִּי אִם עַל-יְדֵי אֱמוּנָה שֶׁאָנוּ בְּנֵי יִשְֹרָאֵל חֲזָקִים בְּאֱמוּנָתֵנוּ שֶׁמָּסְרוּ לָנוּ אֲבוֹתֵינוּ וְרַבּוֹתֵינוּ (כַּמְבֹאָר בְּמָקוֹם אַחֵר). וְזֶה בְּחִינַת עוֹף טָהוֹר נֶאֱכָל בְּמָסֹרֶת, בְּמָסֹרֶת דַּיְקָא לְרַמֵּז שֶׁאִי אֶפְשָׁר לְהַבְחִין ולְהַבְדִּיל בֵּין עוֹפוֹת טְהוֹרִים שֶׁיְּנִיקָתָם מִכַּנְפֵי הַקְּדֻשָּׁה לְעוֹפוֹת טְמֵאִים שֶׁיּוֹנְקִים מִכַּנְפֵי הַנָּחָשׁ, שֶׁהֵם בְּחִינַת הַמְּבוּכוֹת וְהַכְּפִירוֹת שֶׁל חַכְמֵי הַטֶּבַע אִי אֶפְשָׁר לְהַבְחִין וּלְהַבְדִּיל בֵּינֵיהֶם עַל-יְדֵי שׁוּם סִימָן שֶׁהוּא בְּחִינַת מוֹפֵת הַשֵּכֶל הַנִּקְרָאִים, מוֹפֵת חוֹתֵךְ, כִּי אִם עַל-יְדֵי מָסֹרֶת, דְּהַיְנוּ כְּמוֹ שֶׁמָּסְרוּ לָנוּ אֲבוֹתֵינוּ שֶׁאָנוּ סוֹמְכִים וְנִשְׁעָנִים רַק עַל דִּבְרֵי אֲבוֹתֵינוּ כְּמוֹ שֶׁמָּסְרוּ לָנוּ לְהַאֲמִין בַּה' יִתְבָּרַךְ וּבְחִדּוּשׁ הָעוֹלָם וּבִתְחִיַּת הַמֵּתִים וּבִשְׁנֵי עוֹלָמוֹת וְכוּ' בְּלִי שׁוּם סִימָן וּרְאָיָה וְאוֹת וּמוֹפֵת, כִּי אִי אֶפְשָׁר לְבָרֵר כָּל עִקָּרֵי אֱמוּנָתֵנוּ הַקְּדוֹשָׁה עַל-יְדֵי סִימָנִים וּמוֹפְתִים בְּדַרְכֵי הַשֵּכֶל הַנָּבוֹךְ שֶׁלָּהֶם, כִּי אִם עַל-יְדֵי הַקַּבָּלָה וְהַמָּסֹרֶת, כְּמוֹ שֶׁקִּבַּלְנוּ מֵרַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כַּנַּ"ל:

## אות ב

וְעַל-כֵּן הָעוֹפוֹת טְמֵאִים הֵם כ"ד (עֶשְֹרִים וְאַרְבָּעָה), כִּי כָּל הַמְּחַקְּרִים בְּחָכְמַת הַטֶּבַע, הֵם עוֹסְקִים בַּתּוֹרָה שֶׁבִּכְתָב בְּכָל כ"ד סִפְרֵי הַמִּקְרָא, כִּי טוֹב בְּעֵינֵיהֶם הַמְּלִיצָה שֶׁלָּהֶם, אֲבָל הֵם מְפָרְשִׁים הַכֹּל עַל-פִּי דַּרְכֵיהֶם וּמְבוּכוֹתֵיהֶם, אֲבָל עַל תּוֹרָה שֶׁבְּעַל פֶּה הֵם חוֹלְקִים לְגַמְרֵי, מֵחֲמַת שֶׁבַּתּוֹרָה שֶׁבְּעַל פֶּה שֶׁמָּסְרוּ לָנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה מְפֹרָשׁ בְּפֵרוּשׁ לְהֵפֶךְ מִדֵּעוֹתֵיהֶם וּסְבָרוֹתֵיהֶם, כִּי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה קִבְּלוּ פֵּרוּשׁ תּוֹרָה שֶׁבִּכְתָב אִישׁ מִפִּי אִישׁ עַד מֹשֶׁה רַבֵּנוּ, עָלָיו הַשָּׁלוֹם, כְּפִי שֶׁמָּסְרוּ לָנוּ שֶׁזֶּה עִקַּר אֱמוּנָתֵנוּ הַקְּדוֹשָׁה, כִּי מִתּוֹרָה שֶׁבִּכְתָב אִי אֶפְשָׁר לֵידַע דָּבָר בְּלִי תּוֹרָה שֶׁבְּעַל פֶּה כְּמוֹ שֶׁמָּסְרוּ לָנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה. נִמְצָא, שֶׁעִקַּר הִשְׁתַּלְשְׁלוּת יְנִיקַת חַכְמֵי הַטֶּבַע הוּא מִבְּחִינַת תּוֹרָה שֶׁבִּכְתָב, כִּי מֵאַחַר שֶׁהִיא בִּכְתָב יְכוֹלִים לְפָרְשָׁהּ כְּדַעְתָּם הָרָעָה וְלַהֲפֹךְ דִּבְרֵי אֱלֹקִים חַיִּים. אֲבָל בַּתּוֹרָה שֶׁבְּעַל פֶּה שִׁנֵּיהֶם קֵהוֹת, כִּי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה הֵם מְפָרְשִׁים לָנוּ בְּפֵרוּשׁ לְהֵפֶךְ מִדֵּעוֹתֵיהֶם הָרָעוֹת וְעַל-כֵּן בֶּאֱמֶת הֵם חוֹלְקִים עַל דִּבְרֵי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁהֵם תּוֹרָה שֶׁבְּעַל פֶּה, כִּי מִתּוֹרָה שֶׁבְּעַל פֶּה אֵין לָהֶם יְנִיקָה. וְכֵן מְפֹרָשׁ בְּדִבְרֵי רַבֵּנוּ, זִכְרוֹנוֹ לִבְרָכָה (בְּלִקּוּטֵי תִּנְיָנָא סִימָן כח). עַל פָּסוּק, "אֶכְתֹּב לוֹ רוּבֵּי תּוֹרָתִי וְכוּ', שֶׁעִקַּר יְנִיקַת הָעַכּוּ"ם הוּא מִשְׁתַּלְשֵׁל מִבְּחִינַת תּוֹרָה שֶׁבִּכְתָב, עַיֵּן שָׁם. וְהָעַכּוּ"ם הוּא בְּחִינַת חַכְמֵי הַטֶּבַע, כִּי הֵם עוֹסְקִים תָּמִיד בְּחָכְמוֹת אֵלּוּ, אֲבָל לֹא בְּאֵלּוּ חֵלֶק יַעֲקֹב, כִּי אָנוּ מַאֲמִינִים בְּדִבְרֵי רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, כְּפִי מַה שֶּׁמָּסְרוּ לָנוּ פֵּרוּשׁ תּוֹרָה שֶׁבִּכְתָב וּמִפִּיהֶם אָנוּ חַיִּין. וְעַל-כֵּן הָעוֹפוֹת טְמֵאִים הֵם כ"ד, כִּי שֹׁרֶשׁ יְנִיקָתָם שֶׁמִּשְׁתַּלְשֵׁל בַּאֲלָפִים וּרְבָבוֹת הִשְׁתַּלְשְׁלוּת מִבְּחִינַת תּוֹרָה שֶׁבִּכְתָב, שֶׁהֵם כ"ד סִפְרֵי הַמִּקְרָא, כִּי אֶת זֶה לְעֻמַּת זֶה עָשָֹה אֱלֹקִים כַּיָּדוּעַ, וְעַל-כֵּן אִי אֶפְשָׁר לֶאֱכֹל עוֹף טָהוֹר כִּי אִם בְּמָסֹרֶת, שֶׁהוּא בְּחִינַת תּוֹרָה שֶׁבְּעַל פֶּה כְּפִי מַה שֶּׁמָּסְרוּ לָנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כַּנַּ"ל, כִּי אִי אֶפְשָׁר לְהִנָּצֵל מִדַּעְתָּם הָרָעָה כִּי אִם עַל-יְדֵי תּוֹרָה שֶׁבְּעַל פֶּה שֶׁאָנוּ בְּנֵי יִשְֹרָאֵל אֵין סוֹמְכִין עַל שׁוּם פֵּרוּשׁ בַּתּוֹרָה שֶׁבִּכְתָב עַל-פִּי דַּרְכֵי הַמְּלִיצָה רָעָה שֶׁלָּהֶם כִּי אִם כְּמוֹ שֶׁמָּסְרוּ לָנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה בַּתּוֹרָה שֶׁבְּעַל פֶּה, כִּי בְּהַתּוֹרָה אֵין מְפֹרָשׁ כְּלָל דִּינֵי הַתְּפִלִּין אִם יִהְיֶה שֶׁל עוֹר וְהֵיכָן מְקוֹם קְשִׁירָתָם בִּפְרָטִיּוּת אִם עַל כַּף הַיָּד אִם עַל הַקִּבֹּרֶת וְכוּ' וְכֵן בִּשְׁאָרֵי מִצְווֹת. וְכֵן קִבְּלוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה, שֶׁתְּחִיַּת הַמֵּתִים מְפֹרָשׁ בַּתּוֹרָה כְּמוֹ שֶׁלָּמְדוּ מִכַּמָּה פְּסוּקִים. וְכֵן בִּשְׁאָר יְסוֹדֵי הָאֱמוּנָה וּפֵרוּשֵׁי הַתּוֹרָה וְהַמִּצְווֹת, כֻּלָּם אֵין אָנוּ סוֹמְכִים כִּי אִם כְּמוֹ שֶׁפֵּרְשׁוּ לָנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה בִּגְמָרָא וּפוֹסְקִים. וְזֶה מְרֻמָּז בְּדִין זֶה עוֹף טָהוֹר נֶאֱכָל בְּמָסֹרֶת, כִּי דַּיְקָא עַל-יְדֵי מָסֹרֶת בְּחִינַת תּוֹרָה שֶׁבְּעַל פֶּה יְכוֹלִין לָבוֹא לְהִתְרַחֵק מִן הַכּ"ד עוֹפוֹת טְמֵאִים שֶׁמִּשְׁתַּלְשֵׁל יְנִיקָתָם מִכּ"ד סִפְרֵי הַמִּקְרָא, כִּי דַּיְקָא עַל-יְדֵי הַמָּסֹרֶת בְּחִינַת תּוֹרָה שֶׁבְּעַל פֶּה, אָנוּ נִבְדָּלִין מֵהֶם, כִּי הַמָּסֹרֶת בְּחִינַת תּוֹרָה שֶׁבְּעַל פֶּה מְלַמֵּד אוֹתָנוּ לְפָרֵשׁ תּוֹרָה שֶׁבִּכְתָב עַל-פִּי הָאֱמֶת כְּמוֹ שֶׁקִּבְּלוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כַּנַּ"ל:

## אות ג

וְזֶה שֶׁנָּתְנוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה סִימָן לְעוֹפוֹת טְמֵאִים עַל-יְדֵי הָאֲכִילָה, כִּי אִם הוּא דּוֹרֵס וְאוֹכֵל הוּא בְּוַדַּאי טָמֵא, כִּי הָעוֹפוֹת טְמֵאִים יוֹנְקִים מִכַּנְפֵי הַנָּחָשׁ הַנַּ"ל שֶׁיֵּשׁ לָהּ נְמָלָה בְּתוֹךְ פִּיהָ וְכוּ', שֶׁרְצוֹנָהּ לְהִתְגַּבֵּר עַל הַנְּמָלָה, חַס וְשָׁלוֹם, שֶׁהוּא חֲכַם הָאֱמֶת וְכוּ' (כַּמְבֹאָר שָׁם, עַיֵּן שָׁם). וְעַל-כֵּן הַסִּימַן טֻמְאָה הוּא בְּפִיהָּ, שֶׁאִם הוּא דּוֹרֵס וְאוֹכֵל בְּוַדַּאי טָמֵא, כִּי זֶה מַרְאֶה שֶׁרוֹצֶה לְהִתְגַּבֵּר בַּאֲכִילָתָהּ לִהְיוֹת דּוֹרֵס וְאוֹכֵל לִדְרֹס, חַס וְשָׁלוֹם, אֶת הַנְּמָלָה וְכוּ', כִּי מִשָּׁם יְנִיקַת הָעוֹפוֹת טְמֵאִים כַּנַּ"ל, אֲבָל ה' לֹא יַעַזְבֶנּוּ בְּיָדוֹ וְיַצִּיל אֶת הַנְּמָלָה מִפִּיהָ, כִּי שְֹפַת אֱמֶת תִּכּוֹן לָעַד בִּמְהֵרָה בְּיָמֵינוּ אָמֵן:
