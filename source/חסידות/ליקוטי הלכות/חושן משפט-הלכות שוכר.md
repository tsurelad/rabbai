# ליקוטי הלכות/חושן משפט/הלכות שוכר

הֲלָכָה א נִכְלָל בְּהִלְכוֹת פִּקָּדוֹן הֲלָכָה א וּבְהִלְכוֹת אֻמָּנִין הֲלָכָה א אוֹת ד:

# שוכר ב

הַשּוֹכֵר דִּינוֹ כְּשׁוֹמֵר שָֹכָר שֶׁחַיָּב בִּגְנֵבָה וַאֲבֵדָה וּפָטוּר בְּאוֹנְסִין:

## אות א

עַל פִּי הַמַּאֲמָר הַמַּתְחִיל 'תִּקְעוּ בַחֹדֶשׁ שׁוֹפָר' - לְהַמְשִׁיךְ שָׁלוֹם בָּעוֹלָם' בְּסֵפֶר לִקּוּטֵי מוֹהֲרַ"ן הָרִאשׁוֹן סִימָן יד עַיֵּן שָׁם כָּל הַמַּאֲמָר מִתְּחִלָּתוֹ לְסוֹפוֹ:

וּבְקִצּוּר, כִּי מִי שֶׁרוֹצֶה לְהַמְשִׁיךְ שָׁלוֹם בָּעוֹלָם צָרִיךְ לְהַעֲלוֹת אֶת הַכָּבוֹד לְשָׁרְשׁוֹ, דְּהַיְנוּ לְיִרְאָה וְכוּ' וְאִי אֶפְשָׁר לְהַעֲלוֹת אֶת הַכָּבוֹד כִּי אִם עַל יְדֵי תּוֹרַת חֶסֶד, דְּהַיְנוּ עַל יְדֵי שֶׁמְּלַמְּדִין לַאֲחֵרִים וּמַחֲזִירִים אוֹתָם בִּתְשׁוּבָה וְעוֹשִֹין בַּעֲלֵי תְּשׁוּבָה וְגֵרִים עַל יְדֵי זֶה עִקַּר עֲלִיּוֹת הַכָּבוֹד וְכוּ'. וְאִי אֶפְשָׁר לְעָרְרָם בִּתְשׁוּבָה כִּי אִם עַל יְדֵי הַתּוֹרָה, כִּי עַל יְדֵי הַתּוֹרָה מְעוֹרְרִין זִוּוּגֵי הַנְּפָשׁוֹת שֶׁשָּׁרְשָׁם בְּמַחֲשָׁבָה תְּחִלָּה וְכוּ' וְעַל יְדֵי זֶה נַעֲשִֹין בַּעֲלֵי תְּשׁוּבָה וְגֵרִים וְכוּ' וְעַל יְדֵי זֶה עוֹלֶה הַכָּבוֹד לְשֹׁרֶשׁ הַיִּרְאָה וְעַל יְדֵי הַיִּרְאָה נַעֲשֶֹה שָׁלוֹם הַפְּרָטִי בֵּין הַגּוּף וְהַנֶּפֶשׁ וְעַל יְדֵי הַשָּׁלוֹם הַפְּרָטִי יְכוֹלִין לְהִתְפַּלֵּל וְכוּ' וְעַל יְדֵי הַתְּפִלָּה נַעֲשֶֹה הַשָּׁלוֹם הַכְּלָלִי שָׁלוֹם בְּכָל הָעוֹלָמוֹת וַאֲזַי כָּל הַבְּרִיּוֹת מְרַחֲמִין אֶחָד עַל חֲבֵרוֹ כְּמוֹ שֶׁיִּהְיֶה לֶעָתִיד שֶׁיִּהְיֶה הַשָּׁלוֹם הַמֻּפְלָא בָּעוֹלָם, כְּמוֹ שֶׁכָּתוּב, וְגָר זְאֵב עִם כֶּבֶשֹ וְנָמֵר עִם גְּדִי וְכוּ' לֹא יָרֵעוּ וְלֹא יַשְׁחִיתוּ וְכוּ', עַיֵּן שָׁם כָּל זֶה הֵיטֵב. וְאִיתָא שָׁם עַל מַאֲמַר רַבָּה בַּר בַּר חָנָא שַׁקְלֵיתָא לְסַלְתָּאִי וְאַנַּחְתָּא בְּכַוָּתָא דִּרְקִיעַ, כִּי אֲפִלּוּ הַתְּפִלּוֹת הַמְפֹרָשִׁים בַּתְּפִלָּה לְצֹרֶךְ הַגּוּף, כְּגוֹן רְפָאֵנוּ וּבָרֵךְ עָלֵינוּ וְכוּ' צָרִיךְ לְכַוֵּן בָּהֶם גַּם כֵּן רַק עֲבוּר צָרְכֵי הַנְּשָׁמָה, דְּהַיְנוּ עַל פַּרְנָסַת הַנְּשָׁמָה וּרְפוּאָתָהּ וְכוּ', עַיֵּן שָׁם כָּל זֶה הֵיטֵב הֵיטֵב:

## אות ב

וְהִנֵּה עִקַּר הַשְּׁמִירָה מִן הַהֶזֵּקוֹת הוּא עַל יְדֵי שֶׁמַּמְשִׁיכִין שָׁלוֹם בָּעוֹלָם, כִּי עַל יְדֵי זֶה נִשְׁמָרִין מִכָּל הַהֶזֵּקוֹת כְּמוֹ שֶׁיִּהְיֶה לֶעָתִיד שֶׁאָז יִהְיֶה הַשָּׁלוֹם הַמֻּפְלָא בָּעוֹלָם וְאָז יִתְבַּטְּלוּ כָּל הַהֶזֵּקוֹת שֶׁבָּעוֹלָם, כְּמוֹ שֶׁכָּתוּב (יְשַׁעְיָה יא), וְגָר זְאֵב עִם כֶּבֶשֹ וְכוּ' לֹא יָרֵעוּ וְלֹא יַשְׁחִיתוּ, כִּי אֲפִלּוּ בֵּין בְּהֵמוֹת וְחַיּוֹת רָעוֹת יִהְיֶה שָׁלוֹם וְלֹא יַזִּיקוּ זֶה אֶת זֶה כְּלָל, כִּי כָּל הַהֶזֵּקוֹת הֵם בָּאִים מִכַּעַס וּמַחֲלֹקֶת וּמִשָּׁם נִמְשָׁכִין כָּל מִינֵי הֶזֵּקוֹת שֶׁבָּעוֹלָם, אֲבָל עַל יְדֵי שָׁלוֹם מִתְבַּטְּלִין כָּל הַהֶזֵּקוֹת נִמְצָא שֶׁכָּל הַשּׁוֹמְרִים צְרִיכִין לְהַמְשִׁיךְ שָׁלוֹם בָּעוֹלָם כְּדֵי שֶׁיִּתְבַּטְּלוּ כָּל מִינֵי הֶזֵּקוֹת הַבָּאִים מִכַּעַס וּמַחֲלֹקֶת הֵפֶךְ הַשָּׁלוֹם, כִּי עִקַּר אֲחִיזַת הַסִּטְרָא אָחֳרָא שֶׁמִּשָּׁם בָּאִין כָּל מִינֵי הֶזֵּקוֹת הוּא מִבְּחִינַת מַחֲלֹקֶת וְכַעַס שֶׁהֵם בְּחִינַת דִּינִים. אֲבָל עַל יְדֵי שָׁלוֹם בְּחִינַת רַחֲמִים שֶׁכָּל אֶחָד מְרַחֵם עַל חֲבֵרוֹ נִתְבַּטֵּל אֲחִיזַת הַסִּטְרָא אָחֳרָא וְנִתְבַּטְּלִין כָּל מִינֵי הֶזֵּקוֹת הַנִּמְשָׁכִין מֵהֶם. וּלְהַמְשִׁיךְ שָׁלוֹם הוּא עַל יְדֵי הַעֲלָאַת הַכָּבוֹד לְשָׁרְשׁוֹ לִבְחִינַת יִרְאָה וְזֶה נַעֲשֶֹה עַל יְדֵי תּוֹרַת חֶסֶד עַל יְדֵי זִוּוּגֵי הַנְּפָשׁוֹת בְּמַחֲשָׁבָה תְּחִלָּה שֶׁעַל יְדֵי זֶה מַמְשִׁיכִין נַפְשׁוֹת הַבַּעֲלֵי תְּשׁוּבָה וְהַגֵּרִים וְכוּ' וְנִתְעַלֶּה כְּבוֹדוֹ יִתְבָּרַךְ עַל יְדֵי זֶה, עַיֵּן שָׁם. וְזֶה עִקַּר בְּחִינַת הַשְּׁמִירָה לְהַעֲלוֹת הַכָּבוֹד לְשָׁרְשׁוֹ כְּמוֹ יַעֲקֹב שֶׁהָיָה שׁוֹמֵר שֶׁכָּל הַשּׁוֹמְרִים נִלְמָדִין מִמֶּנּוּ, כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה וְכַמּוּבָא בַּשֻּׁלְחָן עָרוּךְ. וְכָל כַּוָּנַת יַעֲקֹב אָבִינוּ בְּעֵת שְׁמִירָתוֹ צֹאן לָבָן הָיָה בִּשְׁבִיל הַעֲלָאַת הַכָּבוֹד לְשָׁרְשׁוֹ עַל יְדֵי הַמְשָׁכַת נַפְשׁוֹת יִשְֹרָאֵל וְהַגֵּרִים. כִּי בֶּאֱמֶת בְּכָל הַחֲפָצִים וְכָל הָעֲשִׁירוּת שֶׁל הָאָדָם יֵשׁ בָּהֶם בְּחִינַת נִיצוֹצֵי הַכָּבוֹד, כִּי הָעֲשִׁירוּת הִוא בְּחִינַת כָּבוֹד, בְּחִינַת מַלְכוּת וּבִשְׁבִיל זֶה רַבִּי מְכַבֵּד עֲשִׁירִים, כִּי שָׁם שֹׁרֶשׁ הָעֲשִׁירוּת בִּבְחִינַת כָּבוֹד וְכַמּוּבָן בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה בִּמְקוֹמוֹת אֲחֵרִים, כִּי הָעֲשִׁירוּת הִוא שֹׁרֶשׁ הַנֶּפֶשׁ בִּבְחִינַת וְאֵלָיו הוּא נוֹשֵֹא אֶת נַפְשׁוֹ, כַּמְבֹאָר בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה בְּמָקוֹם אַחֵר (סִימָן סח) כִּי הָעֲשִׁירוּת הִוא בִּבְחִינַת כָּבוֹד שֶׁשָּׁם עִקַּר שָׁרְשֵׁי הַנְּפָשׁוֹת בִּבְחִינַת בְּסֹדָם אַל תָּבֹא נַפְשִׁי בִּקְהָלָם אַל תֵּחַד כְּבֹדִי, כְּמוֹ שֶׁכָּתַב כָּל זֶה רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה בְּמָקוֹם אַחֵר. וְזֶה שֶׁנֶּאֱמַר בָּעֲשִׁירוּת שֶׁל יַעֲקֹב אָבִינוּ, וּמֵאֲשֶׁר לְאָבִינוּ עָשָֹה אֶת כָּל הַכָּבוֹד הַזֶּה, כִּי הָעֲשִׁירוּת הִוא בְּחִינַת כָּבוֹד כַּנַּ"ל. כִּי יַעֲקֹב כָּל הִשְׁתַּדְלוּתוֹ בְּבֵית לָבָן בְּעֵת שְׁמִירַת צֹאנוֹ הָיָה בִּשְׁבִיל הַעֲלָאַת הַכָּבוֹד לְשָׁרְשׁוֹ לְהַעֲלוֹת כָּבוֹד דִּקְדֻשָּׁה מִזִּילוּתָא דְּגָלוּתָא עַל יְדֵי הַמְשָׁכַת נַפְשׁוֹת יִשְֹרָאֵל וְהַגֵּרִים, כִּי עַל יְדֵי שְׁמִירַת הַצֹּאן זָכָה שֶׁיִּזְדַּוְּגוּ לוֹ שָׁם רָחֵל וְלֵאָה שֶׁהֵם שֹׁרֶשׁ נַפְשׁוֹת יִשְֹרָאֵל וּמֵהֶם נוֹלְדוּ לוֹ שִׁבְטֵי יִשְֹרָאֵל נִמְצָא שֶׁהִמְשִׁיךְ שָׁם בְּבֵית לָבָן נַפְשׁוֹת יִשְֹרָאֵל. גַּם הִמְשִׁיךְ נַפְשׁוֹת גֵּרִים, כִּי כָּל הָאָבוֹת הָיוּ מְגַיְּרֵי גֵּרִים, כְּמוֹ שֶׁכָּתַב רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה (סִימָן רכח) וְכַמּוּבָא בַּמִּדְרָשׁ (בְּרֵאשִׁית רַבָּה פֶּרֶק פד) וְכָל זֶה עָסַק יַעֲקֹב בְּעֵת שְׁמִירָתוֹ אֶת צֹאן לָבָן, כִּי הִשְׁתַּדֵּל לְהַמְשִׁיךְ שָׁלוֹם בָּעוֹלָם שֶׁעַל יְדֵי זֶה עִקַּר מְקוֹמוֹת. וּכְשֶׁיֵּשׁ שָׁלוֹם הַפְּרָטִי זֶה בְּחִינַת שְׁלֵמוּת הַדַּעַת שֶׁנִּתְתַּקֵּן הַדַּעַת וְנִשְׁלָם מִבְּחִינַת פְּגַם וְחֶסְרוֹן הַדַּעַת וַאֲזַי נִתְבַּטֵּל הַגְּנֵבָה וַאֲבֵדָה הַבָּאִין מִפְּגַם וְהִסְתַּלְּקוּת הַדַּעַת כַּנַּ"ל. אֲבָל הָאוֹנְסִין הֵם בָּאִים אֲפִלּוּ בַּיּוֹם אֲפִלּוּ כְּשֶׁדַּעְתּוֹ שְׁלֵמָה. וְאֵין נִתְבַּטְּלִין כִּי אִם עַל יְדֵי רִבּוּי הַשָּׁלוֹם רִבּוּי הַדַּעַת כְּמוֹ שֶׁיִּהְיֶה לֶעָתִיד, כְּמוֹ שֶׁכָּתוּב, וּמָלְאָה הָאָרֶץ דֵּעָה כַּמַּיִם וְכוּ'. שֶׁאָז יִהְיֶה הַשָּׁלוֹם הַמֻּפְלָא בְּחִינַת וְגָר זְאֵב עִם כֶּבֶשֹ וְכוּ', שֶׁזֶּהוּ בְּחִינַת שָׁלוֹם הַכְּלָלִי וְאָז נִתְבַּטְּלִין כָּל הָאוֹנְסִין וְכָל הַהֶזֵּקוֹת שֶׁבָּעוֹלָם וְכַנַּ"ל. וְעַל כֵּן הַשּׁוֹאֵל חַיָּב אֲפִלּוּ בְּאוֹנְסִין, כִּי מֵאַחַר שֶׁחֲבֵרוֹ וִתֵּר נֶגְדּוֹ וְהִשְׁאִיל לוֹ בְּחִנָּם שֶׁזֶּהוּ בְּחִינַת שָׁלוֹם הַכְּלָלִי שֶׁעַל יְדֵי זֶה נִתְבַּטְּלִין כָּל הָאוֹנְסִין, עַל כֵּן הַשּׁוֹאֵל חַיָּב בְּאוֹנְסִין, כִּי בְּוַדַּאי הָאֹנֶס מִצַּד הַשּׁוֹאֵל וּבְגִירָא דִּילֵהּ, כִּי מִצַּד הַמַּשְׁאִיל אֵין שׁוּם הֶזֵּק וְנֵזֶק מֵאַחַר שֶׁוִּתֵּר נֶגְדּוֹ שֶׁזֶּהוּ בְּחִינַת שָׁלוֹם הַכְּלָלִי. אֲבָל שֹוֹכֵר, שֶׁזֶּה בְּחִינַת שָׁלוֹם הַפְּרָטִי, שֶׁעַל יְדֵי זֶה אֵין יְכוֹלִין לְהִשָּׁמֵר כִּי אִם מִגְּנֵבָה וַאֲבֵדָה הַתָּלוּי בְּחֶסְרוֹן הַדַּעַת אֲבָל לֹא מֵאוֹנְסִין כַּנַּ"ל. וְעַל כֵּן הַשּוֹכֵר חַיָּב בִּגְנֵבָה וַאֲבֵדָה וּפָטוּר בְּאוֹנְסִין, כִּי מֵאוֹנְסִין אֵינוֹ יָכוֹל לִשְׁמֹר, כִּי מַה לַּעֲשֹוֹת כְּנֶגֶד הָאֹנֶס כִּי אִם עַל יְדֵי בְּחִינַת שָׁלוֹם הַכְּלָלִי שֶׁעַל יְדֵי זֶה נִצּוֹלִין מִכָּל הָאוֹנְסִין, כִּי נִתְבַּטְּלִין כָּל מִינֵי הֶזֵּק כְּמוֹ שֶׁיִּהְיֶה לֶעָתִיד כַּנַּ"ל וְזֶהוּ רַק בְּמַשְׁאִיל לַחֲבֵרוֹ שֶׁהוּא בְּחִינַת וִתּוּר שֶׁזֶּהוּ בְּחִינַת שָׁלוֹם הַכְּלָלִי כַּנַּ"ל, אֲבָל הַשּוֹכֵר שֶׁהוּא בִּבְחִינַת שָׁלוֹם הַפְּרָטִי אֵינוֹ יָכוֹל לִשְׁמֹר מֵאֹנֶס וְעַל כֵּן הַשּוֹכֵר פָּטוּר מֵאוֹנְסִין כַּנַּ"ל:

## אות ה

וְזֶה בְּחִינַת בִּרְכַּת כֹּהֲנִים. 'יְבָרֶכְךָ ה'' בְּמָמוֹן 'וְיִשְׁמְרֶךָ' מִן הַמַּזִּיקִין (כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה) זֶה נַעֲשֶֹה עַל יְדֵי הַשָּׁלוֹם. שֶׁזֶּהוּ עִקַּר בִּרְכַּת כֹּהֲנִים שֶׁמְּבָרְכִין אֵצֶל שִֹים שָׁלוֹם וּמְסַיְּמִין 'וְיָשֵֹם לְךָ שָׁלוֹם'. כִּי עִקַּר הַשְּׁמִירָה עַל יְדֵי הַשָּׁלוֹם כַּנַּ"ל וְזֶה נִמְשָׁךְ עַל יְדֵי הַתּוֹרָה. וְזֶה בְּחִינַת 'יָאֵר ה' פָּנָיו אֵלֶיךָ', בְּחִינַת אוֹר הַפָּנִים שֶׁהוּא בְּחִינַת הַתּוֹרָה, כְּמוֹ שֶׁכָּתוּב, כִּי בְאוֹר פָּנֶיךָ נָתַתָּ לָנוּ תּוֹרַת חַיִּים וְאַהֲבַת חֶסֶד וּצְדָקָה וְכוּ'. בְּחִינַת תּוֹרַת חֶסֶד וְעַל יְדֵי אוֹר הַפָּנִים בְּחִינַת הַתּוֹרָה עַל יְדֵי זֶה מַעֲלִין הַכָּבוֹד וְזֶה בְּחִינַת וִיחֻנֶּךָּ בְּחִינַת חֵן וְכָבוֹד יִתֵּן ה' וְכוּ' וְעַל יְדֵי זֶה יִשָּא ה' וְכוּ' וְיָשֵֹם לְךָ שָׁלוֹם בְּחִינַת לְהַמְשִׁיךְ שָׁלוֹם עַל יְדֵי הַנַּ"ל וְעַל יְדֵי זֶה נִמְשָׁךְ הַשְּׁמִירָה בְּחִינַת יְבָרֶכְךָ ה' וְיִשְׁמְרֶךָ, כִּי עִקַּר הַשְּׁמִירָה עַל יְדֵי הַשָּׁלוֹם כַּנַּ"ל:

כִּי הַכֹּהֲנִים הֵם בְּחִינַת חֶסֶד וְעִקַּר הַתּוֹרָה הוּא אֶצְלָם, כְּמוֹ שֶׁכָּתוּב (דְּבָרִים לג), יוֹרוּ מִשְׁפָּטֶיךָ לְיַעֲקֹב וְתוֹרָתְךָ לְיִשְֹרָאֵל. וְזֶה בְּחִינַת תּוֹרַת חֶסֶד, כִּי הַכֹּהֵן הוּא אִישׁ הַחֶסֶד וְעַל כֵּן הָרַב שֶׁמְּלַמֵּד תּוֹרָתוֹ לַאֲחֵרִים שֶׁזֶּהוּ בְּחִינַת תּוֹרַת חֶסֶד, כְּמוֹ שֶׁמְּבֹאָר בַּמַּאֲמָר הַנַּ"ל הוּא נִקְרָא כֹּהֵן, כְּמוֹ שֶׁכָּתוּב, כִּי שִֹפְתֵי כֹהֵן יִשְׁמְרוּ דַעַת וְתוֹרָה יְבַקְשׁוּ מִפִּיהוּ כִּי מַלְאַךְ ה' צְבָאוֹת וְכוּ', (כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה מוֹעֵד קָטָן דַּף יז) כִּי הַכֹּהֵן הוּא בִּבְחִינַת תּוֹרַת חֶסֶד כַּנַּ"ל וְעַל יְדֵי תּוֹרַת חֶסֶד מַעֲלִין אֶת הַכָּבוֹד כַּנַּ"ל וְעַל כֵּן עִקַּר הַכָּבוֹד אֵצֶל הַכֹּהֲנִים, כְּמוֹ שֶׁכָּתוּב:

"וְקִדַּשְׁתּוֹ", לִפְתֹּחַ רִאשׁוֹן וּלְבָרֵךְ רִאשׁוֹן. כִּי לַכֹּהֲנִים מַגִּיעַ כָּל הַכָּבוֹד דִּקְדֻשָּׁה, כִּי הֵם מַעֲלִין אֶת הַכָּבוֹד עַל יְדֵי תּוֹרַת חֶסֶד כַּנַּ"ל וְעַל כֵּן הֵם יְכוֹלִין לְהַמְשִׁיךְ שָׁלוֹם כַּנַּ"ל. וְעַל כֵּן הַכֹּהֵן הוּא בִּבְחִינַת שָׁלוֹם, כְּמוֹ שֶׁכָּתוּב הִנְנִי נֹתֵן לוֹ אֶת בְּרִיתִי שָׁלוֹם וְהָיְתָה לוֹ כוּ' בְּרִית כְּהֻנַּת עוֹלָם וְעַל כֵּן הֵם מְבָרְכִין אֶת יִשְֹרָאֵל בְּשָׁלוֹם, כִּי הֵם מַמְשִׁיכִין שָׁלוֹם בָּעוֹלָם וְזֶה עִקַּר בִּרְכַּת כֹּהֲנִים אֵצֶל שִֹים שָׁלוֹם בְּחִינַת וְיָשֵֹם לְךָ שָׁלוֹם. וְעַל יְדֵי זֶה מַמְשִׁיכִין שְׁמִירָה עַל יְדֵי הַשָּׁלוֹם בְּחִינַת יְבָרֶכְךָ ה' וְיִשְׁמְרֶךָ עַל יְדֵי הַשָּׁלוֹם כַּנַּ"ל:

בָּרוּךְ ה' לְעוֹלָם, אָמֵן וְאָמֵן:

# שוכר ג

הַשּוֹכֵר אֶת הַפָּרָה וְהִשְׁאִילָהּ לְאַחֵר וְנֶאֶנְסָה, יְשַׁלֵּם הַשּׁוֹאֵל לַבְּעָלִים וּמְבֹאָר הַטַּעַם בַּמִּשְׁנָה. כֵּיצַד הַלָּה עוֹשֶֹה סְחוֹרָה בְּפָרָתוֹ שֶׁל חֲבֵרוֹ?:

## אות א

עַל פִּי הַתּוֹרָה 'וַיְהִי מִקֵּץ' הַמְדַבֵּר מִשְּׁמִירַת הַזִכָּרוֹן בַּלִּקּוּטֵי הָרִאשׁוֹן סִימָן נד, עַיֵּן שָׁם כָּל הַתּוֹרָה מֵרֹאשׁוֹ לְסוֹפוֹ:

וְהַכְּלָל כִּי צְרִיכִין לִשְׁמֹר מְאֹד אֶת הַזִכָּרוֹן, דְּהַיְנוּ לְאַדַּבְקָא מַחֲשַׁבְתָּא בְּעָלְמָא דְּאָתֵי לִזְכֹּר תָּמִיד בְּמַחֲשַׁבְתּוֹ אֶת עוֹלָם הַנִּצְחִי שֶׁהוּא עוֹלָם הַבָּא וְלֹא יִפֹּל לְשִׁכְחָה שֶׁהִוא מִיתַת הַלֵּב שֶׁלֹּא יִשְׁכַּח אֶת עוֹלָם הַבָּא כָּל יְמֵי חַיָּיו. וּשְׁמִירַת הַזִכָּרוֹן הוּא בִּכְלָל וּבִפְרָט וְכוּ', עַיֵּן שָׁם מַה שֶּׁמְּבֹאָר עַל פָּסוּק ה' אֱלֹקַי גָּדַלְתָּ מְאֹד וְכוּ'. וְאֵצֶל הֲמוֹן עַם נַעֲשֶֹה כָּל זֶה מִמֵּילָא עַל יְדֵי שֵׁנָה וְצִיצִית וּתְפִלִּין וּקְרִיאַת שְׁמַע וּתְפִלָּה וּמַשָּא וּמַתָּן, עַיֵּן שָׁם. וּמְבֹאָר שָׁם שֶׁעַל יְדֵי כָּל הַמַּשָּא וּמַתָּן נַעֲשֶֹה יִחוּד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל שֶׁהוּא בְּחִינַת נוֹטֶה שָׁמַיִם כַּיְרִיעָה, כִּי בְּכָל הַדְּבָרִים יֵשׁ נִיצוֹצוֹת הַקְּדוֹשִׁים וְעַל יְדֵי הַמַּשָּא וּמַתָּן נִתְחַבְּרִין וְנִתְעַלִּין הַנִּיצוֹצוֹת כְּפִי חֶלְקֵי הַנֶּפֶשׁ רוּחַ נְשָׁמָה שֶׁיֵּשׁ לְכָל אֶחָד וְעַל יְדֵי זֶה נַעֲשֶֹה יִחוּד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל, עַיֵּן שָׁם הֵיטֵב. וְלִשְׁמֹר הַזִכָּרוֹן צְרִיכִין לִשְׁמֹר עַצְמוֹ מֵרַע עַיִן וְכוּ' וְגַם צְרִיכִין לִשְׁמֹר עַצְמוֹ מִכֹּחַ הַמְדַמֶּה שֶׁהוּא בְּחִינַת פַּרְעֹה וְכוּ' וְכוּ', עַיֵּן שָׁם כָּל זֶה הֵיטֵב:

## אות ב

כִּי עִקַּר הַשְּׁמִירָה שֶׁל הָאָדָם וַחֲפָצָיו שֶׁיִּהְיוּ נִשְׁמָרִין מִכָּל הֶזֵּקוֹת הוּא עַל יְדֵי שְׁמִירַת הַזִכָּרוֹן שֶׁהוּא לְאַדַּבְקָא מַחֲשַׁבְתָּא בְּעָלְמָא דְּאָתֵי, דְּהַיְנוּ לִזְכֹּר תָּמִיד בַּה' יִתְבָּרַךְ שֶׁזֶּהוּ עִקַּר בְּחִינַת עָלְמָא דְּאָתֵי שֶׁאָז יִתְגַּלֶּה אֱלֹקוּתוֹ לְכָל בָּאֵי עוֹלָם, כַּמְבֹאָר בִּפְסוּקִים רַבִּים וְכַמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל. שֶׁצְּרִיכִין לִזְכֹּר בְּעָלְמָא דְּאָתֵי בִּכְלָל וּבִפְרָט. בִּכְלָל הוּא לִזְכֹּר בְּכָל יוֹם וָיוֹם תֵּכֶף בַּהֲקִיצוֹ מִשְּׁנָתוֹ יִזְכֹּר בְּעָלְמָא דְּאָתֵי שֶׁיֵּשׁ עוֹד עוֹלָם שֶׁעָתִיד לָבֹא שֶׁהוּא הָעִקָּר וְלֹא יִשְׁכַּח וְיַטְעֶה עַצְמוֹ חַס וְשָׁלוֹם, שֶׁזֶּה הָעוֹלָם הוּא עִקָּר חַס וְשָׁלוֹם. וְאַחַר כָּךְ צָרִיךְ לְהַמְשִׁיךְ זֶה הַזִכָּרוֹן בִּפְרָט, דְּהַיְנוּ לְהִסְתַּכֵּל וּלְהָבִין בְּכָל הַמַּחֲשָׁבוֹת וְדִבּוּרִים וַעֲשִֹיּוֹת שֶׁה' יִתְבָּרַךְ מַזְמִין לוֹ בְּכָל יוֹם לְהָבִין מֵהֶם רְמִיזוֹתָיו יִתְבָּרַךְ לְהִתְקַשֵּׁר לַעֲבוֹדָתוֹ יִתְבָּרַךְ כִּי ה' יִתְבָּרַךְ מְצַמְצֵם עַצְמוֹ מֵאֵין סוֹף עַד אֵין תַּכְלִית וּמַזְמִין לְכָל אֶחָד וְאֶחָד מַחֲשָׁבָה דִּבּוּר וּמַעֲשֶֹה בְּכָל יוֹם לְפִי הַזְּמַן וְהַמָּקוֹם וְהָאָדָם, וּמְרַמֵּז לוֹ בָּהֶם רְמָזִים לְקָרְבוֹ לַעֲבוֹדָתוֹ יִתְבָּרַךְ וְכוּ', עַיֵּן שָׁם. וְצָרִיךְ הָאָדָם לְהַגְדִּיל דַּעְתּוֹ וּלְהָבִין אֵלּוּ הָרְמָזִים בְּחִינַת לְאַדַּבְקָא מַחֲשַׁבְתָּא בְּעָלְמָא דְּאָתֵי שֶׁאָז יִתְגַּלֶּה אֱלֹקוּתוֹ הַמְלֻבָּשׁ בְּכָל דָּבָר. וְעַל כֵּן עַל יְדֵי שֶׁשּׁוֹמְרִין אֶת הַזִכָּרוֹן הַזֶּה עַל יְדֵי זֶה נִשְׁמָרִין מִכָּל מִינֵי הֶזֵּקוֹת שֶׁבָּעוֹלָם, כִּי עִקַּר הַשְּׁמִירָה הוּא רַק מֵה' יִתְבָּרַךְ, כְּמוֹ שֶׁכָּתוּב (תְּהִלִּים קכז), אִם ה' לֹא יִשְׁמָר עִיר שָׁוְא שָׁקַד שׁוֹמֵר. וּכְמוֹ שֶׁכָּתוּב (שָׁם קכא) ה' שֹׁמְרֶךָ וְכוּ' וְכַמְבֹאָר כְּבָר כַּמָּה פְּעָמִים וְעַל כֵּן כְּשֶׁזּוֹכֵר בַּה' יִתְבָּרַךְ וּמַמְשִׁיךְ אֱלֹקוּתוֹ בְּכָל דָּבָר עַל יְדֵי זֶה נִשְׁמָר הַדָּבָר מִכָּל רַע, כְּמוֹ שֶׁכָּתוּב ה' יִשְׁמָרְךָ מִכָּל רָע כִּי בְּכָל מָקוֹם שֶׁה' יִתְבָּרַךְ נִמְצָא שָׁם, שָׁם כָּל הַבְּרָכוֹת וְכָל טוֹב וְנִשְׁמָרִין מִכָּל רַע וְעִקַּר הַמְשָׁכַת אֱלֹקוּתוֹ יִתְבָּרַךְ בְּכָל דָּבָר הוּא עַל יְדֵי הַזִכָּרוֹן הַנַּ"ל עַל יְדֵי שֶׁזּוֹכְרִין שֶׁה' יִתְבָּרַךְ בְּכָל דָּבָר. כִּי בֶּאֱמֶת ה' יִתְבָּרַךְ בְּכָל מָקוֹם וּבְכָל דָּבָר רַק כְּשֶׁאֵין זוֹכְרִין אוֹתוֹ חַס וְשָׁלוֹם, אֲזַי הוּא גַּם כֵּן מִתְעַלֵּם וּמַסְתִּיר פָּנָיו בִּבְחִינַת וְאָנֹכִי הַסְתֵּר אַסְתִּיר (דְּבָרִים לא) וַאֲזַי יֵשׁ אֲחִיזָה וּשְׁלִיטָה לְהַקְּלִפּוֹת וּמַזִּיקֵי עָלְמָא חַס וְשָׁלוֹם מֵחֲמַת שֶׁאֱלֹקוּתוֹ נֶעְלָם וְנִסְתָּר שָׁם. אֲבָל תֵּכֶף כְּשֶׁזּוֹכְרִין אוֹתוֹ יִתְבָּרַךְ דְּהַיְנוּ שֶׁמַּאֲמִינִין וְיוֹדְעִין שֶׁאֱלֹקוּתוֹ יִתְבָּרַךְ מְלֻבָּשׁ שָׁם אֲזַי תֵּכֶף נִתְגַּלֶּה אֱלֹקוּתוֹ יִתְבָּרַךְ וּכְשֶׁנִּתְגַּלֶּה אֱלֹקוּתוֹ יִתְבָּרַךְ שָׁם אֲזַי נִמְשָׁךְ כָּל הַבְּרָכוֹת וְכָל טוֹב וְנִשְׁמָרִין מִכָּל רַע בִּבְחִינַת בְּכָל הַמָּקוֹם אֲשֶׁר אַזְכִּיר אֶת שְׁמִי אָבוֹא אֵלֶיךָ וּבֵרַכְתִּיךָ, אַזְכִּיר דַּיְקָא כְּמוֹ שֶׁכָּתוּב יְבָרֶכְךָ ה' וְיִשְׁמְרֶךָ כְּמוֹ שֶׁדָּרְשׁוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה בָּשָֹר וָדָם נוֹתֵן מַתָּנָה לַחֲבֵרוֹ וְאֵינוֹ יָכוֹל לְשָׁמְרוֹ וְכוּ' אֲבָל הַקָּדוֹשׁ בָּרוּךְ הוּא מְבָרֵךְ אֶת הָאָדָם וְשׁוֹמְרוֹ וְכוּ':

נִמְצָא שֶׁעִקַּר הַשְּׁמִירָה עַל יְדֵי הַזִכָּרוֹן וְזֶהוּ בְּחִינַת זָכוֹר וְשָׁמוֹר בְּדִבּוּר אֶחָד כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כִּי זָכוֹר וְשָׁמוֹר הֵם בְּחִינָה אַחַת כִּי הַשְּׁמִירָה הִוא עַל יְדֵי הַזִכָּרוֹן כַּנַּ"ל. וְעִקַּר שְׁמִירַת הַזִכָּרוֹן הוּא עַל יְדֵי בְּחִינַת יוֹסֵף הַצַּדִּיק בְּחִינַת וְיוֹסֵף יָשִׁית יָדוֹ עַל עֵינֶיךָ כַּמְבֹאָר שָׁם הֵיטֵב כִּי שְׁלֵמוּת הַזִכָּרוֹן הוּא כְּשֶׁזּוֹכִין לְזִכָּרוֹן בִּכְלָל וּבִפְרָט בְּכָל הַבְּחִינוֹת וְהַתִּקּוּנִים הַנֶּאֱמָרִין שָׁם בְּהַתּוֹרָה הַנּוֹרָאָה הַנַּ"ל הַמְרֻמָּזִין בַּפָּסוּק ה' אֱלֹקַי גָּדַלְתָּ מְאֹד הוֹד וְהָדָר לָבָשְׁתָּ עֹטֶה אוֹר כַּשַּלְמָה נוֹטֶה שָׁמַיִם כַּיְרִיעָה. וּגְמַר הַתִּקּוּן הוּא בְּחִינַת נוֹטֶה שָׁמַיִם כַּיְרִיעָה שֶׁהוּא בְּחִינַת יִחוּד קֻדְשָׁא בְּרִיךְ הוּא וּשְׁכִינְתֵּהּ שֶׁהוּא בְּחִינַת יִחוּד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל שֶׁנַּעֲשֶֹה עַל יְדֵי הַמַּשָּא וּמַתָּן כַּמְבֹאָר שָׁם עַיֵּן שָׁם. כִּי בְּכָל דָּבָר יֵשׁ נִיצוֹצוֹת הַקְּדוֹשִׁים שֶׁנִּשְׁלָמִין עַל יְדֵי הַמַּשָּא וּמַתָּן שֶׁעַל יְדֵי זֶה נִתְיַחֵד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל וְכוּ' עַיֵּן שָׁם. וְצָרִיךְ כָּל אֶחָד לְהַאֲמִין בְּשָׁעָה שֶׁעוֹשֶֹה הַמַּשָּא וּמַתָּן שֶׁיֵּשׁ נִיצוֹצוֹת קְדוֹשִׁים הַרְבֵּה בְּאֵלּוּ הַדְּבָרִים שֶׁנּוֹשֵֹא וְנוֹתֵן בָּהֶם וְכָל כַּוָּנָתוֹ יִהְיֶה בִּשְׁבִיל בֵּרוּר הַנִּיצוֹצוֹת שֶׁזֶּהוּ בְּחִינַת זִכָּרוֹן שֶׁעַל יְדֵי זֶה נִתְיַחֵד הַצַּדִּיק וּכְנֶסֶת יִשְֹרָאֵל כַּנַּ"ל. וְאָז כְּשֶׁשּׁוֹמֵר אֶת הַזִכָּרוֹן וְזוֹכֵר בִּשְׁעַת הַמַּשָּא וּמַתָּן אֶת ה' יִתְבָּרַךְ שֶׁעַל יְדֵי זֶה זוֹכֶה לְיַחֵד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל עַל יְדֵי הַמַּשָּא וּמַתָּן עַל יְדֵי זֶה נִמְשָׁכִין כָּל הַהַשְׁפָּעוֹת וְהַבְּרָכוֹת וְנִשְׁמָר מִכָּל הֶזֵּקוֹת כַּנַּ"ל וְזֶהוּ בְּחִינַת יוֹסֵף הַצַּדִּיק שֶׁנֶּאֱמַר בּוֹ וְכָל אֲשֶׁר הוּא עוֹשֶֹה ה' מַצְלִיחַ כִּי יוֹסֵף הוּא הַשַּׁלִּיט הוּא הַמַּשְׁבִּיר לְכָל עַם הָאָרֶץ (בְּרֵאשִׁית מב) כִּי יוֹסֵף הוּא הַצַּדִּיק שֶׁעִקַּר הַיִּחוּד עַל יָדוֹ כִּי הוּא בְּחִינַת שְׁמִירַת הַזִכָּרוֹן שֶׁהוּא מַמְשִׁיךְ עַל יְדֵי זֶה כָּל הַהַשְׁפָּעוֹת וְהַבְּרָכָה וְהַשְּׁמִירָה כַּנַּ"ל וְזֶהוּ בְּחִינַת בֵּן פֹּרָת יוֹסֵף בֵּן פֹּרָת עֲלֵי עָיִן פֹּרָת לְשׁוֹן פְּרוּ וּרְבוּ כְּמוֹ שֶׁפֵּרֵשׁ רַשִׁ"י בְּחִינַת יִחוּד שֶׁעַל יָדוֹ נִפְרִין וְנִתְבָּרְכִין וְנִשְׁפָּעִין כָּל הַדְּבָרִים שֶׁבָּעוֹלָם כַּנַּ"ל וְנִשְׁמָרִין מִכָּל רַע כַּנַּ"ל:

## אות ג

נִמְצָא שֶׁעִקַּר שְׁמִירַת הַחֲפָצִים שֶׁל כָּל אֶחָד הוּא כְּפִי שְׁמִירַת הַזִכָּרוֹן שֶׁהוּא בְּחִינַת יוֹסֵף בְּחִינַת בֵּן פֹּרָת יוֹסֵף שֶׁעַל יָדוֹ נִתְיַחֵד בְּחִינַת צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל שֶׁעַל יְדֵי זֶה כָּל הַבְּרָכוֹת וְהַשְּׁמִירָה כַּנַּ"ל. וְזֶה מְרֻמָּז בַּמִּשְׁנָה הַשּוֹכֵר אֶת הַפָּרָה וְהִשְׁאִילָהּ לְאַחֵר וְנֶאֶנְסָה יְשַׁלֵּם לַבְּעָלִים כְּמוֹ שֶׁאָמַר רַבִּי יְהוּדָה כֵּיצַד הַלָּה עוֹשֶֹה סְחוֹרָה בְּפָרָתוֹ שֶׁל חֲבֵרוֹ, בְּפָרָתוֹ דַּיְקָא בְּחִינַת בֵּן פֹּרָת יוֹסֵף כִּי כָּל הַמַּשָּא וּמַתָּן וְהַסְּחוֹרוֹת וְהַשְּׁמִירָה הַכֹּל כְּפִי בְּחִינַת פֹּרָת יוֹסֵף שֶׁזּוֹכֶה כָּל אֶחָד וְאֶחָד לְפִי בְּחִינָתוֹ כְּפִי בְּחִינַת שְׁמִירַת הַזִכָּרוֹן שֶׁזּוֹכֶה שֶׁעַל יְדֵי זֶה נִתְיַחֵד הַצַּדִּיק וּכְנֶסֶת יִשְֹרָאֵל עַל יְדֵי הַמַּשָּא וּמַתָּן שֶׁהוּא בְּחִינַת בֵּן פֹּרָת יוֹסֵף וּכְפִי הַיִּחוּד שֶׁגּוֹרֵם עַל יְדֵי שְׁלֵמוּת הַזִכָּרוֹן שֶׁלּוֹ כְּמוֹ כֵן זוֹכֶה לְהַרְוִיחַ וּלְהִשְׁתַּכֵּר בְּהַסְּחוֹרָה וְהַמַּשָּא וּמַתָּן וְלִהְיוֹת נִשְׁמָר מִכָּל רַע וְזֶהוּ כֵּיצַד הַלָּה עוֹשֶֹה סְחוֹרָה בְּפָרָתוֹ שֶׁל חֲבֵרוֹ בְּפָרָתוֹ דַּיְקָא בְּחִינַת פֹּרָת יוֹסֵף שֶׁעַל יְדֵי זֶה כָּל הָרֶוַח וְהַשְּׁמִירָה כַּנַּ"ל:

וְעוֹד יֵשׁ בָּזֶה דְּבָרִים הַרְבֵּה אַךְ נִשְׁכַּח כִּי לֹא נִכְתַּב בִּזְמַנּוֹ:

## אות ד

וְזֶה בְּחִינַת חֲכִירוּת וְקַבְּלָנוּת וְזֶה בְּחִינַת שְֹכִירוּת פּוֹעֲלִים. בְּחִינַת לֹא תָלִין פְּעֻלַּת שָֹכִיר אִתְּךָ עַד בֹּקֶר. וְזֶה בְּחִינַת פָּרָשַׁת פָּרָה אֲדֻמָּה שֶׁקּוֹרִין אַחַר פּוּרִים קֹדֶם פֶּסַח:

כִּי הַכְּלָל הוּא שֶׁהָעִקָּר לִשְׁמֹר אֶת הַזִכָּרוֹן הַזֶּה הֵיטֵב לִזְכֹּר תָּמִיד בְּכָל יוֹם בְּעָלְמָא דְּאָתֵי שֶׁהוּא אַחְדוּתוֹ יִתְבָּרַךְ כִּי אָז יִהְיֶה ה' אֶחָד וּשְׁמוֹ אֶחָד וְכָל הָעוֹלָם הַזֶּה וְכָל חֲפָצָיו וְתַאֲוֹתָיו וְדֵעוֹתָיו וְכוּ' הַכֹּל יִתְבַּטֵּל וְלֹא יִהְיֶה נִשְׁאַר כִּי אִם ה' יִתְבָּרַךְ לְבַדּוֹ. וְכָל מִי שֶׁיִּזְכֶּה לְהִכָּלֵל בּוֹ כְּפִי מַה שֶּׁשָּׁמַר אֶת הַזִכָּרוֹן הַזֶּה בָּעוֹלָם הַזֶּה וְהִתְרַחֵק אֶת עַצְמוֹ מִתַּאֲוֹת וְהִסְתַּפֵּק אֶת עַצְמוֹ בְּתַכְלִית הַצִּמְצוּם מִכָּל עִנְיְנֵי עוֹלָם הַזֶּה הַמֻּכְרָחִים עַל פִּי הַתּוֹרָה. וְהִנֵּה עִקַּר הַשִּׁכְחָה שֶׁהוּא בְּחִינַת סִטְרָא דְּמוֹתָא בְּחִינַת מִיתַת הַלֵּב כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל נִמְשָׁךְ רַק מֵחֵטְא אָדָם הָרִאשׁוֹן שֶׁעַל יְדֵי זֶה נִגְזַר מִיתָה עַל הָעוֹלָם כְּמוֹ שֶׁכָּתוּב כִּי בְּיוֹם אֲכָלְךָ מִמֶּנּוּ וְכוּ' כִּי מִיתָה הִיא בְּחִינַת שִׁכְחָה כַּנַּ"ל כִּי אָדָם הָרִאשׁוֹן פָּגַם בַּזִּכָּרוֹן עַל יְדֵי בְּחִינַת רַע עַיִן שֶׁהוּא הַנָּחָשׁ הַקַּדְמוֹנִי שֶׁהוּא עִקַּר הָרַע עַיִן בְּעֶצֶם שֶׁנִּתְקַנֵּא בָּהֶם עַל שֶׁרָאָה אוֹתָם עֲרֻמִּים וְכוּ' וְהָיְתָה עֵינוֹ רָעָה בָּהֶם וּכְמוֹ שֶׁפֵּרֵשׁ רַשִׁ"י שָׁם בָּעִנְיָן וְעַל יְדֵי עֵינוֹ הָרָעָה הִתְגָּרָה בָּהֶם וְהִמְשִׁיךְ עֲלֵיהֶם בְּחִינַת רַע עַיִן בִּבְחִינַת וַתֵּרֶא הָאִשָּׁה כִּי טוֹב הָעֵץ וְכוּ' וְכִי תַּאֲוָה הוּא לָעֵינַיִם וְכוּ' שֶׁזֶּהוּ בְּחִינַת פְּגַם הָעֵינַיִם בְּחִינַת רַע עַיִן שֶׁפּוֹגֵם הַזִכָּרוֹן וְזֶהוּ בְּחִינַת וַתִּקַּח מִפִּרְיוֹ וְכוּ' וַתִּקַּח זֶהוּ בְּחִינַת פְּגַם הַיָּד שֶׁפּוֹגֵם גַּם כֵּן הַזִכָּרוֹן כִּי שְׁמִירַת הַזִכָּרוֹן הוּא עַל יְדֵי שְׁנֵי בְּחִינוֹת עַל יְדֵי בְּחִינַת יָד שֶׁמַּכְנִיעַ הַמְדַמֶּה וְעַל יְדֵי הַכְנָעַת הָרַע עַיִן כַּמְבֹאָר הֵיטֵב בְּהַתּוֹרָה הַנַּ"ל וְהִיא פָּגְמָה בִּשְׁנֵיהֶם בִּבְחִינַת עֵינַיִם וּבִבְחִינַת יָד בְּחִינַת וַתֵּרֶא וַתִּקַּח כַּנַּ"ל וְעַל יְדֵי זֶה נִפְגַּם הַזִכָּרוֹן שֶׁל עָלְמָא דְּאָתֵי כִּי אָדָם הָרִאשׁוֹן כָּפַר בָּעִקָּר כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה וּכְשֶׁכָּפַר בּוֹ יִתְבָּרַךְ בְּוַדַּאי כָּפַר בְּעָלְמָא דְּאָתֵי כִּי הַכֹּל אֶחָד כַּנַּ"ל וְעַל כֵּן נִגְזַר עָלָיו מִיתָה שֶׁהִוא בְּחִינַת שִׁכְחָה שֶׁהִוא מִיתַת הַלֵּב כַּנַּ"ל וְעַל יְדֵי זֶה נִגְזַר עָלָיו בְּעִצָּבוֹן תֹּאכֲלֶנָּה בְּזֵעַת אַפֶּךָ תֹּאכַל לֶחֶם שֶׁהִיא בְּחִינַת עַבְדוּת וִיגִיעוֹת שֶׁצְּרִיכִין לְהִתְיַגֵּעַ מְאֹד בְּעַבְדוּת גָּדוֹל קֹדֶם שֶׁמַּמְשִׁיכִין פַּרְנָסָה כִּי הָעַבְדוּת הִוא בְּחִינַת שִׁכְחָה בְּחִינַת שֵׁנָה שֶׁהִיא אֶחָד מִשִּׁשִּׁים בְּמִיתָה שֶׁהִוא בְּחִינַת שִׁכְחָה כַּנַּ"ל כִּי שֵׁנָה הִוא הִסְתַּלְּקוּת הַדַּעַת שֶׁהוּא בְּחִינַת זִכָּרוֹן כִּי אָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה (קִדּוּשִׁין מט) עֲשָֹרָה קַבִּין שֵׁנָה יָרְדוּ לָעוֹלָם תִּשְׁעָה נָטְלוּ עֲבָדִים וְכוּ' נִמְצָא שֶׁעַבְדוּת הִוא ## אות ו

וְזֶה בְּחִינַת חֲכִירוּת וְקַבְּלָנוּת שֶׁכָּל הַהֶזֵּקוֹת שֶׁל הַשָּדֶה חָלִין עַל הַחוֹכֵר כַּמְבֹאָר בַּשֻּׁלְחָן עָרוּךְ הַחוֹכֵר שָֹדֶה וְנִשְׁתַּדְּפָה וְכוּ' כִּי עִקַּר הַבְּרָכָה בְּכָל דָּבָר וּבִפְרָט בַּעֲבוֹדַת הַשָּדֶה הוּא כְּפִי שְׁמִירַת הַזִכָּרוֹן שֶׁעַל יְדֵי זֶה מַמְשִׁיכִין וּמְגַלִּין אֱלֹקוּתוֹ שָׁם שֶׁעַל יְדֵי זֶה נִמְשָׁךְ בְּרָכָה בִּבְחִינַת בְּכָל הַמָּקוֹם אֲשֶׁר אַזְכִּיר אֶת שְׁמִי אָבוֹא אֵלֶיךָ וּבֵרַכְתִּיךָ וְכֵן לְהֵפֶךְ כָּל הַהֶזֵּקוֹת שֶׁבַּשָּדֶה הוּא כְּפִי פְּגַם הַזִכָּרוֹן שֶׁהוּא פְּגַם חֵטְא אָדָם הָרִאשׁוֹן שֶׁעַל יְדֵי זֶה נִגְזַר אֲרוּרָה הָאֲדָמָה וְכוּ' בְּזֵעַת אַפֶּךָ וְכוּ' (בְּרֵאשִׁית ג) שֶׁעַל יְדֵי זֶה נִמְשָׁכִין כָּל הַהֶזֵּקוֹת וְהַהֶפְסֵד שֶׁבַּשָּדֶה כְּשֶׁאֵין זוֹכִין לְהַכְנִיעַ וּלְבַטֵּל זֻהֲמַת הַנָּחָשׁ מֵהַשָּדֶה עַל יְדֵי שְׁמִירַת הַזִכָּרוֹן שֶׁהוּא עִקַּר תִּקּוּן חֵטְא אָדָם הָרִאשׁוֹן כַּנַּ"ל וְעַל כֵּן הַחוֹכֵר שֶׁמְּקַבֵּל עָלָיו עֲבוֹדַת הַשָּדֶה וְנָפַל לִבְחִינַת עַבְדוּת כִּי הוּא צָרִיךְ לַעֲשֹוֹת עֲבוֹדַת חֲבֵרוֹ כְּעֶבֶד וְעַל כֵּן הוּא צָרִיךְ לִשְׁמֹר עַצְמוֹ בְּיוֹתֵר מִפְּגַם הַזִכָּרוֹן כִּי כָּל מַה שֶּׁנִּמְשָׁךְ הָעַבְדוּת וְהָעֲשִֹיָּה בְּיוֹתֵר שָׁם נִמְשָׁךְ הַשִּׁכְחָה בְּיוֹתֵר כִּי עִקַּר הַשִּׁכְחָה הוּא בְּמָקוֹם שֶׁרָחוֹק יוֹתֵר מֵרוּחָנִיּוּת וּכְמוֹ שֶׁכָּתַב רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה עַל פָּסוּק לֹא יָמוּשׁ סֵפֶר הַתּוֹרָה הַזֶּה מִפִּיךָ שֶׁבְּמָקוֹם שֶׁיֵּשׁ מַמָּשׁוּת וְגַשְׁמִיּוּת בְּיוֹתֵר שָׁם שִׁכְחָה בְּיוֹתֵר עַיֵּן שָׁם וּבִפְרָט בִּבְחִינַת עַבְדוּת שֶׁהִוא בְּחִינַת שֵׁנָה בְּחִינַת שִׁכְחָה כִּי עֲשָֹרָה קַבִּין שֵׁנָה יָרְדוּ לָעוֹלָם תִּשְׁעָה נָטְלוּ עֲבָדִים וְעַל כֵּן תּוֹלִין קִלְקוּל הַשָּדֶה בְּהַחוֹכֵר שֶׁהוּא בִּבְחִינַת עֲשִֹיָּה וְעַבְדוּת כִּי עִקַּר קִלְקוּל הַשָּדֶה הוּא עַל יְדֵי פְּגַם הַזִכָּרוֹן שֶׁהוּא נֶאֱחָז בְּיוֹתֵר בִּבְחִינַת עֲשִֹיָּה וְעַבְדוּת כַּנַּ"ל:

## אות ח

וְעַל כֵּן הִזְהִירָה הַתּוֹרָה לִתֵּן הַשְּכִירוּת לְהַשָּכִיר בְּמוֹעֲדוֹ וּבִזְמַנּוֹ וְשֶׁלֹּא לְאַחֵר שְֹכָרוֹ כְּמוֹ שֶׁכָּתוּב (דְּבָרִים כד) בְּיוֹמוֹ תִּתֵּן שְֹכָרוֹ וְכוּ' כִּי עָנִי הוּא וְאֵלָיו הוּא נוֹשֵֹא אֶת נַפְשׁוֹ כִּי תֵּכֶף כְּשֶׁמַּשְׁלִים פְּעֻלָּתוֹ וְיוֹצֵא מִן הָעַבְדוּת שֶׁהִוא בְּחִינַת שִׁכְחָה צְרִיכִין לִתֵּן לוֹ תֵּכֶף שְֹכָרוֹ שֶׁפָּעַל שֶׁהוּא בְּחִינַת בֵּרוּר הַנִּיצוֹצוֹת בְּחִינַת תִּקּוּן נַפְשׁוֹ שֶׁהוּא בְּחִינַת תִּקּוּן הַזִכָּרוֹן שֶׁעַל יְדֵי זֶה מְתַקֵּן מִיָּד פְּגַם הַשִּׁכְחָה שֶׁיָּרַד לְשָׁם עַל יְדֵי בְּחִינַת הָעַבְדוּת כַּנַּ"ל. וְעַל כֵּן הִזְהִירָה הַתּוֹרָה מְאֹד לָתֵת שְֹכָרוֹ בְּיוֹמוֹ דַּיְקָא שֶׁלֹּא לְאַחֵר מִיּוֹם לַחֲבֵרוֹ כִּי תִּקּוּן הַזִכָּרוֹן צְרִיכִין לִזָּהֵר מְאֹד בְּכָל יוֹם וָיוֹם, לִשְׁמֹר אֶת הַזִכָּרוֹן בְּכָל יוֹם דַּיְקָא דְּהַיְנוּ לְהַזְכִּיר אֶת עַצְמוֹ בְּכָל יוֹם וָיוֹם בְּעָלְמָא דְּאָתֵי. כִּי עִקַּר הַשִּׁכְחָה נִמְשָׁךְ עַל יְדֵי רִבּוּי הַזְּמַן כַּנִּרְאֶה בְּחוּשׁ שֶׁעַל יְדֵי אֲרִיכַת הַזְּמַן נִשְׁכַּח הַדָּבָר כְּמוֹ שֶׁכָּתוּב קֹהֶלֶת בִּרְבוֹת הַיָּמִים הַכֹּל נִשְׁכָּח כִּי עִקַּר בְּחִינַת הַשִּׁכְחָה נֶאֱחָז מִבְּחִינַת הַשִּׁנּוּיִים שֶׁיֵּשׁ בָּעוֹלָם בְּכָל יוֹם וּבְכָל עֵת כִּי ה' יִתְבָּרַךְ עוֹשֶֹה נִפְלָאוֹת בְּכָל עֵת וּבְטוּבוֹ מְחַדֵּשׁ בְּכָל יוֹם תָּמִיד מַעֲשֵֹה בְּרֵאשִׁית וּבְכָל יוֹם עוֹלָה הַשֶּׁמֶשׁ בַּבֹּקֶר וְשׁוֹקֵעַ בָּעֶרֶב וְאֵין יוֹם דּוֹמֶה לַחֲבֵרוֹ כִּי יֵשׁ שִׁנּוּיִים גְּדוֹלִים בְּכָל יוֹם וָיוֹם כִּי בְּכָל יוֹם וָיוֹם מִתְבָּרְרִים וְעוֹלִים נִיצוֹצוֹת קְדוֹשִׁים שֶׁלֹּא נִתְבָּרְרוּ עֲדַיִן מִימוֹת עוֹלָם כַּמּוּבָן בַּכַּוָּנוֹת. וְגַם בְּגַשְׁמִיּוּת רוֹאִין בְּחוּשׁ מִי שֶׁמֵּבִין קְצָת שֶׁיֵּשׁ שִׁנּוּיִים בְּכָל יוֹם וְאֵין שׁוּם יוֹם דּוֹמֶה לַחֲבֵרוֹ הֵן בְּעֶצֶם הַיּוֹם שֶׁבְּזֶה הַיּוֹם קַר בְּיוֹתֵר וּבְזֶה הַיּוֹם עָנָן וְגֶשֶׁם וְכַיּוֹצֵא בָּזֶה שִׁנּוּיִים רַבִּים בְּעִנְיַן הָאֲוִיר וְהָרוּחַ וּשְׁאָר הַהִתְהַוּוּת שֶׁבִּזְמַן שֶׁיֵּשׁ בָּהֶם שִׁנּוּיִים רַבִּים בְּכָל יוֹם וָיוֹם כִּי ה' יִתְבָּרַךְ מְשַׁנֶּה עִתִּים בִּתְבוּנָתוֹ. וְכֵן בִּמְאֹרְעוֹת הָאָדָם וּתְנוּעוֹתָיו וּמַחְשְׁבוֹתָיו וְהַנְהָגוֹתָיו יֵשׁ בָּהֶם שִׁנּוּיִים רַבִּים בְּכָל יוֹם וָיוֹם מִי שֶׁמִּסְתַּכֵּל עֲלֵיהֶם. וּמֵרִבּוּי הַשִּׁנּוּיִים הָאֵלֶּה שֶׁבְּכָל יוֹם נִמְשָׁךְ הַשִּׁכְחָה כִּי הֵם מַטְרִידִין אֶת הָאָדָם מִלִּזְכֹּר בְּעָלְמָא דְּאָתֵי וְכַנִּרְאֶה גַּם בְּחוּשׁ קְצָת שֶׁבְּכָל יוֹם וְעֵת הָעוֹלָם מְסַפְּרִים מֵהַשִּׁנּוּיִים שֶׁהַיּוֹם קַר בְּיוֹתֵר וְהַיּוֹם חַם וְהַיּוֹם צְרִיכִין מָטָר וְהַיּוֹם רִבּוּי מָטָר וְכַיּוֹצֵא בָּזֶה וּבָעִתִּים הַלָּלוּ הַפַּרְנָסָה בְּדֹחַק. וּבִפְרָט הַשִּׁנּוּיִים שֶׁיֵּשׁ לְכָל אֶחָד בִּפְרָטִיּוּת שֶׁזֶּה מַטְרִיד אוֹתוֹ הָעֲשִׁירוּת וְהַמַּשָּא וּמַתָּן וְזֶה מַטְרִיד אוֹתוֹ הָעֲנִיּוּת וְהַדַּחֲקוּת וְזֶה טָרוּד לְגַדֵּל בָּנָיו וּבְנוֹתָיו הַרְבֵּה שֶׁיֵּשׁ לוֹ צַעַר גִּדּוּל בָּנִים וְזֶה הוּא חֲשֹוּכֵי בָּנִים וְזֶה מְסַפֵּר מִמְּאֹרְעוֹת הַמִּלְחָמוֹת שֶׁנִּשְׁמָע עַכְשָׁו בָּעוֹלָם וְכַיּוֹצֵא בָּזֶה שִׁנּוּיִים וּפְרָטִים רַבִּים לְאֵין תַּכְלִית שֶׁמַּטְרִידִין אֶת הָאָדָם בְּכָל יוֹם עַד שֶׁנִּדְמֶה לוֹ בְּכָל יוֹם כְּאִלּוּ זֶה הַיּוֹם אֵינוֹ נֶחְשָׁב לוֹ לִזְמַן חִיּוּתוֹ וְזֶה הַיּוֹם קָשֶׁה לוֹ לַעֲבֹד אֶת ה' יִתְבָּרַךְ בִּשְׁלֵמוּת מֵחֲמַת הַשִּׁנּוּי וְהַבִּלְבּוּל שֶׁבְּזֶה הַיּוֹם:

הַשָּׁעָה דַּיְקָא, וְהוּא מַזְמִין לוֹ מַחֲשָׁבָה דִּבּוּר וּמַעֲשֶֹה בְּכָל יוֹם וָיוֹם לְפִי הַיּוֹם וּלְפִי הַמָּקוֹם וּלְפִי הָאָדָם וּמְרַמֵּז לוֹ בָּהֶם דַּיְקָא רְמָזִים לְהִתְקָרֵב לַעֲבוֹדָתוֹ, וּמִי שֶׁזּוֹכֶה לְהַגְדִּיל דַּעְתּוֹ וּלְהָבִין בְּכָל יוֹם וָיוֹם הָרְמָזִים הָאֵלֶּה שֶׁזֶּהוּ בְּחִינַת שְׁמִירַת הַזִכָּרוֹן כַּנַּ"ל אוֹ אֲפִלּוּ כְּשֶׁאֵין לוֹ זֶה הַשֵּכֶל שְׁלֵמָה וַאֲזַי יוּכַל לִזְכֹּר אֶת ה' בְּכָל מָקוֹם וּבְכָל זְמַן וּבְכָל הַמַּדְרֵגוֹת שֶׁבָּעוֹלָם לְמַעְלָה אוֹ לְמַטָּה חַס וְשָׁלוֹם כִּי ה' יִתְבָּרַךְ מְצַמְצֵם עַצְמוֹ בְּכָל יוֹם בְּכָל ## אות ט

נִמְצָא שֶׁצְּרִיכִין לִשְׁמֹר אֶת הַזִכָּרוֹן בְּכָל יוֹם דַּיְקָא וְזֶהוּ בְּחִינַת בְּיוֹמוֹ תִּתֵּן שְֹכָרוֹ וְלֹא תָּבֹא עָלָיו הַשֶּׁמֶשׁ. כִּי שְֹכַר פְּעֻלָּתוֹ הוּא בְּחִינַת תִּקּוּן הַזִכָּרוֹן שֶׁבֵּרֵר עַל יְדֵי עֲבוֹדָתוֹ שֶׁבִּשְׁבִיל זֶה סִכֵּן נַפְשׁוֹ בִּבְחִינַת עַבְדוּת וְשִׁכְחָה שֶׁהִיא בְּחִינַת מִיתָה, כְּדֵי לְבָרֵר זֶה הַשְּכַר פְּעֻלָּה שֶׁהוּא שְׁלֵמוּת נַפְשׁוֹ בְּחִינַת תִּקּוּן הַזִכָּרוֹן בִּבְחִינַת וְאֵלָיו הוּא נוֹשֵֹא אֶת נַפְשׁוֹ וְעַל כֵּן צְרִיכִין לִתֵּן לוֹ זֶה הַשָּכָר מִיָּד בְּיוֹמוֹ דַּיְקָא כִּי צְרִיכִין לִשְׁמֹר אֶת הַזִכָּרוֹן בְּכָל יוֹם דַּיְקָא וְלֹא לִדְחוֹת מִיּוֹם לַחֲבֵרוֹ חַס וְשָׁלוֹם כִּי זֶה עִקַּר שְׁלֵמוּת הַזִכָּרוֹן לְהַזְכִּיר עַצְמוֹ בְּכָל יוֹם דַּיְקָא בְּעָלְמָא דְּאָתֵי כִּי בְּכָל יוֹם יֵשׁ בּוֹ רְמָזִים מְיֻחָדִים לְהִתְקָרֵב לַעֲבוֹדָתוֹ הַמְלֻבָּשִׁים בְּשִׁנּוּי מַחֲשָׁבָה דִּבּוּר וּמַעֲשֶֹה שֶׁבְּאוֹתוֹ הַיּוֹם דַּיְקָא כַּנַּ"ל:

## אות י

וְעַל כֵּן צְרִיכִין לִזְכֹּר יְצִיאַת מִצְרַיִם בִּקְרִיאַת שְׁמַע בְּכָל יוֹם דַּיְקָא כִּי יְצִיאַת מִצְרַיִם שֶׁגָּאַל אוֹתוֹ ה' יִתְבָּרַךְ מִיַּד פַּרְעֹה וַעֲבָדָיו שֶׁהֵם בְּחִינַת כֹּחַ הַמְדַמֶּה, זֶהוּ בְּחִינַת שְׁמִירַת הַזִכָּרוֹן כַּנַּ"ל. וְעַל כֵּן צְרִיכִין לִזְכֹּר זֹאת בְּכָל יוֹם דַּיְקָא, כִּי שְׁמִירַת הַזִכָּרוֹן צָרִיךְ לִהְיוֹת בְּכָל יוֹם דַּיְקָא כַּנַּ"ל:

## אות יא

כְּמוֹ שֶׁאִיתָא בַּזֹּהַר הַקָּדוֹשׁ כִּי יוֹסֵף שֶׁהוּא בְּחִינַת שְׁמִירַת הַזִכָּרוֹן הוּא מַכְנִיעַ בְּחִינַת פַּרְעֹה שֶׁהוּא בְּחִינַת שִׁכְחָה וּמְבָרֵר הַזִכָּרוֹן מִן הַשִּׁכְחָה וְעַל כֵּן הוּא יוֹדֵעַ לִפְתֹּר הַחֲלוֹם שֶׁל הַפָּרוֹת וּלְהַכְנִיעַ וּלְבַטֵּל הַשִּׁכְחָה הַנֶּאֱחָז בָּהֶם שֶׁהֵם שֶׁבַע שְׁנֵי הָרָעָב, כִּי הוּא מַגְבִּיר הַשָּבָע עַל הָרָעָב וּמַמְשִׁיךְ כָּל הַהַשְׁפָּעוֹת וְהַבְּרָכוֹת עַל יְדֵי שְׁמִירַת הַזִכָּרוֹן שֶׁמִּשָּׁם כָּל הַבְּרָכוֹת כַּנַּ"ל. וְזֶהוּ בְּחִינַת פָּרָה אֲדֻמָּה כִּי עַל יְדֵי שֶׁשּוֹרְפִין אֶת הַפָּרָה אֲדֻמָּה שֶׁהִיא תֹּקֶף הַדִּין שֶׁשָּׁם עִקַּר אֲחִיזַת הַשִּׁכְחָה כַּנַּ"ל, עַל יְדֵי זֶה מַכְנִיעִין וּמַמְתִּיקִין אֶת בְּחִינַת הַשִּׁכְחָה שֶׁהִיא סִטְרָא דְּמוֹתָא בְּשָׁרְשׁוֹ וְעַל כֵּן אֵפֶר הַפָּרָה דַּיְקָא מְטַהֵר מְטֻּמְאַת מֵת שֶׁהִיא בְּחִינַת שִׁכְחָה שֶׁנִּמְשָׁךְ מֵחֵטְא אָדָם הָרִאשׁוֹן מֵחֵטְא הָעֵגֶל. וּפָרָה מְטַהֶרֶת מִזֶּה כִּי הַדִּין נִמְתָּק בְּשָׁרְשׁוֹ כַּנַּ"ל וְאָז נִתְבַּטְּלִין שֶׁבַע פָּרוֹת הָרָעוֹת שֶׁהֵם בְּחִינַת הָרַע בְּחִינַת שִׁכְחָה. וְעַל כֵּן פָּרָה כָּל מַעֲשֶֹיהָ שִׁבְעָה שִׁבְעָה כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כְּדֵי לְהַכְנִיעַ וּלְבַטֵּל בְּחִינַת זִכָּרוֹן שֶׁעַל יְדֵי זֶה נִתְיַחֵד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל בִּבְחִינַת נוֹטֶה שָׁמַיִם כַּיְרִיעָה כַּנַּ"ל:

## אות ז

וְזֶהוּ בְּחִינַת לֹא תָלִין פְּעֻלַּת שָֹכִיר אִתְּךָ עַד בֹּקֶר כִּי זֶה הַשָּכִיר שֶׁנִּשְֹכָּר אֵצֶל חֲבֵרוֹ לַעֲשֹוֹת מְלַאכְתּוֹ הוּא מַפִּיל עַצְמוֹ לִבְחִינַת עַבְדוּת שֶׁהִוא בְּחִינַת שִׁכְחָה כַּנַּ"ל וְזֶהוּ בְּחִינַת גָּלוּת מִצְרַיִם בְּחִינַת עֲבָדִים הָיִינוּ לְפַרְעֹה בְּמִצְרָיִם, כִּי פַּרְעֹה הוּא בְּחִינַת כֹּחַ הַמְדַמֶּה בְּחִינַת פְּגַם הַזִכָּרוֹן בְּחִינַת שִׁכְחָה כַּמְבֹאָר שָׁם וְעַל כֵּן רָצָה לְהַגְבִּיר בְּחִינַת הָעַבְדוּת עַל יִשְֹרָאֵל כְּדֵי לְהַפִּיל עֲלֵיהֶם שִׁכְחָה כְּדֵי שֶׁלֹּא יוּכְלוּ לְדַבֵּק מַחֲשַׁבְתָּם בְּעָלְמָא דְּאָתֵי וּכְמוֹ שֶׁכָּתוּב כִּי נִרְפִּים הֵם עַל כֵּן הֵם צוֹעֲקִים לֵאמֹר נֵלְכָה נִזְבְּחָה לֵאלֹקֵינוּ תִּכְבַּד הָעֲבוֹדָה כוּ' וְיַעֲשֹוּ בָּהּ וְאַל יִשְׁעוּ וְכוּ'. נִמְצָא זֶה שֶׁנִּשְֹכַּר אֵצֶל חֲבֵרוֹ בִּשְׁבִיל פַּרְנָסָתוֹ הוּא מוֹרִיד עַצְמוֹ לִבְחִינַת עַבְדוּת שֶׁהִוא בְּחִינַת שִׁכְחָה אֲבָל כַּוָּנָתוֹ בִּשְׁבִיל פַּרְנָסָתוֹ כִּי בְּהֶכְרֵחַ לְבָרֵר הַפַּרְנָסָה עַל יְדֵי הַל"ט [הַשְּׁלֹשִׁים וְתֵשַׁע] מְלָאכוֹת כִּי מֵחֲמַת חֵטְא אָדָם הָרִאשׁוֹן נִפְגַּם הַזִכָּרוֹן וְנָפַל לִבְחִינַת שִׁכְחָה וְעַל יְדֵי זֶה נִתְקַלֵּל בְּעִצָּבוֹן תֹּאכֲלֶנָּה בְּזֵעַת אַפֶּךָ תֹּאכַל כַּנַּ"ל. וְעַל כֵּן נִקְרָא זִכָּרוֹן בְּחִינַת זָכוֹר בְּחִינַת יוֹסֵף הַצַּדִּיק שֶׁהוּא בְּחִינַת נָהָר הַיּוֹצֵא מֵעֵדֶן שֶׁהוּא צִנּוֹר שֶׁל כָּל הַהַשְׁפָּעוֹת כַּיָּדוּעַ. וְעַל כֵּן בֶּאֱמֶת הַתּוֹרָה דְּבֵקִים בְּעָלְמָא דְּאָתֵי, עַל כֵּן פַּרְנָסָתוֹ בְּנָקֵל יוֹתֵר כִּי מַעֲבִירִין מִמֶּנּוּ עֹל וְכוּ' כִּי כָּל הַשֶּׁפַע וְהַפַּרְנָסָה מִבְּחִינַת שְׁמִירַת הַזִכָּרוֹן הַנַּ"ל:

וְזֶה שֶׁאִיתָא בַּזֹּהַר הַקָּדוֹשׁ שָׁטוּ הָעָם וְלָקְטוּ בִּשְׁטוּתָא כִּי עִקַּר יְגִיעַת הַפַּרְנָסָה הוּא מֵחֲמַת שְׁטוּת וְחֶסְרוֹן הַדַּעַת מֵחֲמַת שֶׁאֵינוֹ מַגְדִּיל דַּעְתּוֹ לְדַבֵּק מַחֲשַׁבְתּוֹ בְּעָלְמָא דְּאָתֵי שֶׁזֶּהוּ עִקַּר הַגְדָּלַת הַדַּעַת כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל, וְעַל יְדֵי זֶה פַּרְנָסָתוֹ בִּיגִיעָה גְּדוֹלָה בְּחִינַת שָׁטוּ הָעָם וְלָקְטוּ בִּשְׁטוּתָא כַּנַּ"ל. וְכָל הַמָּמוֹן וְהַפַּרְנָסָה שֶׁהָאָדָם מַרְוִיחַ עַל יְדֵי יְגִיעַ כַּפּוֹ זֶה הַמָּמוֹן הוּא בְּחִינַת בֵּרוּר הַנִּיצוֹצוֹת שֶׁבֵּרֵר מִן הַסִּטְרָא אָחֳרָא שֶׁהוּא בְּחִינַת בֵּרוּר הַזִכָּרוֹן וְכַמְבֹאָר שָׁם בְּהַתּוֹרָה הַנַּ"ל עַל פָּסוּק נוֹטֶה שָׁמַיִם כַּיְרִיעָה שֶׁהוּא בְּחִינַת יִחוּד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל שֶׁנַּעֲשֶֹה עַל יְדֵי הַמַּשָּא וּמַתָּן שֶׁעַל יְדֵי כָּל הַחֲפָצִים שֶׁהָאָדָם זוֹכֶה לָהֶם בִּקְדֻשָּׁה עַל יְדֵי זֶה עוֹלִין הַנִּיצוֹצוֹת וְנִכְלָלִין בְּקוֹמַת אָדָם וְנַעֲשֶֹה מֵהַיִחוּד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל הַנַּ"ל שֶׁזֶּהוּ בְּחִינַת גְּמַר תִּקּוּן שְׁלֵמוּת הַזִכָּרוֹן. וּמֵחֲמַת שֶׁעִקַּר הַמָּמוֹן וְהַפַּרְנָסָה הוּא עַל יְדֵי בֵּרוּר הַזִכָּרוֹן עַל כֵּן כָּל מִי שֶׁרָחוֹק מִזֶּה הַזִכָּרוֹן דְּהַיְנוּ שֶׁאֵינוֹ מְדַבֵּק ## אות ח

וְעַל כֵּן הִזְהִירָה הַתּוֹרָה לִתֵּן הַשְּכִירוּת לְהַשָּכִיר בְּמוֹעֲדוֹ וּבִזְמַנּוֹ וְשֶׁלֹּא לְאַחֵר שְֹכָרוֹ כְּמוֹ שֶׁכָּתוּב (דְּבָרִים כד) בְּיוֹמוֹ תִּתֵּן שְֹכָרוֹ וְכוּ' כִּי עָנִי הוּא וְאֵלָיו הוּא נוֹשֵֹא אֶת נַפְשׁוֹ כִּי תֵּכֶף כְּשֶׁמַּשְׁלִים פְּעֻלָּתוֹ וְיוֹצֵא מִן הָעַבְדוּת שֶׁהִוא בְּחִינַת שִׁכְחָה צְרִיכִין לִתֵּן לוֹ תֵּכֶף שְֹכָרוֹ שֶׁפָּעַל שֶׁהוּא בְּחִינַת בֵּרוּר הַנִּיצוֹצוֹת בְּחִינַת תִּקּוּן נַפְשׁוֹ שֶׁהוּא בְּחִינַת תִּקּוּן הַזִכָּרוֹן שֶׁעַל יְדֵי זֶה מְתַקֵּן מִיָּד פְּגַם הַשִּׁכְחָה שֶׁיָּרַד לְשָׁם עַל יְדֵי בְּחִינַת הָעַבְדוּת כַּנַּ"ל. וְעַל כֵּן הִזְהִירָה הַתּוֹרָה מְאֹד לָתֵת שְֹכָרוֹ בְּיוֹמוֹ דַּיְקָא שֶׁלֹּא לְאַחֵר מִיּוֹם לַחֲבֵרוֹ כִּי תִּקּוּן הַזִכָּרוֹן צְרִיכִין לִזָּהֵר מְאֹד בְּכָל יוֹם וָיוֹם, לִשְׁמֹר אֶת הַזִכָּרוֹן בְּכָל יוֹם דַּיְקָא דְּהַיְנוּ לְהַזְכִּיר אֶת עַצְמוֹ בְּכָל יוֹם וָיוֹם בְּעָלְמָא דְּאָתֵי. כִּי עִקַּר הַשִּׁכְחָה נִמְשָׁךְ עַל יְדֵי רִבּוּי הַזְּמַן כַּנִּרְאֶה בְּחוּשׁ שֶׁעַל יְדֵי אֲרִיכַת הַזְּמַן נִשְׁכַּח הַדָּבָר כְּמוֹ שֶׁכָּתוּב קֹהֶלֶת בִּרְבוֹת הַיָּמִים הַכֹּל נִשְׁכָּח כִּי עִקַּר בְּחִינַת הַשִּׁכְחָה נֶאֱחָז מִבְּחִינַת הַשִּׁנּוּיִים שֶׁיֵּשׁ בָּעוֹלָם בְּכָל יוֹם וּבְכָל עֵת כִּי ה' יִתְבָּרַךְ עוֹשֶֹה נִפְלָאוֹת בְּכָל עֵת וּבְטוּבוֹ מְחַדֵּשׁ בְּכָל יוֹם תָּמִיד מַעֲשֵֹה בְּרֵאשִׁית וּבְכָל יוֹם עוֹלָה הַשֶּׁמֶשׁ בַּבֹּקֶר וְשׁוֹקֵעַ בָּעֶרֶב וְאֵין יוֹם דּוֹמֶה וּבְזֶה הַיּוֹם עָנָן וְגֶשֶׁם וְכַיּוֹצֵא בָּזֶה שִׁנּוּיִים רַבִּים בְּעִנְיַן הָאֲוִיר וְהָרוּחַ וּשְׁאָר הַהִתְהַוּוּת שֶׁבִּזְמַן שֶׁיֵּשׁ בָּהֶם שִׁנּוּיִים רַבִּים בְּכָל יוֹם וָיוֹם כִּי ה' יִתְבָּרַךְ מְשַׁנֶּה עִתִּים בִּתְבוּנָתוֹ. וְכֵן בִּמְאֹרְעוֹת הָאָדָם וּתְנוּעוֹתָיו וּמַחְשְׁבוֹתָיו וְהַנְהָגוֹתָיו יֵשׁ בָּהֶם שִׁנּוּיִים רַבִּים בְּכָל יוֹם וָיוֹם מִי שֶׁמִּסְתַּכֵּל עֲלֵיהֶם. וּמֵרִבּוּי הַשִּׁנּוּיִים הָאֵלֶּה שֶׁבְּכָל יוֹם נִמְשָׁךְ הַשִּׁכְחָה כִּי הֵם מַטְרִידִין אֶת הָאָדָם מִלִּזְכֹּר בְּעָלְמָא דְּאָתֵי וְכַנִּרְאֶה גַּם בְּחוּשׁ קְצָת שֶׁבְּכָל יוֹם וּבְנוֹתָיו הַרְבֵּה שֶׁיֵּשׁ לוֹ צַעַר גִּדּוּל בָּנִים וְזֶה הוּא חֲשֹוּכֵי בָּנִים וְזֶה מְסַפֵּר מִמְּאֹרְעוֹת הַמִּלְחָמוֹת שֶׁנִּשְׁמָע עַכְשָׁו בָּעוֹלָם וְכַיּוֹצֵא בָּזֶה שִׁנּוּיִים ## אות ט

## אות י

הַזֻּהֲמָא שֶׁל מִצְרַיִם וְעַל יְדֵי זֶה זָכוּ יִשְֹרָאֵל לָצֵאת מִשָּׁם. כִּי עִקַּר יְצִיאַת מִצְרַיִם הָיָה בְּכֹחַ יוֹסֵף הַצַּדִּיק כְּמוֹ שֶׁכָּתוּב וַיִּקַּח מֹשֶׁה אֶת עַצְמוֹת יוֹסֵף עִמּוֹ כְּמוֹ שֶׁאִיתָא בַּזֹּהַר הַקָּדוֹשׁ כִּי יוֹסֵף שֶׁהוּא בְּחִינַת שְׁמִירַת הַזִכָּרוֹן הוּא מַכְנִיעַ בְּחִינַת פַּרְעֹה שֶׁהוּא בְּחִינַת שִׁכְחָה וּמְבָרֵר הַזִכָּרוֹן מִן הַשִּׁכְחָה וְעַל כֵּן הוּא יוֹדֵעַ לִפְתֹּר הַחֲלוֹם שֶׁל הַפָּרוֹת וּלְהַכְנִיעַ וּלְבַטֵּל הַשִּׁכְחָה הַנֶּאֱחָז בָּהֶם שֶׁהֵם שֶׁבַע שְׁנֵי הָרָעָב, כִּי הוּא מַגְבִּיר הַשָּבָע עַל הָרָעָב וּמַמְשִׁיךְ כָּל הַהַשְׁפָּעוֹת וְהַבְּרָכוֹת עַל יְדֵי שְׁמִירַת הַזִכָּרוֹן שֶׁמִּשָּׁם כָּל הַבְּרָכוֹת כַּנַּ"ל. וְזֶהוּ בְּחִינַת פָּרָה אֲדֻמָּה כִּי עַל יְדֵי שֶׁשּוֹרְפִין אֶת הַפָּרָה אֲדֻמָּה שֶׁהִיא תֹּקֶף הַדִּין שֶׁשָּׁם עִקַּר אֲחִיזַת הַשִּׁכְחָה כַּנַּ"ל, עַל יְדֵי זֶה מַכְנִיעִין וּמַמְתִּיקִין אֶת בְּחִינַת הַשִּׁכְחָה שֶׁהִיא סִטְרָא דְּמוֹתָא בְּשָׁרְשׁוֹ וְעַל כֵּן אֵפֶר הַפָּרָה דַּיְקָא מְטַהֵר מְטֻּמְאַת מֵת שֶׁהִיא בְּחִינַת שִׁכְחָה שֶׁנִּמְשָׁךְ מֵחֵטְא אָדָם הָרִאשׁוֹן מֵחֵטְא הָעֵגֶל. וּפָרָה מְטַהֶרֶת מִזֶּה כִּי הַדִּין נִמְתָּק בְּשָׁרְשׁוֹ כַּנַּ"ל וְאָז נִתְבַּטְּלִין שֶׁבַע פָּרוֹת הָרָעוֹת שֶׁהֵם בְּחִינַת הָרַע בְּחִינַת שִׁכְחָה. וְעַל כֵּן פָּרָה כָּל מַעֲשֶֹיהָ שִׁבְעָה שִׁבְעָה כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה כְּדֵי לְהַכְנִיעַ וּלְבַטֵּל שֶׁבַע פָּרוֹת הָרָעוֹת וּלְהַגְבִּיר שֶׁבַע פָּרוֹת הַטּוֹבוֹת שֶׁהֵם בְּחִינַת יוֹסֵף הַצַּדִּיק בְּחִינַת זִכָּרוֹן בְּחִינַת בֵּן פֹּרָת יוֹסֵף בְּחִינַת אֲנִי קֵל שַׁדַּי פְּרֵה וּרְבֵה שֶׁזֶּהוּ בְּחִינַת זִכָּרוֹן שֶׁעַל יְדֵי זֶה נִתְיַחֵד צַדִּיק וּכְנֶסֶת יִשְֹרָאֵל בִּבְחִינַת נוֹטֶה שָׁמַיִם כַּיְרִיעָה כַּנַּ"ל:

## אות יב

וְעַל כֵּן יוֹסֵף כְּשֶׁפֵּרַשׁ מֵאָבִיו הָיָה עוֹסֵק בְּפָרָשַׁת עֶגְלָה עֲרוּפָה כְּמוֹ שֶׁכָּתוּב וַיַּרְא אֶת הָעֲגָלוֹת כִּי הָרוֹצֵחַ הִמְשִׁיךְ סִטְרָא דְּמוֹתָא לָעוֹלָם בְּחִינַת שִׁכְחָה עַל כֵּן צְרִיכִין לְהָבִיא עֶגְלָה עֲרוּפָה וְלָקֹץ אוֹתָהּ מִן הָעֹרֶף שֶׁהוּא אוֹתִיּוֹת פַּרְעֹה לְהַכְנִיעַ הָעֹרֶף שֶׁהוּא הֵפֶךְ הַדַּעַת שֶׁהוּא בְּחִינַת שִׁכְחָה שֶׁהִיא פְּגַם וְהֵפֶךְ הַדַּעַת וְעַל כֵּן עוֹרְפִין עֶגְלָה דַּיְקָא. הַיְנוּ עַל פִּי הַנַּ"ל כִּי עִקַּר הַשִּׁכְחָה נֶאֱחֶזֶת בִּבְחִינַת בַּהֲמִיּוּת וּבִבְחִינַת שׁוֹר בְּיוֹתֵר שֶׁהוּא רֹאשׁ לַבְּהֵמוֹת, וְהָעֶגְלָה אֲשֶׁר עֲדַיִן לֹא לֻמְּדָה כְּלָל בְּשׁוּם עֲבוֹדָה שָׁם כֹּחַ הַבַּהֲמִיּוּת עוֹד בְּתֹקֶף יוֹתֵר וְיוֹתֵר כִּי זֶה יָדוּעַ שֶׁעַל יְדֵי כָּל הָעֲבוֹדוֹת שֶׁבְּנֵי אָדָם עוֹבְדִין עִם הַבְּהֵמוֹת לְצָרְכָּם עַל יְדֵי זֶה מַכְנִיעִין כֹּחַ הַבַּהֲמִיּוּת וְעַל כֵּן הֵם נַעֲשִֹין רְגִילִין עִם בְּנֵי הָאָדָם וִיכוֹלִים לְלַמְּדָם עֲבוֹדַת הָאָדָם שֶׁכָּל זֶה הוּא בְּחִינַת הַכְנָעַת הַבַּהֲמִיּוּת קְצָת. אֲבָל הָעֵגֶל אֲשֶׁר עֲדַיִן לֹא לֻמַּד כְּלָל, שָׁם הַבַּהֲמִיּוּת בְּיוֹתֵר וְיוֹתֵר וְזֶה שֶׁהוֹכִיחַ אֶת יִשְֹרָאֵל, בָּעֵגֶל לֹא לֻמַּד וְעַל כֵּן עָשֹוּ הָעֵרֶב רַב עֵגֶל דַּיְקָא כְּדֵי לְהַגְבִּיר הַבַּהֲמִיּוּת וְהַשִּׁכְחָה בְּיוֹתֵר וְעַל כֵּן צְרִיכִין לַעֲרֹף עֶגְלָה דַּיְקָא כְּדֵי לְהַכְנִיעַ וּלְשַׁבֵּר תֹּקֶף הַשִּׁכְחָה שֶׁהִיא סִטְרָא דְּמוֹתָא שֶׁנִּמְשַׁךְ עַל יְדֵי הָרוֹצֵחַ הַזֶּה שֶׁהָרַג אֶת זֶה הֶחָלָל הַנִּמְצָא. וְעַל כֵּן יוֹסֵף כְּשֶׁפֵּרַשׁ מֵאָבִיו הִנִּיחוֹ בְּפָרָשַׁת עֶגְלָה עֲרוּפָה וְאַחַר כָּךְ זָכָה לִגְדֻלָּה עַל יְדֵי הַחֲלוֹם שֶׁל שֶׁבַע הַפָּרוֹת שֶׁהֵם סוֹד פָּרָה אֲדֻמָּה כַּנַּ"ל, זֶה בְּחִינַת תָּבוֹא אִמּוֹ וּתְקַנַּח עַל בְּנָהּ כִּי פָּרָה מְכַפֵּר עַל חֵטְא הָעֵגֶל כִּי עַל יְדֵי כָּל הַמִּצְוֹת שֶׁל פָּרָה אֲדֻמָּה מַכְנִיעִין וּמְבַטְּלִין הַשִּׁכְחָה כַּנַּ"ל וְנִתְבַּטְּלָה הַמִּיתָה שֶׁזֶּהוּ בְּחִינַת עֲרִיפַת הָעֶגְלָה שֶׁכָּל זֶה הוּא בְּחִינַת:

הַכְנָעַת וּבִטּוּל פַּרְעֹה שֶׁרָצָה לְהַגְבִּיר הַשִּׁכְחָה עַל יִשְֹרָאֵל חַס וְשָׁלוֹם כַּנַּ"ל שֶׁזֶּהוּ כָּל גָּלוּת מִצְרַיִם וּכְלַל כָּל הַגָּלֻיּוֹת. כִּי עִקַּר הַגָּלוּת הוּא גָּלוּת הַנֶּפֶשׁ דְּהַיְנוּ מַה שֶּׁרוֹצִין לְהִתְגַּבֵּר עַל יְדֵי הַגָּלוּת לְהַרְחִיק אֶת יִשְֹרָאֵל חַס וְשָׁלוֹם מֵעָלְמָא דְּאָתֵי שֶׁזֶּהוּ עִקַּר הַגָּלוּת וְהַצָּרָה הַגְּדוֹלָה מִכָּל הַצָּרוֹת כַּמְבֹאָר בְּדִבְרֵי רַבֵּנוּ זִכְרוֹנוֹ לִבְרָכָה בְּמָקוֹם אַחֵר שֶׁזֶּהוּ עִקַּר צָרָתָן שֶׁל יִשְֹרָאֵל. וְהַתִּקּוּן עַל יְדֵי יוֹסֵף שֶׁעַל יָדוֹ נִמְשָׁךְ הַזִכָּרוֹן כַּנַּ"ל:

## אות יג

וְעַל כֵּן קוֹרִין פָּרָשַׁת פָּרָה אַחַר פּוּרִים קֹדֶם פֶּסַח כִּי כָּל עִנְיַן פּוּרִים וּפֶסַח הוּא לְהַכְנִיעַ עֲמָלֵק וּפַרְעֹה שֶׁהֵם בְּחִינַת פְּגַם הַזִכָּרוֹן כִּי פַּרְעֹה הוּא בְּחִינַת כֹּחַ הַמְדַמֶּה שֶׁפּוֹגֵם בַּזִּכָּרוֹן וַעֲמָלֵק גָּרוּעַ יוֹתֵר כִּי הוּא בְּחִינַת הָרַע עַיִן בְּעַצְמוֹ שֶׁהוּא עִקַּר פְּגַם הַזִכָּרוֹן וְכַמְבֹאָר מִזֶּה בְּמָקוֹם אַחֵר (בְּהִלְכוֹת תַּלְמוּד תּוֹרָה עַיֵּן שָׁם). וְעַל כֵּן מִלְחָמָה לַה' בַּעֲמָלֵק מִדּוֹר דּוֹר כִּי אֵינוֹ נִכְנָע לְגַמְרֵי כִּי אִם בְּדוֹרוֹ שֶׁל מָשִׁיחַ שֶׁיָּבוֹא בִּמְהֵרָה בְּיָמֵינוּ כִּי הָרַע עַיִן אִי אֶפְשָׁר לְהַכְנִיעַ וּלְבַטֵּל כִּי אִם עַל יְדֵי מַלְכוּת מָשִׁיחַ כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל. וְעַל כֵּן צְרִיכִין לִזְכֹּר מַעֲשֵֹה עֲמָלֵק וְלִבְלִי לִשְׁכֹּחַ כִּי עִקַּר הַכְנָעַת הָמָן עֲמָלֵק הוּא עַל יְדֵי הַזִכָּרוֹן כַּנַּ"ל וְזֶהוּ בְּחִינַת וְהַיָּמִים הָאֵלֶּה נִזְכָּרִים וְנַעֲשִֹים וְכוּ' (אֶסְתֵּר ט) כַּנַּ"ל. וְעַל כֵּן הָמָן הִפִּיל פּוּר שֶׁהוּא בְּחִינַת שִׁכְחָה כַּמּוּבָא וּבִקֵּשׁ לְהַשְׁמִיד וְכוּ' לְהַגְבִּיר סִטְרָא דְּמוֹתָא חַס וְשָׁלוֹם כִּי הוּא עִקַּר זֻהֲמַת הַנָּחָשׁ בְּחִינַת הֲמִן הָעֵץ כְּמוֹ שֶׁאָמְרוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה. וְה' יִתְבָּרַךְ הֵפֵר מַחֲשַׁבְתּוֹ וְהֵשִׁיב לוֹ גְּמוּלוֹ בְּרֹאשׁוֹ וְנִתְהַפֵּךְ מֵהֵפֶךְ אֶל הֵפֶךְ כִּי לֹא דַּי שֶׁלֹּא פָּעַל מַחֲשַׁבְתּוֹ לְהַשְׁכִּיחַ אֶת הַתּוֹרָה וְעָלְמָא דְּאָתֵי חַס וְשָׁלוֹם אַדְּרַבָּא זָכוּ יִשְֹרָאֵל אָז לְהַגְבִּיר אֶת הַזִכָּרוֹן דִּקְדֻשָּׁה יוֹתֵר וְיוֹתֵר בִּבְחִינַת קִיְּמוּ וְקִבְּלוּ שֶׁקִּבְּלוּ אָז אֶת הַתּוֹרָה מֵחָדָשׁ שֶׁהוּא בְּחִינַת זִכָּרוֹן דִּקְדֻשָּׁה וְהַיָּמִים הָאֵלֶּה נִזְכָּרִים לְעוֹלָם כְּמוֹ שֶׁכָּתוּב (שָׁם) וְזִכְרָם לֹא יָסוּף וְדָרְשׁוּ רַבּוֹתֵינוּ זִכְרוֹנָם לִבְרָכָה שֶׁיְּמֵי הַפּוּרִים לֹא יִבָּטְּלוּ לְעוֹלָם כִּי נִתְגַּבֵּר אָז הַזִכָּרוֹן דִּקְדֻשָּׁה יוֹתֵר וְיוֹתֵר כַּנַּ"ל. וְעַל כֵּן אַחַר פּוּרִים קוֹרִין פָּרָשַׁת פָּרָה שֶׁהִיא מְטַהֶרֶת מְטֻּמְאַת מֵת שֶׁהִיא בְּחִינַת שִׁכְחָה שֶׁעַל יְדֵי זֶה יְכוֹלִין לַעֲשֹוֹת הַקָּרְבַּן פֶּסַח וְנִגְאָלִין מִפַּרְעֹה וּמִצְרַיִם שֶׁהֵם גַּם כֵּן בְּחִינַת פְּגַם הַזִכָּרוֹן כַּנַּ"ל:

כִּי הָמָן הָרָשָׁע רָצָה לְהִתְגַּבֵּר בִּשְׁעַת שֵׁנָה בְּסוֹף הַגָּלוּת שֶׁהוּא בְּחִינַת שֵׁנָה וְהוּא סָבַר שֶׁבִּשְׁעַת שֵׁנָה הִסְתַּלְּקוּת הַדַּעַת אָז יֵשׁ לוֹ כֹּחַ לְהַגְבִּיר הַשִּׁכְחָה חַס וְשָׁלוֹם וּמֵחֲמַת שֶׁשֵּׁנָה הִוא אֶחָד מִשִּׁשִּׁים בְּמִיתָה כַּנַּ"ל. אֲבָל הוּא לֹא יָדַע שֶׁיִּשְֹרָאֵל קְדוֹשִׁים אַדְּרַבָּא בִּשְׁעַת שֵׁנָה הֵם דְּבֵקִים בְּעָלְמָא דְּאָתֵי כִּי הֵם מְקַדְּשִׁים עַצְמָן בִּשְׁעַת שֵׁנָה גַּם כֵּן כִּי קוֹרִין קְרִיאַת שְׁמַע קֹדֶם וּמַפְקִידִים רוּחָם בְּיָדוֹ יִתְבָּרַךְ וְעַל כֵּן דַּיְקָא אָז בִּשְׁעַת שֵׁנָה בְּסוֹף הַגָּלוּת הָיָה הֶאָרַת מָרְדְּכַי שֶׁהוּא הֶאָרָה גְּדוֹלָה מְאֹד מְאֹד שֶׁמֵּאִיר בִּשְׁעַת שֵׁנָה דַּיְקָא וְעַל יְדֵי הַהֶאָרָה הַגְּדוֹלָה הַזֹּאת דַּיְקָא הִכְנִיעוּ אוֹתוֹ כִּי יִשְֹרָאֵל דְּבֵקִים בִּשְׁעַת שֵׁנָה דַּיְקָא בְּעָלְמָא דְּאָתֵי כַּמְבֹאָר בְּהַתּוֹרָה הַנַּ"ל:
